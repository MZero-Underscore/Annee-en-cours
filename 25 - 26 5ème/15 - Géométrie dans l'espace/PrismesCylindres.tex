\section{Vocabulaire et définitions}
\subsection{Le prisme droit}

    \begin{definition}
        Un \textbf{prisme droit} est un solide géométrique constitué :
        \begin{itemize}
            \item de deux faces parallèles et superposables qui sont des polygones : ce sont les \textbf{bases} ;
            \item de faces latérales qui sont toutes des \textbf{rectangles}.
        \end{itemize}
    \end{definition}
    
    \noindent
    \begin{minipage}{0.55\textwidth}
        \begin{exemple}
        En vous aidant de la figure ci-contre (un prisme droit à base triangulaire), complétez les phrases :
        \begin{itemize}
            \item Les deux bases sont les triangles : \\
            \points[5] et \points[5]
            \item Les faces latérales sont des \points[5] \\
            comme par exemple : \points[8]
            \item La hauteur est le segment : \points[6]
            \item Les arêtes latérales sont toutes de même \points[6]
        \end{itemize}
        \end{exemple}
    \end{minipage}
    \hfill

    \begin{remarque}
        La distance entre les deux bases s'appelle la \textbf{hauteur} du prisme droit. Le nombre de faces latérales est toujours égal au nombre de côtés du polygone de base.
    \end{remarque}

\subsection{Le cylindre de révolution}
    \begin{definition}
        Un \textbf{cylindre de révolution} est un solide obtenu en faisant tourner un rectangle autour de l'un de ses côtés. Il est constitué :
        \begin{itemize}
            \item de deux faces parallèles et superposables qui sont des \textbf{disques} : ce sont les \textbf{bases} ;
            \item d'une surface latérale \textbf{courbe} qui, développée, est un rectangle.
        \end{itemize}
    \end{definition}

    \begin{minipage}{0.55\textwidth}

            \item Les deux bases sont des disques de centres respectifs : \points[4] et \points[4]
            \item La hauteur est le segment : \points[6]
            \item Le segment $[OA]$ est un \points[6] de la base.
        \end{itemize}
        \end{exemple}

\section{Volumes}

    \begin{proprieteTitre}{Formule du volume}
        Pour calculer le volume $\mathcal{V}$ d'un prisme droit ou d'un cylindre de révolution, on utilise la formule générale suivante :
        \[ \mathcal{V} = \text{Aire de la base} \times \text{Hauteur} = \mathcal{B} \times h \]
        \begin{itemize}
            \item $\mathcal{B}$ est l'aire de la base du solide ($\text{cm}^2$, $\text{m}^2$, etc.)
            \item $h$ est la hauteur du solide ($\text{cm}$, $\text{m}$, etc.)
        \end{itemize}
    \end{proprieteTitre}

    \begin{exemple}
        Complétez les deux exemples d'application ci-dessous :
        
        \vspace{0.2cm}
        \textbf{1. Cas du prisme droit à base triangulaire} \\
        Calculer le volume d'un prisme droit de hauteur $h = 7\text{ cm}$ ayant pour base un triangle rectangle dont les côtés de l'angle droit mesurent $3\text{ cm}$ et $4\text{ cm}$.
        \begin{itemize}
            \item[$\bullet$] \textbf{Étape 1 :} Calcul de l'aire de la base $\mathcal{B}$ (triangle rectangle) : \\
            $\mathcal{B} = \frac{\points[6] \times \points[4]}{2} = \frac{\points[4]}{2} = \points[4]\text{ cm}^2$
            \item[$\bullet$] \textbf{Étape 2 :} Calcul du volume $\mathcal{V}$ du prisme : \\
            $\mathcal{V} = \points[6] \times \points[4] = \points[6]\text{ cm}^3$
        \end{itemize}
        
        \vspace{0.4cm}
    
        \textbf{2. Cas du cylindre de révolution} \\
        Calculer le volume d'un cylindre de révolution de hauteur $h = 10\text{ cm}$ ayant pour base un disque de rayon $r = 3\text{ cm}$.
        \begin{itemize}
            \item[$\bullet$] \textbf{Étape 1 :} Calcul de l'aire de la base $\mathcal{B}$ (disque) : \\
            $\mathcal{B} = \pi \times \points[4]^2 = \points[5]\pi\text{ cm}^2$
            \item[$\bullet$] \textbf{Étape 2 :} Calcul du volume $\mathcal{V}$ du cylindre : \\
            $\mathcal{V} = \points[5]\pi \times \points[4] = \points[6]\pi \approx \points[6]\text{ cm}^3$ (arrondi au dixième).
        \end{itemize}
    \end{exemple}

\section{Problème d'application}

    \begin{exerciceTitre}{La citerne d'eau de pluie}
        Pour arroser son potager, un jardinier installe une grande citerne de récupération d'eau de pluie en forme de cylindre de révolution. 
        Cette citerne possède un diamètre de base de $1,2\text{ m}$ et une hauteur de $1,5\text{ m}$.
        
        \begin{enumerate}
            \item Déterminer le rayon de la base de cette citerne.
            \item Calculer l'aire exacte de la base du cylindre, puis donner sa valeur arrondie au centième près de $\text{m}^2$.
            \item Calculer le volume de cette citerne en $\text{m}^3$. On arrondira le résultat au millième près.
            \item Sachant que $1\text{ m}^3 = 1000\text{ litres}$, quelle est la capacité maximale de cette citerne d'eau arrondie au litre près ?
        \end{enumerate}
    \end{exerciceTitre}
