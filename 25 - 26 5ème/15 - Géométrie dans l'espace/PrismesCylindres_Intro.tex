    \begin{exerciceTitre}{Partie 2 : Défi Patron n°1 (Le Prisme)}
        \begin{minipage}{0.65\textwidth}
            Ton objectif est de tracer le patron d'un prisme droit en vraie grandeur.
            \begin{itemize}
                \item Sa base est un \textbf{trapèze rectangle} dont les dimensions sont données sur le schéma ci-contre.
                \item La \textbf{hauteur} du prisme (la distance entre les deux bases) est de \textbf{$6\text{ cm}$}.

    \begin{exerciceTitre}{Partie 3 : Défi Patron n°2 (Le Cylindre)}
        Ton objectif est maintenant de tracer le patron d'un cylindre de révolution.
        \begin{itemize}
            \item Le \textbf{rayon} de sa base est de \textbf{$2\text{ cm}$}.
            \item Sa \textbf{hauteur} est de \textbf{$5\text{ cm}$}.
        \end{itemize}
        
        \textbf{Attention, il y a un piège !} La surface latérale est un grand rectangle. L'une de ses dimensions est bien $5\text{ cm}$ (la hauteur). Mais quelle est l'autre dimension ?
        
        \begin{enumerate}
            \item À quoi doit correspondre la longueur de ce rectangle pour qu'il s'enroule parfaitement autour du disque de base sans le dépasser et sans qu'il manque de papier ? \\
            \dotfill
            \item Calcule cette longueur (tu peux utiliser la touche $\pi$ de ta calculatrice et arrondir ton résultat au millimètre près) : \\
            \dotfill
            \item Trace le patron complet ci-dessous en vraie grandeur.
        \end{enumerate}
    \end{exerciceTitre}
