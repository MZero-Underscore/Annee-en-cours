\section{Priorité des opérations}
\subsection{cas général}

    \begin{convention}
        Dans une suite de calculs, on effectue :
        \begin{itemize}
            \item en premier les calculs entre parenthèses,
            \item puis les multiplications et les divisions dans le sens de lecture,
            \item on termine par les additions et les soustractions dans le sens de lecture.
        \end{itemize}
    \end{convention}
    
    \begin{exemple}
        Effectuer les calculs suivants.
        
        \(A = 16 + 4 \times 5\div2 \) \quad 
        \( B = 3\times[87-(3+2\times7)]\)
        puis $C=5+5\times1,7$
    \end{exemple}

\subsection{Cas particuliers}
    \begin{propriete}
        Dans une expression \textbf{comportant uniquement des additions}, on peut effectuer les calculs dans l'ordre que l'on veut.
    \end{propriete}

    \begin{exemple}
        Effectuer les calculs suivants dans l'ordre le plus judicieux.

        $A=12+ 99+ 188+ 1 \quad
        B=13,7+4,25+4,3+1,75$
    \end{exemple}

    \begin{propriete}
        Dans une expression \textbf{comportant uniquement des multiplications}, on peut effectuer les calculs dans l'ordre que l'on veut.
    \end{propriete}

    \begin{exemple}
        Effectuer les calculs suivants dans l'ordre le plus judicieux.

        $A=100\times 0,4 \times 0,03 \times 10 \quad 
        B=4\times5\times200\times25$
    \end{exemple}

\section{Vocabulaire}

    \begin{definition}
        Une expression est :
        \begin{itemize}
            \item  une \textbf{somme} si le \textbf{dernier calcul} à effectuer est une \textbf{addition}.
            \item  une \textbf{différence} si le \textbf{dernier calcul} à effectuer est une \textbf{soustraction}.
            \item  un \textbf{produit} si le \textbf{dernier calcul} à effectuer est une \textbf{multiplication}.
            \item  un \textbf{quotient} si le \textbf{dernier calcul} à effectuer est une \textbf{division}.
        \end{itemize}
    \end{definition}


\newpage


    \begin{exemple}
        \begin{enumerate}
            \item Associer chaque expression à la phrase qui la décrit :

            \begin{center}
            \begin{tabular}{l c p{1cm} c r}
                 \((8 - 3) : 2\) & $\bullet$ & & $\bullet$ & La différence de 8 et du quotient de 3 par 2. \\
                \(8 - 3 : 2\) & $\bullet$ & & $\bullet$ & La somme de 8 et du produit de 3 par 2. \\
                \(8 \times (3 + 2)\) & $\bullet$ & & $\bullet$ & Le quotient de la différence de 8 et de 3 par 2. \\
                \(8 + 3 \times 2\) & $\bullet$ & & $\bullet$ & Le produit de 8 par la somme de 3 et de 2. \\
            \end{tabular}
            \end{center}

            \item \begin{enumerate}
                \item Quel est le produit de \(-4\) par la somme de \(7\) et de \(-9\) ?
            
                \item Quelle est la somme du produit de \(-12\) par \(3\) et de \(-15\) ?
            
                \item Quel est le quotient de \(-24\) par le produit de \(6\) par \(-2\) ?
            
                \item Quelle est la différence entre le produit de \(5\) par \(-8\) et la somme de \(-12\) et de \(-7\) ?
            \end{enumerate}
        \end{enumerate}
    \end{exemple}


\section{Distributivité de la multiplication}

    \begin{remarque}
        La distributivité de la multiplication sur l'addition et la soustraction est une propriété fondamentale en mathématiques. Nous nous contenterons, pour le moment de l'utiliser pour effectuer certains calculs de façon \textit{astucieuse}.
    \end{remarque}


    \begin{exemple}
        Effectuer les calculs suivants sans utiliser la calculatrice et sans les poser :

        $A=70\times 99  \quad B=57\times 1001 \quad C=98 \times 55$ et $D=77\times202$
    \end{exemple}
