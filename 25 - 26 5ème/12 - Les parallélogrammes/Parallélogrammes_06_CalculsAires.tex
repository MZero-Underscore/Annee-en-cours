    \begin{exercice}[Retrouver une dimension manquante]
        \begin{enumerate}
            \item L'aire d'un parallélogramme est de $45\text{ cm}^2$. Sa base mesure $9\text{ cm}$. Quelle est la longueur de sa hauteur correspondante ?
            \item L'aire d'un triangle est de $24\text{ cm}^2$. Sa hauteur mesure $6\text{ cm}$. Quelle est la longueur de la base correspondante ?
        \end{enumerate}
    \end{exercice}
    
    
    \begin{exercice}[Problème : Le champ de l'agriculteur]
        Un agriculteur possède un champ en forme de parallélogramme. Un des côtés du champ mesure $120\text{ m}$. La distance perpendiculaire entre ce côté et le côté opposé est de $85\text{ m}$.
        \begin{enumerate}
            \item Fais un croquis à main levée du champ en y plaçant les dimensions.
            \item Calcule l'aire de ce champ en mètres carrés ($\text{m}^2$).
            \item Sachant qu'un hectare ($1\text{ ha}$) correspond à $10\,000\text{ m}^2$, exprime l'aire de ce champ en hectares.
            \item L'agriculteur souhaite semer du blé. Il faut $15\text{ kg}$ de graines par hectare. Quelle quantité de graines devra-t-il acheter ?
        \end{enumerate}
    \end{exercice}
    
    
    \begin{exercice}[Problème : La voile de bateau]
        Une voilerie fabrique une grand-voile pour un bateau. Le plan de la voile est modélisé par un grand triangle avec un angle obtus. La base de la voile mesure $3,5\text{ m}$ et le point le plus haut de la voile se situe à une distance perpendiculaire de $6\text{ m}$ par rapport à l'axe de cette base.
        \begin{enumerate}
            \item Calcule l'aire de la voile en $\text{m}^2$.
            \item Le tissu spécifique pour cette voile coûte $45$\,\euro{} le mètre carré. Quel sera le coût total du tissu nécessaire pour fabriquer cette voile ?
        \end{enumerate}
    \end{exercice}
    
    
    \begin{exercice}[Problème de synthèse : Figure composée]
        On souhaite peindre la façade d'un hangar qui a la forme suivante : un rectangle surmonté d'un parallélogramme de même largeur.
        \begin{itemize}
            \item La largeur totale de la façade rectangulaire est de $12\text{ m}$.
            \item La partie rectangulaire a une hauteur de $5\text{ m}$.
            \item La partie supérieure (en forme de parallélogramme) a une hauteur de $3\text{ m}$.
        \end{itemize}
        \begin{enumerate}
            \item Calcule l'aire de la partie rectangulaire.
            \item Calcule l'aire de la partie en forme de parallélogramme.
            \item Déduis-en l'aire totale de la façade.
            \item Un pot de peinture permet de couvrir $15\text{ m}^2$. Combien de pots entiers faut-il acheter pour peindre toute la façade ?
        \end{enumerate}
    \end{exercice}
