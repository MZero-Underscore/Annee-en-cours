\subsection{Utiliser les propriétés du parallélogramme}

    \begin{definition}
        Un \textbf{parallélogramme} est un quadrilatère dont les côtés opposés sont parallèles deux à deux.
    \end{definition}
    
    
    \begin{propriete}
        Si un quadrilatère est un parallélogramme, alors :
        \begin{itemize}
            \item Ses côtés opposés sont de \textbf{même longueur}.
            \item Ses diagonales se coupent en \textbf{leur milieu}. (Ce milieu est le centre de symétrie de la figure).
            \item Ses angles opposés sont de \textbf{même mesure}.
            \item Ses angles consécutifs sont \textbf{supplémentaires} (leur somme fait $180^{\circ}$).
        \end{itemize}
    \end{propriete}

\subsection{Reconnaître un parallélogramme}

    \begin{proprieteTitre}{Propriétés réciproques (Comment reconnaître un parallélogramme ?)}
    Si un quadrilatère (non croisé) possède :
    \begin{itemize}
        \item ses diagonales qui se coupent \textbf{en leur milieu},
        \item OU ses côtés opposés de \textbf{même longueur} deux à deux,
        \item OU deux côtés opposés \textbf{parallèles et de même longueur},
    \end{itemize}
    alors c'est un \textbf{parallélogramme}.
    \end{proprieteTitre}

\subsection{Utiliser les propriétés des parallélogrammes particuliers}
    
    \begin{definition}
        \begin{itemize}
            \item Un \textbf{rectangle} est un quadrilatère qui possède $4$ angles droits.
            \item Un \textbf{losange} est un quadrilatère qui possède $4$ côtés de même longueur.
            \item Un \textbf{carré} est un quadrilatère qui possède $4$ angles droits \textbf{et} $4$ côtés de même longueur (c'est à la fois un rectangle et un losange).
        \end{itemize}
        
        \begin{center}
            \hspace{2cm}
            % --- RECTANGLE ---
            \begin{tikzpicture}[scale=0.6]
                \draw[thick, fill=blue!5] (0,0) -- (3,0) -- (3,2) -- (0,2) -- cycle;
                % Codage des angles droits
                \draw[red, thick] (0.3,0) -- (0.3,0.3) -- (0,0.3);
                \draw[red, thick] (2.7,0) -- (2.7,0.3) -- (3,0.3);
                \draw[red, thick] (2.7,2) -- (2.7,1.7) -- (3,1.7);
                \draw[red, thick] (0.3,2) -- (0.3,1.7) -- (0,1.7);
                \node[below] at (1.5,-0.2) {\textbf{Rectangle}};
            \end{tikzpicture}
            \hfill
            % --- LOSANGE ---
            \begin{tikzpicture}[scale=0.6]
                \draw[thick, fill=green!5] (0,1.5) -- (2,0) -- (4,1.5) -- (2,3) -- cycle;
                % Codages des côtés (double trait perpendiculaire)
                \draw[red, thick] (0.85, 0.65) -- (1.05, 0.95); \draw[red, thick] (0.95, 0.55) -- (1.15, 0.85); % Bas Gauche
                \draw[red, thick] (2.95, 0.95) -- (3.15, 0.65); \draw[red, thick] (2.85, 0.85) -- (3.05, 0.55); % Bas Droite
                \draw[red, thick] (2.85, 2.15) -- (3.05, 2.45); \draw[red, thick] (2.95, 2.05) -- (3.15, 2.35); % Haut Droite
                \draw[red, thick] (0.95, 2.45) -- (1.15, 2.15); \draw[red, thick] (0.85, 2.35) -- (1.05, 2.05); % Haut Gauche
                \node[below] at (2,-0.2) {\textbf{Losange}};
            \end{tikzpicture}
            \hfill
            % --- CARRÉ ---
            \begin{tikzpicture}[scale=0.6]
                \draw[thick, fill=orange!5] (0,0) -- (2.2,0) -- (2.2,2.2) -- (0,2.2) -- cycle;
                % Codage des angles droits
                \draw[red, thick] (0.3,0) -- (0.3,0.3) -- (0,0.3);
                \draw[red, thick] (1.9,0) -- (1.9,0.3) -- (2.2,0.3);
                \draw[red, thick] (1.9,2.2) -- (1.9,1.9) -- (2.2,1.9);
                \draw[red, thick] (0.3,2.2) -- (0.3,1.9) -- (0,1.9);
                % Codages des côtés (trait simple)
                \draw[red, thick] (1.1,-0.15) -- (1.1,0.15);
                \draw[red, thick] (2.05,1.1) -- (2.35,1.1);
                \draw[red, thick] (1.1,2.05) -- (1.1,2.35);
                \draw[red, thick] (-0.15,1.1) -- (0.15,1.1);
                \node[below] at (1.1,-0.2) {\textbf{Carré}};
            \end{tikzpicture}
            \hspace{2cm}
            \text{ }
        \end{center}
    \end{definition}
    
    
    \begin{proprieteTitre}{Propriétés des diagonales}
        Le rectangle, le losange et le carré sont des \textbf{parallélogrammes}. Leurs diagonales se coupent donc \textbf{en leur milieu}. De plus :
        \begin{itemize}
            \item Les diagonales d'un \textbf{rectangle} sont de \textbf{même longueur}.
            \item Les diagonales d'un \textbf{losange} sont \textbf{perpendiculaires}.
            \item Les diagonales d'un \textbf{carré} sont \textbf{de même longueur ET perpendiculaires}.
        \end{itemize}

        \begin{center}
            \hspace{2cm}
            % --- RECTANGLE (Diagonales) ---
            \begin{tikzpicture}[scale=0.6]
                \draw[thick, fill=blue!5] (0,0) -- (3,0) -- (3,2) -- (0,2) -- cycle;
                % Diagonales
                \draw (0,0) -- (3,2);
                \draw (0,2) -- (3,0);
                % Codages des mêmes longueurs (1 trait)
                \draw[red, thick] (0.65, 0.65) -- (0.85, 0.35);
                \draw[red, thick] (2.15, 1.65) -- (2.35, 1.35);
                \draw[red, thick] (0.65, 1.35) -- (0.85, 1.65);
                \draw[red, thick] (2.15, 0.35) -- (2.35, 0.65);
                \node[below] at (1.5,-0.2) {\textbf{Rectangle}};
            \end{tikzpicture}
            \hfill
            % --- LOSANGE (Diagonales) ---
            \begin{tikzpicture}[scale=0.6]
                \draw[thick, fill=green!5] (0,1.5) -- (2,0) -- (4,1.5) -- (2,3) -- cycle;
                % Diagonales
                \draw (0,1.5) -- (4,1.5);
                \draw (2,0) -- (2,3);
                % Angle droit au centre
                \draw[red, thick] (2, 1.7) -- (2.2, 1.7) -- (2.2, 1.5);
                % Codages milieu diag horizontale (1 trait vertical)
                \draw[red, thick] (1, 1.35) -- (1, 1.65);
                \draw[red, thick] (3, 1.35) -- (3, 1.65);
                % Codages milieu diag verticale (2 traits horizontaux)
                \draw[red, thick] (1.85, 0.7) -- (2.15, 0.7); \draw[red, thick] (1.85, 0.8) -- (2.15, 0.8);
                \draw[red, thick] (1.85, 2.2) -- (2.15, 2.2); \draw[red, thick] (1.85, 2.3) -- (2.15, 2.3);
                \node[below] at (2,-0.2) {\textbf{Losange}};
            \end{tikzpicture}
            \hfill
            % --- CARRÉ (Diagonales) ---
            \begin{tikzpicture}[scale=0.6]
                \draw[thick, fill=orange!5] (0,0) -- (2.2,0) -- (2.2,2.2) -- (0,2.2) -- cycle;
                % Diagonales
                \draw (0,0) -- (2.2,2.2);
                \draw (0,2.2) -- (2.2,0);
                % Angle droit au centre (incliné à 45° pour suivre les diagonales)
                \draw[red, thick] (1.25, 1.25) -- (1.1, 1.4) -- (0.95, 1.25);
                % Codages des mêmes longueurs (1 trait)
                \draw[red, thick] (0.45, 0.65) -- (0.65, 0.45);
                \draw[red, thick] (1.55, 1.75) -- (1.75, 1.55);
                \draw[red, thick] (0.45, 1.55) -- (0.65, 1.75);
                \draw[red, thick] (1.55, 0.45) -- (1.75, 0.65);
                \node[below] at (1.1,-0.2) {\textbf{Carré}};

            \text{ }

    \end{proprieteTitre}

\subsection{Reconnaître un quadrilatère particulier}

    \begin{proprieteTitre}{Reconnaître un quadrilatère particulier (par ses côtés et angles)}
    \begin{itemize}
        \item Si un parallélogramme possède \textbf{deux côtés consécutifs perpendiculaires} (c'est-à-dire un angle droit), alors c'est un \textbf{rectangle}.
        \item Si un parallélogramme possède \textbf{deux côtés consécutifs de même longueur}, alors c'est un \textbf{losange}.
        \item Si un parallélogramme possède \textbf{un angle droit ET deux côtés consécutifs de même longueur}, alors c'est un \textbf{carré}.
    \end{itemize}
    \end{proprieteTitre}

\section{Aire d'un parallélogramme et application aux triangles}
% ==========================================

    \begin{proprieteTitre}{Aire d'un parallélogramme}
        Pour calculer l'aire d'un parallélogramme, on multiplie la longueur d'un côté (appelé \textbf{\textcolor{blue}{Base}}) par la \textbf{\textcolor{red}{hauteur}} relative à ce côté :
        \begin{center}
            \large $\mathcal{A} = \textcolor{blue}{\text{Base}} \times \textcolor{red}{\text{Hauteur}}$
        \end{center}
        
        \vspace{0.2cm}
        \begin{center}
        \begin{tikzpicture}[scale=0.75]
            % Parallélogramme
            \draw[thick, fill=blue!5] (0,0) -- (5,0) -- (6,2.5) -- (1,2.5) -- cycle;
            % Base (en bleu épais)
            \draw[line width=1.5pt, blue] (0,0) -- (5,0) node[midway, below] {\textbf{Base ($b$)}};
            % Hauteur (en rouge pointillé épais)
            \draw[line width=1.2pt, dashed, red] (1,2.5) -- (1,0) node[midway, right] {\textbf{Hauteur ($h$)}};
            % Angle droit
            \draw[red, thick] (1,0.2) -- (1.2,0.2) -- (1.2,0);
        \end{tikzpicture}
        \end{center}
    \end{proprieteTitre}

    \begin{proprieteTitre}{Aire d'un triangle}
        Un triangle quelconque correspond toujours à la \textbf{moitié} d'un parallélogramme. Son aire est donc :
        \begin{center}
            \large $\mathcal{A} = \dfrac{\textcolor{blue}{\text{Base}} \times \textcolor{red}{\text{Hauteur}}}{2}$
        \end{center}
        
        \vspace{0.2cm}
        \begin{center}
        \begin{tikzpicture}[scale=0.75]
            % Parallélogramme "fantôme" pour l'explication visuelle
            \draw[dashed, gray!70] (0,0) -- (5,0) -- (7,3) -- (2,3) -- cycle;
            % Triangle mis en valeur
            \draw[thick, fill=green!5] (0,0) -- (5,0) -- (2,3) -- cycle;
