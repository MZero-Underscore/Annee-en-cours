\section{Distance à zéro d'un nombre relatif}

    \begin{definitionTitre}{Distance à zéro}
        \begin{minipage}{0.49\linewidth}
            La \textbf{distance à zéro} d'un nombre relatif \( a \) est la distance \( OA \) où :
            \begin{itemize}
                \item \( O \) est l'origine d'une droite graduée,
                \item \( A \) est le point d'abscisse \( a \) sur cette droite.
            \end{itemize}

            On note ce nombre $|a|$.
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.49\linewidth}
            \begin{center}
                \includegraphics[scale=0.65]{images/DistanceZero.png}
            \end{center}
        \end{minipage}
    \end{definitionTitre}

    
    \begin{exemple} 
        \begin{minipage}{0.5\linewidth}
            1. Donner la distance à zéro des nombres suivants :
            \begin{itemize}
                \item $ |-12| = \dots$
    
                \medskip
                
                \item $ |+5,4| = \dots$
                
                \medskip
                
                \item $ |0| = \dots$
                
                \medskip
                
                \item $ |-125,78| = \dots$
                
                \medskip
            \end{itemize}
        \end{minipage}
        \hfill
        \begin{minipage}{0.49\linewidth}
            2. Trouver tous les nombres relatifs $x$ dans les cas suivants :
            \begin{enumerate}
                \item $|x| = 9$

                \bigskip
                
                \item $|x| = 1,5$

                \bigskip
                
                \item $|x| = 0$

                \medskip
            \end{enumerate}
        \end{minipage}

        3. Compléter avec les symboles < ou > en justifiant par les calculs des distances à zéro.

        \medskip
        
        \begin{minipage}{0.24\linewidth}
            $|-8| \ldots |-5|$

            \vspace{2cm}
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.24\linewidth}
            $|-14| \ldots |+12|$

            \vspace{2cm}
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.24\linewidth}
            $|+3,5| \ldots |-3,6|$

            \vspace{2cm}
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.24\linewidth}
            $|-12| \ldots 2$

            \vspace{2cm}
        \end{minipage}

    \end{exemple}


\section{Additionner}
\subsection{Additionner deux nombres de même signe}
    
    \begin{definitionTitre}{Addition}
        Pour \textbf{additionner} deux nombres de \textbf{même signe} :
        \begin{itemize}
            \item on garde le signe commun,
            \item on effectue la somme des distances à zéro.
        \end{itemize}
    \end{definitionTitre}

    
    \begin{exemple}
        Effectuer les calculs suivants.

        \medskip
        
        \begin{minipage}{0.24\linewidth}
            $A = (-5)+ (-13)$

            \vspace{2cm}
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.24\linewidth}
            $B=-45+(-23)$

            \vspace{2cm}
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.24\linewidth}
            $C=-24,5+(-45,3)$

            \vspace{2cm}
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.24\linewidth}
            $D=-12,01+(-89,99)$

            \vspace{2cm}
        \end{minipage}
    \end{exemple}


\subsection{Additionner deux nombres de signes différents}
    
    \begin{definitionTitre}{Addition}
        Pour \textbf{additionner} deux nombres de \textbf{signes différents} :
        \begin{itemize}
            \item on garde le signe du nombre qui a la plus grande distance à zéro,
            \item on effectue la différence des distances à zéro.
        \end{itemize}
    \end{definitionTitre}
    
    \begin{exemple}
        Effectuer les calculs suivants.

        \medskip
        
        \begin{minipage}{0.24\linewidth}
            $A = 17+ (-13)$

            \vspace{2cm}
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.24\linewidth}
            $B=65+(-115)$

            \vspace{2cm}
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.24\linewidth}
            $C=-24,1+45,3$

            \vspace{2cm}
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.24\linewidth}
            $D=12,4+(-17,1)$

            \vspace{2cm}
        \end{minipage}
    \end{exemple}

    
\section{Soustraire}
\subsection{Nombres opposés}

    \begin{definitionTitre}{Opposés}
        Deux nombres sont \textbf{opposés} si leur somme vaut 0.

        \medskip

        L'opposé d'un nombre relatif $a$ se note $-a$. On a toujours $a+(-a)=0$
    \end{definitionTitre}
    
        
    \begin{exemple}
        1. Compléter le tableau suivant :
        \begin{center}
        \begin{tabular}{|c|c|c|c|c|c|}
            \hline
            Nombre $a$ & $+5$ & $-12,4$ & $0$ & $-(-3)$ & $a$ \\ \hline
            Opposé de $a$ & \dots & \dots & \dots & \dots & \dots \\ \hline
        \end{tabular}
        \end{center}

        \medskip
    
        \begin{minipage}{0.49\linewidth}
            2. Calculer les expressions suivantes en utilisant la définition des nombres opposés :
            \begin{itemize}
                \item $A = (+45) + (-45) = \dots$

                \smallskip
                
                \item $B = (-7,2) + (+7,2) = \dots$

                \smallskip
                
                \item $C = (+15) + (-8) + (-15) = \dots$

                \smallskip
            \end{itemize}    
        \end{minipage}
        \hfill \vline \hfill
        \begin{minipage}{0.49\linewidth}
            3. Répondre par vrai ou faux.
            \begin{enumerate}
                \item L'opposé d'un nombre négatif est toujours positif.
                \item Deux nombres opposés ont toujours la même distance à zéro.
                \item Si la somme de deux nombres est nulle, alors ces nombres sont opposés.
            \end{enumerate}
        \end{minipage}
    \end{exemple}
    
    
    \begin{propriete} 
        Soustraire un nombre revient à ajouter son opposé.
    \end{propriete}

\subsection{Enchaînement d'additions et de soustractions}
    
    \begin{definitionTitre}{somme algébrique}
        Le résultat d'un enchaînement d’additions et/ou de soustractions s'appelle une \textbf{somme algébrique}.
    \end{definitionTitre}

    \begin{methode}
        Pour calculer une somme algébrique,
        \begin{enumerate}
            \item On simplifie au maximum l'expression : plus de parenthèses ni de soustraction de nombre négatif.
            \item On entoure les termes positifs d'une même couleur.
            \item On entoure les termes négatifs d'une autre couleur.
            \item On procède à l'addition des nombres positifs puis à celle des nombres négatifs.
            \item On termine par l'addition des deux sommes précédentes.
        \end{enumerate}
    \end{methode}

    \begin{exemple}
        Effectuer les calculs suivants.

        \medskip

        \begin{minipage}{0.30\linewidth}
            $A=(-10)+(+24)-6+7+(-6)$

            \vspace{3cm}
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.30\linewidth}
            $B=17+(+3)+(-6)+(-45)+2$

            \vspace{3cm}
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.30\linewidth}
            $C=-5-(-6)+23-17$

            \vspace{3cm}
        \end{minipage}
    \end{exemple}
