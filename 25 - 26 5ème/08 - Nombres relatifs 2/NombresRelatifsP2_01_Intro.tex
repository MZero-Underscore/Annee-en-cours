    \begin{tcolorbox}[colback=blue!5,colframe=blue!75,title=1. La Règle d'Or]
        Chaque nombre est représenté par des jetons :
        \begin{itemize}
            \item Un jeton \textbf{Bleu} \pos représente $(+1)$.
            \item Un jeton \textbf{Rouge} \negat représente $(-1)$.
            \item \textbf{La paire nulle :} \pos + \negat = $0$. Ils s'annulent et quittent le plateau.
        \end{itemize}
    \end{tcolorbox}
    
    \section{L'Addition (Réunir)}
    \textit{Pour additionner, on met tous les jetons ensemble et on simplifie les paires nulles.}

    \section{La Soustraction (Enlever)}
    \textit{Pour soustraire, on retire les jetons demandés. Si on n'en a pas assez, on peut ajouter des paires nulles (\pos + \negat).}

        \includegraphics[scale=0.30]{images/Activite01_Exemple02.png}
