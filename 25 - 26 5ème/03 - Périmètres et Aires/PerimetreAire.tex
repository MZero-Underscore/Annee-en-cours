\section{Calculer un périmètre}
\subsection{Définition}
    \begin{definitionTitre}{Périmètre}
        Le \textbf{périmètre} d'une figure est la \textbf{longueur} de son contour. L’unité de mesure est le mètre ou l'un de ses multiples ou sous-multiples.
    \end{definitionTitre}

\subsection{Cas des polygones}
    \begin{propriete}
        Le périmètre d'un polygone est la somme des longueurs de ses côtes.
    \end{propriete}

\subsection{Cas du cercle}
    \begin{propriete}
        \begin{minipage}{0.5\linewidth}
            Le périmètre d'un cercle de rayon $r$  ou de diamètre $d$ s'obtient grâce à la formule suivante :

            Périmètre = $2 \times \pi \times r = d \times \pi$
        \end{minipage}
        \hfill
        \begin{minipage}{0.49\linewidth}
            \begin{center}
                \includegraphics[scale=0.4]{images/Cercle.png}
            \end{center}      
        \end{minipage}
    \end{propriete}


    \begin{remarque}
        \begin{itemize}
            \item Le nombre $\pi$ est le coefficient de proportionnalité liant le diamètre et le périmètre du cercle. Il est connu depuis au moins \np{4000} ans.

            \item La lettre $\pi$ vient de l'alphabet grec. C'est notre P majuscule. L'utilisation de cette lettre pour désigner ce nombre est relativement récente et date d'environ 400 ans.

            \item \underline{Ce nombre n'est pas décimal}, mais on peut en donner une valeur approchée : $\pi\simeq 3,1415 \ldots$
        \end{itemize}
    \end{remarque}

\section{Calculer une aire}
\subsection{Définition}
    \begin{definitionTitre}{Définition}
        L'\textbf{aire d'une figure} est la mesure de la surface contenue à l'intérieur de la figure. Une unité usuelle est le $m^2$ ou l'un de ses multiples ou sous-multiples.
    \end{definitionTitre}


    \begin{methode}
        Mesurer l'aire d'une figure en cm$^2$, c'est compter combien il faut de carrés de 1 cm de côté pour paver l'intérieur de la figure.
    \end{methode}
