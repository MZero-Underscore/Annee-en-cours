            \begin{ZoneReponse}{2.5cm}
                % 1,35 * 150 = 202,5 tonnes
            \end{ZoneReponse}

            \item Pour brûler, le moteur de la fusée mélange de l'hydrogène liquide avec de l'oxygène liquide dans un \textbf{ratio $1:6$}. On parle de mélange \textbf{hydrogène-oxygène cryogénique}.

            Si la fusée utilise $200$ kg d'hydrogène, quelle masse d'oxygène doit-elle emporter pour que le mélange soit parfait ?
            
            \begin{ZoneReponse}{2.5cm}
                % Calcul : 200 * 6 = 1200 kg d'oxygène.
            \end{ZoneReponse}
        \end{enumerate}
    \end{exerciceNote}

    \begin{exerciceNote}{5}
        \textbf{Méthode visuelle pour les ratios :}
        Imaginez des "parts" ou des "cases" égales.
        
        \begin{enumerate}
            \item \textbf{Partage de bonbons (56 bonbons, ratio $3:5$) :}
            Julie a 3 parts, Nathan a 5 parts. Total = 8 parts.
            
            Une part vaut : $56 \div 8 = 7$ bonbons.
            
            Julie : $3 \times 7 = \mathbf{21}$ bonbons. Nathan : $5 \times 7 = \mathbf{35}$ bonbons.
            
            \item \textbf{Recette de cocktail ($4:3:1$) :}
            L'ananas (3 parts) représente $60$ cL.
            
            Une part vaut : $60 \div 3 = 20$ cL.
            
            Orange (4 parts) : $4 \times 20 = \mathbf{80 \text{ cL}}$.
            Grenadine (1 part) : $1 \times 20 = \mathbf{20 \text{ cL}}$.
            
            Volume total : $80 + 60 + 20 = \mathbf{160 \text{ cL}}$.
            
            \item \textbf{Économies ($2:3$) :}
            Total des parts = 5. Une part = $120 \div 5 = 24$ \euro.
            
            Pierre (2 parts) : $48$ \euro. Paul (3 parts) : $72$ \euro.
            
            Différence : $72 - 48 = 24$ \euro. \textbf{C'est Vrai.}
        \end{enumerate}
    \end{exerciceNote}
