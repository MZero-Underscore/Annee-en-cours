        \begin{enumerate}[leftmargin=*]
            \item \textbf{( - 4 ) + ( + 2 ) = - 2}
            \begin{ZoneReponse}{2cm}       
                Dessin : \negat \negat \negat \negat \pos \pos 

                \medskip

                On barre deux paires (\pos\negat). Il reste deux jetons rouges : \negat \negat.
            \end{ZoneReponse}

            \item On veut calculer : {\large \textbf{( + 3 ) - ( - 2 )}}
            \begin{enumerate}[leftmargin=*]
                \item Au départ : \pos \pos \pos. 
                \item Pour enlever 2 rouges alors qu'il n'y en a pas :
                \item \textbf{Dessine} : 
                \begin{ZoneReponse}{2cm}       
                    On ajoute deux paires nulles : (\pos \negat) (\pos \negat).

                    \medskip
                    
                    Le plateau devient : \pos \pos \pos \pos \pos \negat \negat.
                \end{ZoneReponse}
                
                \item Après avoir enlevé les 2 rouges, il reste 5 bleus. Résultat : \textbf{+ 5}.
            \end{enumerate}

            \item Si je retire un jeton rouge \negat, le résultat va \textbf{augmenter}.
            \begin{ZoneReponse}{2cm}       
                Justification : Retirer une dette ou un signe négatif revient à ajouter de la valeur. Mathématiquement, $-(-1) = +1$.
            \end{ZoneReponse}
        \end{enumerate}
    \end{exerciceNote}
