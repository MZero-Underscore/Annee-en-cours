\section{Expressions littérales}

    \begin{definition}
        Une \textbf{expression littérale} est une expression dans laquelle un ou plusieurs nombres sont remplacés par des \textbf{lettres}. Ces lettres désignent des nombres dont on ne connaît pas la valeur.
        On les appelle des \textbf{variables}.
    \end{definition}

\subsection{Simplification d'écriture}
    
    \begin{convention}
        Pour alléger les écritures, on peut \textbf{supprimer le signe $\times$} :
        \begin{itemize}
            \item Devant une \textbf{lettre} : $3 \times a$ s'écrit $3a$.
            \item Devant une \textbf{parenthèse} : $5 \times (x + 2)$ s'écrit $5(x + 2)$.
        \end{itemize}
    \end{convention}
    
    
    \begin{remarque}
        \begin{itemize}
            \item On ne supprime jamais le signe $\times$ entre deux nombres ($3 \times 5$ ne s'écrit pas $35$ !).
            \item $1 \times x$ s'écrit simplement $x$.
            \item $x \times x$ se note $x^2$ (se lit « $x$ au carré »).
        \end{itemize}
    \end{remarque}
    
    
    \begin{exemple}
        Simplifier les écritures suivantes.

        \medskip

        \begin{minipage}{0.24\linewidth}
            $A=3\times a$

            \vspace{1cm}
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.24\linewidth}
            $B=8\times a+5\times b$

            \vspace{1cm}
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.24\linewidth}
            $C=x\times 5$

            \vspace{1cm}
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.24\linewidth}
            $D=3\times (x+5)$

            \vspace{1cm}
        \end{minipage}
    \end{exemple}


    \begin{convention}
        Pour noter le produit d'un nombre par lui-même, on utilise la notation \textbf{puissance} :
        \begin{itemize}
            \item $x \times x$ se note \textbf{$x^2$} (et se lit « $x$ au carré »).
            \item $x \times x \times x$ se note \textbf{$x^3$} (et se lit « $x$ au cube »).
        \end{itemize}
    \end{convention}
    
    
    \begin{exemple}
        Simplifier les écritures suivantes.

        \medskip

        \begin{minipage}{0.24\linewidth}
            $A=x\times x\times x \times 4$

            \vspace{1cm}
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.24\linewidth}
            $B=2x\times x$

            \vspace{1cm}
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.24\linewidth}
            $C=7\times x\times y \times x \times 3$

            \vspace{1cm}
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.24\linewidth}
            $D=x\times3+10\times x\times x$

            \vspace{1cm}
        \end{minipage}
    \end{exemple}
    
    
    \begin{remarque}
        Attention à ne pas confondre $x + x$ et $x \times x$ :
        \begin{itemize}
            \item $x + x = 2x$ (c'est le double de $x$).
            \item $x \times x = x^2$ (c'est le carré de $x$).
        \end{itemize}
    \end{remarque}

\subsection{Évaluer une expression}
    
    \begin{definitionTitre}{Évaluer}
        \textbf{Évaluer} une expression littérale, c'est calculer sa valeur numérique en \textbf{remplaçant les variables} par leur valeur dans cette expression littérale.
    \end{definitionTitre}

\subsection{Tester une égalité}
    
    \begin{definitionTitre}{Égalité}
        Une \textbf{égalité} est constituée de deux membres séparés par un signe « $=$ ».
    \end{definitionTitre}
    
    
    \begin{methode}
        Pour tester si une égalité est \textbf{vraie} pour des valeurs données :
        \begin{enumerate}
            \item On calcule le membre de gauche séparément.
            \item On calcule le membre de droite séparément.
            \item On compare les deux résultats.
        \end{enumerate}
    \end{methode}
