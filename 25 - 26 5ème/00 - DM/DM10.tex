    \begin{exerciceTitre}{Problème de fractions}
        \begin{enumerate}
            \item Le trajet complet représente $\frac{5}{5}$. Après le premier jour, il reste : $\frac{5}{5} - \frac{2}{5} = \frac{3}{5}$ du trajet.
            \item On calcule la fraction du reste : $\frac{3}{4} \times \frac{3}{5} = \frac{9}{20}$. Ils effectuent $\frac{9}{20}$ du trajet total le second jour.
            \item Total parcouru : $\frac{2}{5} + \frac{9}{20} = \frac{8}{20} + \frac{9}{20} = \frac{17}{20}$ du trajet.
            \item La part restante est $\frac{20}{20} - \frac{17}{20} = \frac{3}{20}$. Comme un quart équivaut à $\frac{1}{4} = \frac{5}{20}$ et que $\frac{3}{20} < \frac{5}{20}$, il leur restera moins d'un quart du parcours pour le dernier jour.
        \end{enumerate}
    \end{exerciceTitre}

        \end{tikzpicture}

        \begin{enumerate}
            \setcounter{enumi}{4}
            \item Dans le quadrilatère $ABCD$, les diagonales $[AC]$ et $[BD]$ se coupent en leur milieu $I$. C'est donc un parallélogramme. De plus, ses diagonales $(AC)$ et $(BD)$ sont perpendiculaires. Or, si un parallélogramme a ses diagonales perpendiculaires, c'est un losange. Donc $ABCD$ est un losange.
            
            \item Les diagonales d'un carré doivent être de même longueur. Ici, $AC = 8 \text{ cm}$ et $BD = 6 \text{ cm}$. Les diagonales n'étant pas de même longueur ($AC \neq BD$), ce losange n'est pas un carré.
        \end{enumerate}

    \begin{exerciceTitre}{Calculs d'aires}
        \begin{enumerate}
            \item $\text{Aire}(ABCD) = \text{Longueur} \times \text{largeur} = 10 \times 6 = 60 \text{ m}^2$.
            \item $ABCD$ est un rectangle, ses côtés opposés ont la même longueur, donc $CD = AB = 6 \text{ m}$. \\
            $CDE$ est rectangle en $C$ : $\text{Aire}(CDE) = \frac{CD \times CE}{2} = \frac{6 \times 4}{2} = 12 \text{ m}^2$.
            \item Pour le parallélogramme $ADGF$, la base est $[AD]$. Comme $ABCD$ est un rectangle, ses côtés opposés sont de même longueur, d'où $AD = BC = 10 \text{ m}$. La hauteur relative à cette base mesure $3 \text{ m}$. \\
            $\text{Aire}(ADGF) = \text{base} \times \text{hauteur} = 10 \times 3 = 30 \text{ m}^2$.
            \item $\text{Aire totale} = 60 + 12 + 30 = 102 \text{ m}^2$.
            \item On effectue la division : $102 \div 15 = 6,8$. Avec 6 sacs, la surface couverte n'est que de $90 \text{ m}^2$. Il est donc nécessaire d'entamer un 7\textsuperscript{e} sac. Ils devront acheter au minimum \textbf{7 sacs}.
        \end{enumerate}
    \end{exerciceTitre}
