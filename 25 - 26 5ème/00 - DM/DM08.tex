        \begin{enumerate}
            \item On remplit un ballon de volley-ball avec $3$ litres d'air. 
            \begin{enumerate}
                \item Construire un tableau de proportionnalité permettant de calculer la masse d'air ajoutée dans le ballon.
                \item Quelle est la masse d'air ajoutée dans le ballon ?
            \end{enumerate}

            \begin{ZoneReponse}{4cm}
                \begin{tabular}{|l|c|c|c|} \hline
                Volume (L) & 5 & 3 & \np{160000} \\ \hline
                Masse (g)  & 6,5 & ? & ?\\ \hline
                \end{tabular} \\

                \medskip
                
                Coefficient : $6,5 \div 5 = 1,3$. Masse pour 3L : $3 \times 1,3 = \mathbf{3,9}$ \textbf{g}.
            \end{ZoneReponse}
            
            \item La salle de classe contient environ $160$ m$^3$ d'air. 
            \begin{enumerate}
                \item Sachant que $1$ m$^3 = \np{1000}$ litres, compléter le tableau de proportionnalité du 1. pour calculer la masse d'air contenue dans la salle de classe.
                \item Calculer la masse de l'air contenue dans la salle en kilogrammes. 
            \end{enumerate}
            

            \begin{ZoneReponse}{4cm}
                $160$ m$^3 = 160 \times \np{1000} = \np{160000}$ Litres. \\[2ex]
                Masse en g : $\np{160000} \times 1,3 = \np{208000}$ g. \\[2ex]
                Masse en kg : $\np{208000} \div \np{1000} = \mathbf{208}$ \textbf{kg}.
            \end{ZoneReponse}
        \end{enumerate}
        

    \end{exercice}
    
        
% --- EXERCICE 5 : LES RATIOS ---
    \begin{exercice}[Partage et Ratios]
        Pour préparer un cocktail de fruits, on mélange du jus d'orange, du jus d'ananas et du jus de mangue dans le ratio $3 : 2 : 1$.
        On souhaite préparer $1,2$ litre de ce cocktail.
        \medskip
        
        \begin{enumerate}
            \item Quelle quantité de chaque jus doit-on verser ?
        \end{enumerate}
        
        \begin{ZoneReponse}{3cm}
            Nombre total de parts : $3 + 2 + 1 = 6$ parts. \\
            Valeur d'une part : $1,2 \div 6 = 0,2$ litre. \\
            Jus d'orange : $3 \times 0,2 = \mathbf{0,6}$ \textbf{L}. \\
            Jus d'ananas : $2 \times 0,2 = \mathbf{0,4}$ \textbf{L}. \\
            Jus de mangue : $1 \times 0,2 = \mathbf{0,2}$ \textbf{L}.
        \end{ZoneReponse}

        \begin{enumerate}[resume]
            \item Un autre jour, on prépare le même cocktail (ratio $3:2:1$). Cette fois-ci, on utilise exactement $45$ cL de jus d'orange.
            \begin{enumerate}
                \item Quel est le volume total de ce nouveau mélange ?
                \item Quelle quantité de jus de mangue a été utilisée ?
            \end{enumerate}
        \end{enumerate}

        \begin{ZoneReponse}{3cm}
            a) Le jus d'orange représente 3 parts. Une part vaut donc : $45 \div 3 = 15$ cL. \\
            Le mélange total contient 6 parts ($3+2+1$), donc : $6 \times 15 = \mathbf{90}$ \textbf{cL}. 

            \medskip
            
            b) Le jus de mangue représente 1 part, soit : $1 \times 15 = \mathbf{15}$ \textbf{cL}.
        \end{ZoneReponse}
    \end{exercice}
