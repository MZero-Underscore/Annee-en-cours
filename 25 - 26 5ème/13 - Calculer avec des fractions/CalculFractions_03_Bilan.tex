    \begin{exerciceTitre}{Additions et soustractions (même dénominateur)}
        Calcule les expressions suivantes et donne le résultat sous la forme d'une fraction (simplifiée si possible) :
        \begin{multicols}{2}
        \begin{enumerate}[label=\alph*)]
            \item $A = \dfrac{3}{7} + \dfrac{2}{7}$
            \item $B = \dfrac{11}{15} - \dfrac{4}{15}$
            \item $C = \dfrac{13}{8} + \dfrac{5}{8}$
            \item $D = \dfrac{27}{100} - \dfrac{17}{100}$
        \end{enumerate}
        \end{multicols}
    \end{exerciceTitre}
    
    
    \begin{exerciceTitre}{Additions et soustractions (dénominateurs multiples)}
        Calcule les expressions suivantes en respectant les étapes (transformer pour avoir le même dénominateur avant de calculer) :
        \begin{multicols}{2}
        \begin{enumerate}[label=\alph*)]
            \item $E = \dfrac{1}{3} + \dfrac{5}{6}$
            \item $F = \dfrac{7}{12} - \dfrac{1}{4}$
            \item $G = \dfrac{3}{10} + \dfrac{2}{5}$
            \item $H = 1 - \dfrac{3}{4}$
        \end{enumerate}
        \end{multicols}
    \end{exerciceTitre}
    
    
    \begin{exerciceTitre}{Multiplications de fractions}
        Calcule les produits suivants (rappel : multiplier les numérateurs entre eux et les dénominateurs entre eux) :
        \begin{multicols}{2}
        \begin{enumerate}[label=\alph*)]
            \item $I = \dfrac{2}{3} \times \dfrac{4}{5}$
            \item $J = \dfrac{7}{2} \times \dfrac{3}{5}$
            \item $K = 5 \times \dfrac{4}{7}$
            \item $L = \dfrac{1}{10} \times \dfrac{1}{10}$
        \end{enumerate}
        \end{multicols}
    \end{exerciceTitre}
    
    
    \begin{exerciceTitre}{Problème : Le partage du gâteau}
        Pour son anniversaire, Yanis a acheté un grand gâteau. 
        \begin{itemize}
            \item Il en donne d'abord $\dfrac{1}{3}$ à sa petite sœur.
            \item Il en donne ensuite $\dfrac{3}{8}$ à ses amis.
        \end{itemize}
        
        \begin{enumerate}
            \item Quelle fraction du gâteau a été mangée au total ?
            \item Quelle fraction du gâteau reste-t-il pour Yanis ?
        \end{enumerate}
    \end{exerciceTitre}
    
    
    \begin{exerciceTitre}{Problème : La bibliothèque}
        Dans une bibliothèque, $\dfrac{2}{5}$ des livres sont des bandes dessinées. Parmi ces bandes dessinées, $\dfrac{3}{4}$ sont des mangas.
        \begin{enumerate}
            \item Quelle fraction du total des livres de la bibliothèque les mangas représentent-ils ? (Indice : calculer les $\dfrac{3}{4}$ de $\dfrac{2}{5}$).
            \item Si la bibliothèque contient $1\,000$ livres, combien y a-t-il de mangas ?
        \end{enumerate}
    \end{exerciceTitre}
    
    
    \begin{exerciceTitre}{Problème : Le réservoir d'essence}
        Le réservoir d'une voiture est rempli aux $\dfrac{2}{3}$. Le conducteur consomme $\dfrac{1}{6}$ de la capacité totale du réservoir pour faire un trajet.
        \begin{enumerate}
            \item Quelle fraction de la capacité totale reste-t-il dans le réservoir après le trajet ?
            \item Si le réservoir contient $60$ litres au total, combien de litres reste-t-il ?
        \end{enumerate}
    \end{exerciceTitre}
