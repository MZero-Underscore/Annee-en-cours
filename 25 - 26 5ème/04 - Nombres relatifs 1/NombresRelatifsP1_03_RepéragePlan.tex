    \begin{exercice}
        \begin{center}
            \includegraphics[scale=0.30]{images/ReperagePlan.png}
        \end{center}

        \begin{enumerate}
            \item Donner l'abscisse des points A, B et C. On dit que l'on repère ces points sur l'axe tracé.
            
            \textbf{On utilise parfois le mot \og coordonnée \fg pour remplacer le mot \og abscisse \fg.}

            \vspace{0.75cm}

            \item On souhaite repérer le point D, c'est à dire indiquer précisément son emplacement sur la figure. Comment peut-on procéder?

            \vspace{0.75cm}

            \item Faire de même pour les point E, F et G.

            \vspace{0.75cm}

            \item Adapter les coordonnées de A, B et C pour qu'elles aient la même forme que celle de D, E, F et G.             
        \end{enumerate}
    \end{exercice}
