\section{Recueillir et organiser des données}
    
    \begin{definition}
        On appelle \textbf{donnée} une information collectée (nombre, mot, couleur). Pour les analyser, on les regroupe souvent dans un \textbf{tableau} ou on les représente par un \textbf{graphique}. 
    \end{definition}
    
    
    \begin{exemple}[Exploitation de données (Calcul mental)]
        Voici les notes obtenues par une classe de 5ème lors d'un test : \\
        \textbf{12 ; 15 ; 8 ; 12 ; 10 ; 15 ; 12 ; 8 ; 10 ; 12.}
    
        \begin{enumerate}[label=\arabic*/]
            \item \textbf{Organiser les données dans le tableau suivant :}
            \begin{center}
            \begin{tabular}{|l|c|c|c|c|c|}
                \hline
                \textbf{Note} & 8 & 10 & 12 & 15 & \textbf{Total} \\ \hline
                \textbf{Nombre d'élèves (Effectif)} & \points[2] & \points[2] & \points[2] & \points[2] & \points[2] \\ \hline
            \end{tabular}
            \end{center}
    
            \item \textbf{Représenter ces résultats par un diagramme en bâtons :}
            \begin{ZoneReponse}{6cm}
                \begin{center}
                \begin{tikzpicture}[scale=0.8]
                    % Grille de lecture grise et légère
                    \draw[step=0.5cm, gray!20, very thin] (0,0) grid (9,5);

\section{Effectifs et Fréquences}
    
    \begin{definition}
        \begin{itemize}
            \item L'\textbf{effectif} d'une donnée est le nombre de fois où cette donnée apparaît. 
            \item L'\textbf{effectif total} est la somme de tous les effectifs (le nombre total de données). 
            \item La \textbf{fréquence} est le quotient de l'effectif par l'effectif total : $ \text{Fréquence} = \frac{\text{Effectif}}{\text{Effectif total}} $. 
        \end{itemize}
    \end{definition}
    
    
    \begin{methodeTitre}{Lire une fréquence en pourcentage}
        Pour obtenir la fréquence en pourcentage, on peut écrire la fréquence sous forme décimale et **lire le nombre à partir du chiffre des centièmes comme unité**.
    \end{methodeTitre}
    
    
    \begin{exemple}
        \begin{minipage}{0.45\linewidth}
            Dans une classe de 20 élèves, 3 élèves portent des lunettes.
            \begin{itemize}
                \item La fréquence est : $\frac{3}{20} = 0,15 = \dfrac{15}{100}$. 
                \item On lit « 15 centièmes », \\
                \text{ } \hfill ce qui correspond à **15 \%**.
            \end{itemize}
        \end{minipage}
        \hfill \vline \hfill
        \begin{minipage}{0.45\linewidth}
            Autre cas : si la fréquence est $0,4$.
            \begin{itemize}
                \item On peut écrire $0,4 = 0,40 = \dfrac{40}{100}$.
                \item On lit « 40 centièmes », \\
                \text{ } \hfill ce qui correspond à **40 \%**.
            \end{itemize}
        \end{minipage}
    \end{exemple}
    
    
    \begin{remarque}
        Cette méthode souligne que « pour-cent » signifie littéralement « pour cent ». Si on a $0,125$; on lit $12,5$ centièmes, soit $12,5\%$. 
    \end{remarque}

\section{Moyenne et moyenne pondérée}

    \begin{definition}
        La \textbf{moyenne} d'une série est le nombre obtenu en additionnant toutes les données et en divisant le résultat par l'effectif total.
    \end{definition}
    
    
    \begin{theoremeTitre}{Calcul avec les effectifs (Moyenne pondérée)}
        Pour calculer une moyenne à partir d'un tableau, on additionne les produits de chaque donnée par son effectif, puis on divise par l'effectif total :
        \[ \text{Moyenne} = \frac{\text{Somme des (Valeur $\times$ Effectif)}}{\text{Effectif Total}} \]
    \end{theoremeTitre}
    
    
    \begin{exemple}
        Reprenons les notes du test de calcul mental du début de chapitre :
        \begin{center}
        \begin{tabular}{|l|c|c|c|c|c|}
        \hline
        \textbf{Note} & 8 & 10 & 12 & 15 & \textbf{Total} \\ \hline
        \textbf{Effectif} & 2 & 2 & 4 & 2 & \textbf{10} \\ \hline
        \end{tabular}
        \end{center}
    
        \textbf{Calcul de la moyenne :}
        \[ M = \frac{8 \times \points[1] + 10 \times \points[1] + 12 \times \points[1] + 15 \times \points[1]}{10} \]
        \[ M = \frac{\points[1] + \points[1] + \points[1] + \points[1]}{10} = \frac{\points[2]}{10} = \points[2] \]
        
        
        La moyenne de la classe à ce test est de \points[2] / 20.
    \end{exemple}
    
    \begin{remarque}
        La moyenne pondérée donne le même résultat que si on additionnait les 10 notes une par une, mais le calcul est plus rapide et mieux organisé.
    \end{remarque}

            \begin{itemize}[leftmargin=*]
                \item \textbf{Produits :} Valeur $\times$ Effectif.
                \item \textbf{Somme :} Additionner les produits.
                \item \textbf{Division :} Diviser par le total.
            \end{itemize}
            \textit{Exemple : $\frac{120}{10} = 12$}
        }

    \begin{cultureMaths}
        Le mot \textbf{« Statistique »} tire son origine du latin \textit{status}, qui signifie « état ». À l'origine, cette discipline n'était pas une branche des mathématiques, mais un \textbf{\textit{instrument de gouvernance}} au service des souverains.
        
        Il servait avant tout à \textbf{dénombrer} les sujets, les récoltes et les armées afin d'administrer le pays avec précision. Ce n'est qu'à partir du \textbf{XVII\textsuperscript{e} siècle} que la rencontre entre ces inventaires et le \textit{\textbf{calcul des probabilités}} a donné naissance à la science que nous étudions aujourd'hui : une méthode pour transformer des données brutes en \textbf{prévisions fiables}.
    \end{cultureMaths}
