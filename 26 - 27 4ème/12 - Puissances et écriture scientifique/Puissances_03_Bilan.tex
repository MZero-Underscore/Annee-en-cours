    \begin{exercice}{Comprendre l'exposant négatif}
    Compléter les égalités suivantes pour faire apparaître l'inverse, puis donner l'écriture décimale :
    
    \vspace{0.2cm}
    \renewcommand{\arraystretch}{2.5}
    \begin{enumerate}
        \item $2^{-3} = \dfrac{1}{2^{\dots}} = \dfrac{1}{\dots} = \makebox[2.5cm]{\dotfill}$ \medskip
        \item $5^{-2} = \dfrac{1}{\dots\dots} = \makebox[3cm]{\dotfill}$ \medskip
        \item $10^{-3} = \dfrac{1}{\dots\dots} = \makebox[3cm]{\dotfill}$ \medskip
        \item $(-2)^{-4} = \dfrac{1}{\dots\dots} = \makebox[3cm]{\dotfill}$ \medskip
        \item $(-5)^{-1} = \makebox[3cm]{\dotfill}$
    \end{enumerate}

    \begin{exercice}{Le cas particulier des puissances de 10}
        Cet exercice prépare la prochaine notion du cours : le cas particulier du nombre 10.
        
        \vspace{0.2cm}
        \textbf{Partie A : Exposants positifs}
        \begin{enumerate}
            \item Calculer sous forme d'un nombre entier : \medskip
            \begin{itemize}
                \renewcommand{\itemsep}{0.2cm}
                \item $10^1 = \makebox[3cm]{\dotfill}$ \hspace{2cm} $\bullet $ $10^3 = \makebox[3cm]{\dotfill}$
                \item $10^2 = \makebox[3cm]{\dotfill}$ \hspace{2cm} $\bullet $ $10^4 = \makebox[3cm]{\dotfill}$
            \end{itemize}
            \item Que remarque-t-on concernant le lien entre l'exposant et le résultat ?
            
            \begin{ZoneReponse}{2cm}\end{ZoneReponse}
            \item En déduire, sans poser de calcul, l'écriture décimale de $10^9$ : \makebox[5cm]{\dotfill}
        \end{enumerate}
        
        
        \textbf{Partie B : Exposants négatifs}
        \begin{enumerate}
            \item Écrire sous forme de fraction décimale puis de nombre décimal : \medskip
            \begin{itemize}
                \renewcommand{\itemsep}{0.2cm}
                \item $10^{-1} = \makebox[2cm]{\dotfill} = \makebox[2.5cm]{\dotfill}$ \hspace{2cm} $\bullet $ $10^{-3} = \makebox[3cm]{\dotfill}$ \medskip
                \item $10^{-2} = \makebox[2cm]{\dotfill} = \makebox[2.5cm]{\dotfill}$ \hspace{2cm} $\bullet $ $10^{-4} = \makebox[3cm]{\dotfill}$
            \end{itemize}
            \item Que remarque-t-on concernant le lien entre l'exposant négatif et l'écriture décimale du résultat ?
            
            \begin{ZoneReponse}{2cm}\end{ZoneReponse}
            \item En déduire, sans poser de calcul, l'écriture décimale de $10^{-6}$ : \makebox[4cm]{\dotfill}
        \end{enumerate}
    \end{exercice}
