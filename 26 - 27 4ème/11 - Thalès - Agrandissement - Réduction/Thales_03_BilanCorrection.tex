% CORRECTION PROBLÈME 1
% ==========================================
\begin{exercice}[Correction - Problème 1]
    \begin{minipage}{0.65\textwidth}
        \textbf{1.} Dans le triangle $ABC$, le côté le plus long est $[AC]$ avec $AC = 2,5$ m.
        D'une part : $AC^2 = 2,5^2 = 6,25$. \\
        D'autre part : $AB^2 + BC^2 = 2,4^2 + 0,7^2 = 5,76 + 0,49 = 6,25$.
        
        Puisque $AC^2 = AB^2 + BC^2$, d'après la réciproque du théorème de Pythagore, le triangle $ABC$ est rectangle en $B$.
        Le mur de la rampe est donc parfaitement vertical au sol.
    \end{minipage}
    \hfill
    \begin{minipage}{0.3\textwidth}
        \begin{center}
            \begin{tikzpicture}[scale=1.5]
                % Triangle ABC
                \coordinate (B) at (0,0);
                \coordinate (A) at (2.4,0);
                \coordinate (C) at (0,0.7);
                \draw[thick] (A) -- (B) -- (C) -- cycle;
                
                % Angle droit en B
                \draw (0.15,0) -- (0.15,0.15) -- (0,0.15);
                
                % Points D et E
                \coordinate (D) at (1.6,0); % AD = 0.8 -> distance depuis A
                \coordinate (E) at (1.58, 0.24); % Point sur AC
                
                \draw[red, thick] (D) -- (E);
                
                % Labels
                \node[below] at (A) {$A$};
                \node[below] at (B) {$B$};
                \node[left] at (C) {$C$};
                \node[below] at (D) {$D$};
                \node[above right] at (E) {$E$};
            \end{tikzpicture}
        \end{center}
    \end{minipage}

    \vspace{0.3cm}
    \textbf{2.} Les points $A, D, B$ d'une part, et $A, E, C$ d'autre part, sont alignés dans le même ordre.\\
    Comparons les rapports $\frac{AD}{AB}$ et $\frac{AE}{AC}$ en utilisant l'égalité des produits en croix : \\
    D'une part : $AD \times AC = 0,8 \times 2,5 = 2$ \\
    D'autre part : $AE \times AB = 0,85 \times 2,4 = 2,04$
    
    Puisque les produits en croix ne sont pas égaux ($2 \neq 2,04$), on en déduit que $\frac{AD}{AB} \neq \frac{AE}{AC}$.\\
    D'après la contraposée du théorème de Thalès, les droites $(DE)$ et $(BC)$ \textbf{ne sont pas parallèles}.
\end{exercice}

\begin{exercice}[Correction - Problème 2]
    \begin{minipage}{0.5\textwidth}
        \begin{center}
            \begin{tikzpicture}[scale=0.12]
                % Coordonnées
                \coordinate (O) at (0,0);
                \coordinate (B) at (10,0); % Echelle adaptée pour le schéma
                \coordinate (N) at (40,0);
                \coordinate (T) at (10,7.5);
                \coordinate (S) at (40,30);
                
                % Lignes
                \draw[thick] (O) -- (N);
                \draw[thick, blue] (B) -- (T);
                \draw[thick, blue] (N) -- (S);
                \draw[dashed] (O) -- (S);
                
                % Angles droits
                \draw (B)++(0,2) -- ++(2,0) -- ++(0,-2);
                \draw (N)++(0,2) -- ++(-2,0) -- ++(0,-2);
                
                % Labels
                \node[below] at (O) {$O$};
                \node[below] at (B) {$B$};
                \node[below] at (N) {$N$};
                \node[above left] at (T) {$T$};
                \node[above] at (S) {$S$};
            \end{tikzpicture}
        \end{center}
    \end{minipage}
    \hfill
    \begin{minipage}{0.45\textwidth}
        \textbf{1.} On sait que le repère et l'éolienne sont perpendiculaires au sol. Donc $(BT) \perp (ON)$ et $(SN) \perp (ON)$. \\
        Or, si deux droites sont perpendiculaires à une même troisième, alors elles sont parallèles entre elles. Donc $(BT) \parallel (SN)$.
        
        \vspace{0.3cm}
        \textbf{2.} Les points $O, B, N$ sont alignés dans cet ordre, donc : \\
        $ON = OB + BN = 2 + 38 = 40$ m.
    \end{minipage}

    \vspace{0.3cm}
    \textbf{3.} Dans les triangles $OBT$ et $ONS$, on sait que les points $O, B, N$ et $O, T, S$ sont alignés et que les droites $(BT)$ et $(SN)$ sont parallèles.\\
    D'après le théorème de Thalès, on a l'égalité des rapports : $\frac{OB}{ON} = \frac{OT}{OS} = \frac{BT}{SN}$.\\
    En remplaçant par les valeurs : $\frac{2}{40} = \frac{1,5}{SN}$.\\
    Grâce au produit en croix : $SN = \frac{40 \times 1,5}{2} = \frac{60}{2} = 30$ m. L'éolienne mesure 30 mètres de haut.
    
    \vspace{0.3cm}
    \textbf{4.} Le triangle $ONS$ est rectangle en $N$. D'après le théorème de Pythagore : \\
    $OS^2 = ON^2 + SN^2$ \\
    $OS^2 = 40^2 + 30^2 = 1600 + 900 = 2500$. \\
    $OS = \sqrt{2500} = 50$ m. Le rayon laser parcourt 50 mètres.
\end{exercice}

\begin{exercice}[Correction - Problème 3]
    \begin{minipage}{0.65\textwidth}
        \textbf{1.} Le triangle $SPA$ est rectangle en $P$. D'après le théorème de Pythagore :\\
        $SA^2 = SP^2 + PA^2$\\
        $3,5^2 = 2,8^2 + PA^2$\\
        $12,25 = 7,84 + PA^2$\\
        $PA^2 = 12,25 - 7,84 = 4,41$\\
        $PA = \sqrt{4,41} = 2,1$ m. La poutre est fixée à 2,1 m du mur.
        
        \vspace{0.3cm}
        \textbf{2.} Les points $S, N, P$ d'une part et $S, M, A$ d'autre part sont alignés dans le même ordre.
        Comparons les rapports $\frac{SN}{SP}$ et $\frac{SM}{SA}$ avec le produit en croix :\\
        D'une part : $SN \times SA = 1,2 \times 3,5 = 4,2$\\
        D'autre part : $SM \times SP = 1,5 \times 2,8 = 4,2$
        
        Puisque $SN \times SA = SM \times SP$, les rapports sont égaux ($\frac{SN}{SP} = \frac{SM}{SA}$).\\
        D'après la \textbf{réciproque du théorème de Thalès}, les droites $(MN)$ et $(PA)$ sont parallèles. L'étagère sera parfaitement horizontale.
    \end{minipage}
    \hfill
    \begin{minipage}{0.3\textwidth}
        \begin{center}
            \begin{tikzpicture}[scale=1.2]
                \coordinate (P) at (0,0);
                \coordinate (S) at (0,2.8);
                \coordinate (A) at (2.1,0);
                \draw[thick] (P) -- (S) -- (A) -- cycle;
                \draw (0.2,0) -- (0.2,0.2) -- (0,0.2);
                \coordinate (N) at (0, 1.6); 
                \coordinate (M) at (0.9, 1.6); 
                \draw[purple, thick] (N) -- (M);
                \node[below] at (P) {$P$}; \node[above] at (S) {$S$}; \node[below] at (A) {$A$};
                \node[left] at (N) {$N$}; \node[above right] at (M) {$M$};
            \end{tikzpicture}
        \end{center}
    \end{minipage}
\end{exercice}

\begin{exercice}[Correction - Problème 4]
    \begin{minipage}{0.65\textwidth}
        \textbf{1.} Dans $RST$ rectangle en $R$, d'après Pythagore : \\
        $ST^2 = RS^2 + RT^2 = 4,8^2 + 6,4^2 = 64$. \\
        Donc $ST = 8$ cm.
        
        \vspace{0.3cm}
        \textbf{2.} Les points $S, M, R$ et $S, N, T$ sont alignés dans cet ordre. \\
        Calculons $SM = SR - RM = 4,8 - 3 = 1,8$ cm.\\
        Comparons $\frac{SM}{SR}$ et $\frac{SN}{ST}$ :\\
        $SM \times ST = 1,8 \times 8 = 14,4$ \\
        $SN \times SR = 3 \times 4,8 = 14,4$ \\
        Comme $SM \times ST = SN \times SR$, les rapports sont égaux. D'après la \textbf{réciproque du théorème de Thalès}, $(MN) // (RT)$.
        
        \vspace{0.3cm}
        \textbf{3.} On applique le théorème de Thalès : $\frac{MN}{RT} = \frac{SM}{SR} = \frac{1,8}{4,8}$.\\
        $MN = \frac{1,8 \times 6,4}{4,8} = 2,4$ cm.
        
        \vspace{0.3cm}
        \textbf{4.} Dans $SMN$, le plus long côté est $[SN]$. $SN^2 = 3^2 = 9$.\\
        $SM^2 + MN^2 = 1,8^2 + 2,4^2 = 3,24 + 5,76 = 9$.\\
        $SN^2 = SM^2 + MN^2$, donc d'après la réciproque de Pythagore, $SMN$ est rectangle en $M$.
    \end{minipage}
    \hfill
    \begin{minipage}{0.3\textwidth}
        \begin{center}
            \begin{tikzpicture}[scale=0.6]
                \coordinate (R) at (0,0); \coordinate (S) at (0,4.8); \coordinate (T) at (6.4,0);
                \draw[thick] (R) -- (S) -- (T) -- cycle;
                \draw (0.4,0) -- (0.4,0.4) -- (0,0.4);
                \coordinate (M) at (0,3); \coordinate (N) at (2.4,3);
                \draw[blue, thick] (M) -- (N);
                \draw (0.4, 3) -- (0.4, 3.4) -- (0, 3.4);
                \node[below left] at (R) {$R$}; \node[above] at (S) {$S$}; \node[below right] at (T) {$T$};
                \node[left] at (M) {$M$}; \node[above right] at (N) {$N$};
            \end{tikzpicture}
        \end{center}
    \end{minipage}
\end{exercice}
