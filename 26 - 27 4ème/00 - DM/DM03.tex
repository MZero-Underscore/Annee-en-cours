    \begin{exercice}
        \begin{minipage}{0.5\linewidth}
            On considère un cube d'arête 5 cm.

            \begin{enumerate}
                \item Calculer la valeur exacte de la longueur AC.

                Dans le triangle ADC, rectangle en D, d'après le théorème de Pythagore.

                $AC^2=AD^2+DC^2$\\
                $AC^2=5^2+5^2$\\
                $AC^2=25+25$\\
                $AC^2=50$\\
                Donc $AC=\sqrt{50}$

                \item En déduire la valeur exacte de longueur de la grande diagonale du cube EC.

                \item Donner l'arrondi de EC au mm près.

                Dans le triangle AEC, rectangle en A, d'après le théorème de Pythagore.

                $EC^2=AE^2+AC^2$\\
                $EC^2=5^2+50$\\
                $EC^2=25+50$\\
                $EC^2=75$\\
                Donc $EC=\sqrt{75}\simeq 8,7$ cm (arrondi au mm près)
            \end{enumerate}
        \end{minipage}
