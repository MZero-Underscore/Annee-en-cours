                     5 & 12 \\ \hline
                     125 & 
                \end{array}$

            \end{ZoneReponse}

            \begin{ZoneReponse}{1.5cm}
                $\begin{array}{c|c}
                     5 &  \\ \hline
                     125 & 350
                \end{array}$

                \quad Il faut 14L de peinture blanche.
            \end{ZoneReponse}
        \end{enumerate}

            \end{ZoneReponse}
        \end{enumerate}
