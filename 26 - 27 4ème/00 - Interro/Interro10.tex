        \textbf{Le sac de billes}
        Un sac contient 5 billes rouges, 3 billes bleues et 2 billes vertes.
        \begin{enumerate}
            \item Quelle est la probabilité de tirer une bille bleue ? \\
            \textbf{Il y a 10 billes au total. $P(\text{bleue}) = \frac{3}{10} = \mathbf{0,3}$.}
            \item On ajoute des billes noires. $P(\text{noire}) = \frac{1}{3}$. Combien de billes noires ? \\
            \textbf Il y a 10 billes avant l'ajout des billes noires.
            
            On cherche le nombre $x$ de billes noires telles que $\frac{x}{10+x}=\frac{1}{3}$\\
            Par talonnement, on s'oriente rapidement vers $x=5$ ce qui nous donne $\frac{5}{15}=\frac{1}{3}$

            Il y a donc 5 billes noires dans le sac.
            
        \end{enumerate}
    \end{exerciceNote}

        $\frac{OA}{OC} = \frac{OB}{OD} \implies \frac{5}{12,5} = \frac{4}{OD} \implies OD = \frac{4 \times 12,5}{5} = \frac{50}{5} = \mathbf{10}$ \textbf{cm}.
    \end{exerciceNote}

    \begin{exerciceNote}{4}
        \textbf{Problème : La statue en bronze}
        \begin{enumerate}
            \item Montrer que le coefficient d'agrandissement $k$ est égal à 5. \\
            \textbf{$k = \frac{\text{Hauteur réelle}}{\text{Hauteur maquette}} = \frac{100 \text{ cm}}{20 \text{ cm}} = \mathbf{5}$.}
            \item Par quel nombre est multiplié le volume ? \textbf{Par $k^3 = 5^3 = \mathbf{125}$.}
            \item Volume final : \textbf{$V = 150 \times 125 = \mathbf{18\,750}$ cm$^3$ (soit 18,75 litres).}
            \item \textbf{Défi :} Masse de la statue ? \\
            \textbf{La masse est proportionnelle au volume. $M = 1,2 \text{ kg} \times 125 = \mathbf{150}$ kg.}
        \end{enumerate}
    \end{exerciceNote}

    \begin{exerciceNote}{4}
        \textbf{Exercice Expert : Le logo}
        \begin{enumerate}
            \item Montrer que $BC = 13$ cm. \\
            \textbf{Dans le triangle $ABC$ rectangle en $A$, d'après le théorème de Pythagore : \\
            $BC^2 = AB^2 + AC^2 = 5^2 + 12^2 = 25 + 144 = 169$. Donc $BC = \sqrt{169} = \mathbf{13}$ cm.}
            \item Rapport exact de la réduction : \textbf{$k = \frac{MN}{BC} = \frac{3,9}{13} = \mathbf{0,3}$.}
            \item Aire de $ABC$ : \textbf{$\frac{5 \times 12}{2} = \mathbf{30}$ cm$^2$.}
            \item Aire de $AMN$ : \textbf{Aire$(AMN) = \text{Aire}(ABC) \times k^2 = 30 \times 0,3^2 = 30 \times 0,09 = \mathbf{2,7}$ cm$^2$.}
        \end{enumerate}
    \end{exerciceNote}

                    \draw (2,0.8) node[above]{$B$} -- (2,0) node[below]{$A$};
                \end{tikzpicture}
            \end{center}
        \end{minipage}
        
        \textbf{Réponse :} Les droites $(AB)$ et $(CD)$ sont parallèles. D'après le \textbf{théorème de Thalès} : \\
        $\frac{OA}{OC} = \frac{OB}{OD} \implies \frac{6}{14} = \frac{4,5}{OD} \implies OD = \frac{4,5 \times 14}{6} = \frac{63}{6} = \mathbf{10,5}$ \textbf{cm}.
    \end{exerciceNote}

    \begin{exerciceNote}{4}
        \textbf{Problème : Le vase en céramique}
        \begin{enumerate}
            \item Montrer que le coefficient d'agrandissement $k$ est égal à 4. \\
            \textbf{$k = \frac{\text{Hauteur réelle}}{\text{Hauteur maquette}} = \frac{60 \text{ cm}}{15 \text{ cm}} = \mathbf{4}$.}
            \item Par quel nombre est multiplié le volume ? \textbf{Par $k^3 = 4^3 = \mathbf{64}$.}
            \item Volume final : \textbf{$V = 200 \times 64 = \mathbf{12\,800}$ cm$^3$ (soit 12,8 litres).}
            \item \textbf{Défi :} Masse du vase ? \\
            \textbf{La masse est proportionnelle au volume. $M = 0,8 \text{ kg} \times 64 = \mathbf{51,2}$ kg.}
        \end{enumerate}
    \end{exerciceNote}

    \begin{exerciceNote}{4}
        \textbf{Exercice Expert : Le logo}
        \begin{enumerate}
            \item Montrer que $BC = 10$ cm. \\
            \textbf{Dans le triangle $ABC$ rectangle en $A$, d'après le théorème de Pythagore : \\
            $BC^2 = AB^2 + AC^2 = 6^2 + 8^2 = 36 + 64 = 100$. Donc $BC = \sqrt{100} = \mathbf{10}$ cm.}
            \item Rapport exact de la réduction : \textbf{$k = \frac{MN}{BC} = \frac{4}{10} = \mathbf{0,4}$ (ou $\frac{2}{5}$).}
            \item Aire de $ABC$ : \textbf{$\frac{6 \times 8}{2} = \mathbf{24}$ cm$^2$.}
            \item Aire de $AMN$ : \textbf{Aire$(AMN) = \text{Aire}(ABC) \times k^2 = 24 \times 0,4^2 = 24 \times 0,16 = \mathbf{3,84}$ cm$^2$.}
        \end{enumerate}
    \end{exerciceNote}
