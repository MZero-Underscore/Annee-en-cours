    % --- PARTIE 1 : RÉDUCTION ---
    \begin{exercice}[Réduction d'expressions]
    Réduire les expressions suivantes :
    
    \begin{minipage}[t]{0.48\textwidth}
    \begin{itemize}[label=\textbullet]
        \item $A = 5x + 3x$
        \item $B = 12y - 4y$
        \item $C = 7x + 2 + 4x - 5$
        \item $D = 3x^2 + 5x - x^2 + 2$
    \end{itemize}
    \end{minipage}
    \hfill
    \begin{minipage}[t]{0.48\textwidth}
    \begin{itemize}[label=\textbullet]
        \item $A' = -4a + 10a$
        \item $B' = 8b - 11b$
        \item $C' = 5 + 3x - 2 - 7x$
        \item $D' = 4x^2 - 2x + 3x^2 + 5x$
    \end{itemize}
    \end{minipage}
    \end{exercice}

    % --- PARTIE 2 : DÉVELOPPEMENT ---
    \begin{exercice}[Développement (Simple Distributivité)]
    Développer puis réduire chaque expression en utilisant la distributivité.
    
    \begin{minipage}[t]{0.48\textwidth}
    \begin{itemize}[label=\textbullet]
        \item $E = 3(x + 4)$
        \item $F = 5(2x - 3)$
    \end{itemize}
    \end{minipage}
    \hfill
    \begin{minipage}[t]{0.48\textwidth}
    \begin{itemize}[label=\textbullet]
        \item $G = x(x + 7)$
        \item $H = -2(4x - 5)$
    \end{itemize}
    \end{minipage}

    \end{exercice}
    
    \begin{ZoneReponse}{3cm}
    \end{ZoneReponse}
    
    % --- PARTIE 3 : MÉLANGE ET CALCULS ---
    \begin{exercice}[Calcul de la valeur d'une expression]
    On considère l'expression $I = 4(x + 2) - 3x$.
    \begin{enumerate}
        \item Développer et réduire l'expression $I$.
        \item Calculer la valeur de $I$ pour $x = 5$.
    \end{enumerate}
    \end{exercice}
