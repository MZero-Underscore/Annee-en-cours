    % --- EXERCICE 2 ---
    \begin{exercice}[Utiliser les propriétés]
        Soit $ABCD$ un parallélogramme tel que $AB = 6$~cm, $AD = 4$~cm et l'angle $\widehat{BAD} = 60^{\circ}$.
        
        \begin{enumerate}
            \item Réaliser une figure à main levée.
            \item Quelle est la longueur des segments $[CD]$ et $[BC]$ ? Justifier la réponse.
            \item Quelle est la mesure des angles $\widehat{BCD}$ et $\widehat{ADC}$ ? Justifier.
        \end{enumerate}
    \end{exercice}
    

    % --- EXERCICE 3 : CONSTRUCTIONS ---
    \begin{exercice}[Constructions de parallélogrammes]
    Effectuer les constructions suivantes sur votre cahier (pensez à faire un schéma à main levée au brouillon) :
    
    \begin{enumerate}
        \item \textbf{Version Côtés / Angle :} Construire $EFGH$ tel que $EF = 6$~cm, $EH = 4$~cm et $\widehat{FEH} = 65^{\circ}$.
        \item \textbf{Version Diagonales :} Construire $IJKL$ de centre $O$ tel que ses diagonales mesurent $IK = 10$~cm et $JL = 6$~cm, avec l'angle $\widehat{IOJ} = 50^{\circ}$.
        \item \textbf{Version Côtés / Diagonale :} Construire $MNOP$ tel que $MN = 5$~cm, $MP = 3$~cm et la diagonale $NP = 7$~cm.
    \end{enumerate}
    \end{exercice}

    % --- EXERCICE 5 : DÉMONSTRATIONS (RECONNAÎTRE) ---
    \begin{exercice}[Reconnaître un parallélogramme]
    Dans chaque cas, réaliser une figure à main levée puis expliquer pourquoi le quadrilatère est un parallélogramme :
    \begin{enumerate}
        \item Un quadrilatère $RSTU$ dont les diagonales $[RT]$ et $[SU]$ se coupent en un point $K$ tel que $RK = KT$ et $SK = KU$.
        \item Un quadrilatère dont les côtés opposés sont parallèles deux à deux.
    \end{enumerate}
    \end{exercice}

    \begin{exercice}[Reconnaître un parallélogramme]
        On considère un triangle EFG quelconque. On construit I le milieu de [EG] puis H le symétrique de F par rapport à I.

        Quelle est la nature du quadrilatère EFGH ainsi construit ? 
    \end{exercice}


    \begin{exercice}[Un cas particulier]

        Construire deux segments [AB] et [CD] de sorte que les propriétés suivantes soient vérifiées :
        \begin{itemize}
            \item AB=DC
            \item $(AB)$ et $(DC)$ sont parallèles.
            \item Les points A et D sont sur la partie gauche de la feuille et les points B et C sur la partie plus à droite de la feuille.
        \end{itemize}
        
    \end{exercice}
