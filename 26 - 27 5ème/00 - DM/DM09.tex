    \begin{exercice}[Data Centers : Les données et l'énergie] 
        \textbf{On pourra faire les calculs à la calculatrice !}
        
        Un \textit{Data Center} télécharge et traite en permanence des données sur Internet. 
        
        \begin{enumerate}
            \item \textbf{Le téléchargement des données} \\
            Ce centre traite \textbf{300 Go} (Gigaoctets) de données en \textbf{4 heures}.
            \begin{enumerate}
                \item En supposant que le débit est régulier, quel volume de données est traité en 1 heure ?
                \item \textit{Pour se rendre compte :} Un film en très haute définition pèse environ 5 Go. Combien de films ce Data Center traite-il en une seule heure ?
                \item Combien de temps lui faut-il pour traiter \textbf{1 To} (Téraoctet), sachant que $1 \text{ To} = 1\,000 \text{ Go}$ ?
            \end{enumerate}
            
            \item \textbf{La consommation d'énergie} \\
            Pour fonctionner, les serveurs informatiques ont besoin d'électricité. La consommation est proportionnelle au temps d'utilisation. \\
            Ce Data Center consomme \textbf{60 MWh} (Mégawattheures) en \textbf{10 heures}.
            \begin{enumerate}
                \item Quelle est sa consommation électrique pour une journée complète (24 heures) ?
                \item Quelle est sa consommation pour un mois complet de 30 jours ?
                \item \textit{Pour se rendre compte :} Un collège moyen (avec le chauffage, l'éclairage et la cantine) consomme environ \textbf{400 MWh en une année entière}. \\
                À combien d'années de fonctionnement de ce collège correspond la consommation de ce Data Center en \textbf{un seul mois} ?
            \end{enumerate}
        \end{enumerate}
    \end{exercice}

    \begin{exercice}[Des périmètres mystérieux]
        On considère deux figures géométriques :
        \begin{itemize}
            \item \textbf{Figure 1 :} Un rectangle de longueur $2x$ et de largeur $x + 3$.
            \item \textbf{Figure 2 :} Un triangle de côtés $3x$, $x + 5$ et $2x + 1$.
        \end{itemize}
        \begin{enumerate}
            \item Écrire l'expression du périmètre de la Figure 1, puis la réduire. \\
            \textcolor{red}{$P_1 = 2x + (x+3) + 2x + (x+3) = 2x + x + 3 + 2x + x + 3 \mathbf{ = 6x + 6}$.}
            \item Écrire l'expression du périmètre de la Figure 2, puis la réduire. \\
            \textcolor{red}{$P_2 = 3x + (x + 5) + (2x + 1) = 3x + x + 5 + 2x + 1 \mathbf{ = 6x + 6}$.}
            \item Que remarque-t-on ? \\
            \textcolor{red}{On remarque que les deux figures ont \textbf{le même périmètre} quelle que soit la valeur de $x$.}
            \item Calculer ce périmètre lorsque $x = 4$ cm. \\
            \textcolor{red}{$P = 6 \times 4 + 6 = 24 + 6 \mathbf{ = 30 \text{ cm}}$.}
        \end{enumerate}
    \end{exercice}
    
    
    \begin{exercice}[Data Centers : Les données et l'énergie] 
    Un \textit{Data Center} télécharge et traite en permanence des données sur Internet. 
    
    \begin{enumerate}
        \item \textbf{Le téléchargement des données} \\
        Ce centre traite \textbf{300 Go} (Gigaoctets) de données en \textbf{4 heures}.
        \begin{enumerate}
            \item En supposant que le débit est régulier, quel volume de données est traité en 1 heure ? \\
            \textcolor{red}{\textbf{Réponse :} $300 \div 4 = \mathbf{75 \text{ Go}}$. En une heure, 75 Go sont traités.}
            
            \item \textit{Pour se rendre compte :} Un film en très haute définition pèse environ 5 Go. Combien de films ce Data Center traite-il en une seule heure ? \\
            \textcolor{red}{\textbf{Réponse :} $75 \div 5 = \mathbf{15 \text{ films}}$. Cela correspond à 15 films par heure.}
            
            \item Combien de temps lui faut-il pour traiter \textbf{1 To} (Téraoctet), sachant que $1 \text{ To} = 1\,000 \text{ Go}$ ? \\
            \textcolor{red}{\textbf{Réponse :} $1\,000 \div 75 \approx \mathbf{13,33 \text{ heures}}$ (soit environ 13h et 20 minutes).}
        \end{enumerate}
        
        \item \textbf{La consommation d'énergie} \\
        Pour fonctionner, les serveurs informatiques ont besoin d'électricité. La consommation est proportionnelle au temps d'utilisation. \\
        Ce Data Center consomme \textbf{60 MWh} (Mégawattheures) en \textbf{10 heures}.
        \begin{enumerate}
            \item Quelle est sa consommation électrique pour une journée complète (24 heures) ? \\
            \textcolor{red}{\textbf{Réponse :} Consommation pour 1h : $60 \div 10 = 6$ MWh. \\ Pour 24h : $6 \times 24 = \mathbf{144 \text{ MWh}}$.}
            
            \item Quelle est sa consommation pour un mois complet de 30 jours ? \\
            \textcolor{red}{\textbf{Réponse :} $144 \times 30 = \mathbf{4\,320 \text{ MWh}}$.}
            
            \item \textit{Pour se rendre compte :} Un collège moyen (avec le chauffage, l'éclairage et la cantine) consomme environ \textbf{400 MWh en une année entière}. \\
            À combien d'années de fonctionnement de ce collège correspond la consommation de ce Data Center en \textbf{un seul mois} ? \\
            \textcolor{red}{\textbf{Réponse :} $4\,320 \div 400 = \mathbf{10,8 \text{ années}}$. La consommation mensuelle du Data Center représente presque \textbf{11 ans} de la vie du collège !}
        \end{enumerate}
    \end{enumerate}
    \end{exercice}
