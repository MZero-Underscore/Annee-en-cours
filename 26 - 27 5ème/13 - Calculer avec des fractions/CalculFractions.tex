\section{Rappels}

    \begin{definition}
        Soit $a$ et $b$ deux nombres quelconques ($b$ est non nul).
        
        Le nombre qui, multiplié par $b$, donne $a$, s'appelle le \textbf{quotient} de $a$ par $b$. On le note $\frac{a}{b}$.
    \end{definition}
    
    
    \begin{remarque}
        \begin{itemize}
            \item Le nombre $\frac{a}{b}$ est le résultat de la division de $a$ par $b$.
            \item Lorsque les nombres $a$ et $b$ sont deux nombres entiers ($b$ non nul), on dit que $\frac{a}{b}$ est une \textbf{fraction}.
        \end{itemize}
    \end{remarque}
    
    
    \begin{exemple}
        Compléter les égalités suivantes pour qu'elles soient vraies.\medskip

        \begin{minipage}{0.49\linewidth}
            \begin{itemize}
                \item $4 \times \points[2] = 28$ \medskip
                \item $\points[2] \times 7 = 2$
            \end{itemize}
        \end{minipage}
        \hfill
        \begin{minipage}{0.49\linewidth}
            \begin{itemize}
                \item $3 \times \points[2] = 14$ \medskip
                \item $10 \times \points[2] = 5$
            \end{itemize}
        \end{minipage}
    \end{exemple}
    
    
    \begin{definition}[Simplification]
        \textbf{Simplifier une fraction}, c'est l'écrire avec les plus petits nombres entiers possibles au numérateur et au dénominateur.
    \end{definition}
        
        
    \begin{exemple}

\section{Additionner et soustraire}
    
    \begin{propriete}
        Pour \textbf{additionner} (ou \textbf{soustraire}) deux fractions:
        \begin{itemize}
            \item On réduit les nombres au \textbf{même dénominateur}.
            \item On additionne (ou on soustrait) les numérateurs et on garde le dénominateur commun.
        \end{itemize}
    \end{propriete}

\section{Multiplier}
\subsection{Cas général}
    
    \begin{propriete}
        Pour \textbf{multiplier} des fractions, on multiplie les numérateurs entre eux et les dénominateurs entre eux.
    \end{propriete}

    
    \begin{exemple}
        Calculer les expressions suivantes. Donner les résultats sous forme simplifiée.\medskip

        \begin{minipage}{0.24\linewidth}
            $A = \frac{4}{5} \times \frac{3}{7}$

            \vspace{4cm}
        \end{minipage}
        \hfill\vline \hfill
        \begin{minipage}{0.24\linewidth}
            $B = \frac{4}{9} \times \frac{5}{3}$

            \vspace{4cm}
        \end{minipage}
        \hfill\vline \hfill
        \begin{minipage}{0.24\linewidth}

\subsection{Simplifier avant de calculer}
    
    \begin{remarque}
        Lorsqu'on doit multiplier deux fractions avec des nombres entiers un peu grands, on commence par décomposer ces nombres sous forme de produits et on cherche des simplifications avant de se lancer dans les multiplications.
    \end{remarque}
    
    
    \begin{exemple}
        Effectuer les opérations suivantes en simplifiant dès le départ.\medskip
