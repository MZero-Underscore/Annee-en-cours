        \begin{enumerate}[leftmargin=*]
            \item Compléter la phrase suivante :

            Dans une suite de calculs,
            \begin{itemize}
                \item on commence par effectuer \textcolor{red}{les calculs entre parenthèses;}

                \medskip
                
                \item ensuite, on effectue \textcolor{red}{les multiplications et les divisions dans le sens de lecture;}

                \medskip
                \item pour terminer, on effectue \textcolor{red}{les additions et les soustractions de gauche à droite.}
            \end{itemize}

            \medskip

            \item Donner la définition du périmètre d'une figure.

            \medskip
            
            Le périmètre d'une figure est \textcolor{red}{la longueur de son contour.}

            \item Compléter avec la bonne formule.

            \medskip
            
            \begin{minipage}{0.5\linewidth}
                \begin{itemize}
                    \item Périmètre d'un rectangle : \textcolor{red}{$2\times$ longueur +$2\times$ Largeur}

                    \medskip 
                    
                    \item Périmètre d'un cercle : \textcolor{red}{$2\times\pi\times$rayon}
                \end{itemize}
            \end{minipage}
            \hfill
            \vline
            \hfill
            \begin{minipage}{0.49\linewidth}
                \begin{itemize}
                    \item Aire d'un carré : \textcolor{red}{côté$\times$côté}

                    \medskip 
                    
                    \item Aire d'un disque : \textcolor{red}{$\pi\times$rayon$^2$}
                \end{itemize}
            \end{minipage}

        \end{enumerate}

                \smallskip
    
                \item Calculer l'aire de la surface de gazon qui sera arrosée par les deux canons à eau. (Arrondir au m$^2$)

                \textcolor{red}{Chaque canon arrose une surface qui est un quart de disque de rayon 15 m.}\\
                \textcolor{red}{Si on superpose les 2 quarts de disques, on obtient la surface du carré + la surface arrosée 2 fois (en orange)}\\
                \textcolor{red}{On calcule alors l'aire arrosée par les deux quarts de disque puis celle du carré.}\\
                \textcolor{red}{Il restera à faire la soustraction pour avoir l'aire demandée.}

                 \medskip

                \textcolor{red}{Aire des quarts de disques = $(\pi\times15^2)\div4\times2\simeq 353$ m$^2$ \\ Aire du carré = $15\times15=225$ m$^2$}

                \textcolor{red}{On trouve alors Aire demandée $\simeq 353-225=128$ m$^2$.}
                
            \end{enumerate}
