\section{Rappels}

    \begin{definition}[Proportionnalité]
        Deux grandeurs sont \textbf{proportionnelles} si l'on peut passer de l'une à l'autre en multipliant par un même nombre non nul, appelé le \textbf{coefficient de proportionnalité}.
    \end{definition}

    
    \begin{exemple}
        On considère le tableau suivant représentant le prix payé en fonction de la masse de fruits achetés :
        \begin{center}
        \renewcommand{\arraystretch}{1.5}
        \begin{tabular}{|l|c|c|c|}
        \hline
        Masse (en kg) & 2 & 3 & 5 \\ \hline
        Prix (en \euro) & 4,40 & 6,60 & 11 \\ \hline
        \end{tabular}
        \end{center}

\section{Échelle}
    \begin{definition}[Échelle]
        L'\textbf{échelle} d'une carte ou d'un plan est le coefficient de proportionnalité qui permet de passer des distances réelles aux distances correspondantes sur le plan.
        \[ \text{Échelle} = \frac{\text{Distance sur le plan (ou la carte)}}{\text{Distance réelle}} \]
    \end{definition}


    \begin{remarque}
        Pour appliquer cette formule, les deux distances doivent obligatoirement être exprimées dans la \textbf{même unité} (le plus souvent en centimètres). \\
        Par exemple, une carte est à l'échelle $\dfrac{1}{25\,000}$. Cela signifie que $1$ cm sur la carte représente $25\,000$ cm (soit $250$ m) dans la réalité.
    \end{remarque}


    \begin{exemple}
        \textbf{Calculer une distance réelle} \\[0.1cm]

\section{Pourcentages}
    \begin{propriete}[Appliquer un pourcentage]
        Appliquer un pourcentage de $x\,\%$ à une quantité revient à multiplier cette quantité par la fraction $\dfrac{x}{100}$.
    \end{propriete}

    \begin{exemple}
        Un jeu de société coûte $45$ \euro. Le magasin accorde une remise immédiate de $20\,\%$. Calculer le montant de la réduction puis le nouveau prix du jeu de société.

    \begin{propriete}[Calculer un pourcentage]
        Calculer un pourcentage revient à trouver une proportion ou une fréquence, puis à l'exprimer sous la forme d'une fraction dont le dénominateur est égal à $100$.
    \end{propriete}
