\section{Généralités}

    Une \textbf{fraction} est un nombre qui s'écrit sous la forme $\boxed{\dfrac{\text{un nombre entier }}{\text{un nombre entier} (\neq 0)}}$.

    \medskip

    On peut les utiliser dans plusieurs situations.
    \begin{itemize}
        \item Pour représenter un partage équitable d'une unité;
        \item pour représenter une proportion; 
        \item en tant que nombre;
        \item \ldots
    \end{itemize}


    \begin{definition}
        Une \textbf{fraction} se note $\dfrac{a}{b}$ où $a$ et $b$ sont deux nombres entiers $(b\neq 0)$.

        Le nombre $a$ s'appelle le \textbf{numérateur} de la fraction.\\
        Le nombre $b$ s'appelle le \textbf{dénominateur} de la fraction. Il ne peut pas être nul\\
    \end{definition}

\section{Représenter une proportion}
\subsection{Représenter une proportion}
    Il est fréquent d'utiliser les fractions pour représenter une proportion.

    Par exemple, si dans une classe de 30 élèves, il y a 18 filles, on peut dire qu'il y a une proportion de $\dfrac{18}{30}$ (dix-huit-trentièmes) de filles dans la classe.

    \smallskip

    \begin{exemple}
        \begin{enumerate}
            \item Un restaurant propose à ses clients un plat du jour. Parmi les 72 clients qui ont déjeuné hier dans ce restaurant, 24 ont pris le plat du jour. 
            
            Quelle est la proportion de clients ayant pris le plat du jour ?

            \medskip

            \points[40]
            

            \item Un fleuriste propose des bouquets composés de huit tulipes et cinq roses et des bouquets composés de sept marguerites et quatre roses. 
            
            Quelle est la proportion de roses dans chacun de ces bouquets ?
            
            \medskip

            \points[40]\points[40]
        \end{enumerate}        
    \end{exemple}
    

\subsection{Calculer la fraction d'une quantité}
    Dans les exercices, une fois une proportion exprimée, on demande régulièrement de donner la quantité représentée par cette proportion. Voici un exemple :

    Dans un collège de 450 élèves, les $\dfrac{2}{5}$ sont demi-pensionnaires. 

    Combien d'élèves sont demi-pensionnaires ? 

    On dit que l'on va \textbf{calculer la fraction d'une quantité}, ici les $\dfrac{2}{5}$ de 450.
        

    \begin{propriete}
        Pour \textbf{calculer la fraction d'une quantité,} on \textbf{multiplie la fraction par la quantité}.

        Il y a trois méthodes de calcul pour aboutir au résultat. Il n'y en a pas une plus efficace que les autres.
    \end{propriete}


    \begin{methode}
        On souhaite calculer les $\dfrac{9}{4}$ de 12, c'est à dire $\dfrac{9}{4}\times12$.

        \medskip

        \begin{minipage}{0.32\linewidth}
            \underline{Méthode 1}:
            \begin{itemize}[leftmargin=*]
                \item On compte 9 fois la valeur 12
                \item On sépare le résultat en 4 parties égales.
            \end{itemize}

            \vspace{2cm}
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.32\linewidth}
            \underline{Méthode 2}:
            \begin{itemize}[leftmargin=*]
                \item On sépare 12 en 4 parties égales.
                \item On compte 9 fois la valeur trouvée.
            \end{itemize}

            \vspace{2cm}
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.32\linewidth}
            \underline{Méthode 3}:
            \begin{itemize}[leftmargin=*]
                \item On sépare 9 en 4 parties égales.
                \item On multiplie le résultat par 12.

            \end{itemize}

            \vspace{2cm}
        \end{minipage}
    \end{methode}
    

\section{Représenter un quotient}
    \begin{definition}
        Soit $a$ et $b$ deux nombres entiers $(b\neq 0)$. 
        Le nombre qui, multiplié par $b$ donne $a$ est le quotient $\dfrac{a}{b}$.

        \[b\times\boxed{\dfrac{a}{b}}=a\]
    \end{definition}


\newpage


    \begin{exemple}
        Compléter les égalités suivantes.

\section{Manipuler des fractions}
\subsection{Fractions égales}
    \begin{propriete}
        \underline{On ne change pas} la valeur d'une fraction \textbf{en multipliant ou divisant} son numérateur \textbf{et} son dénominateur par un \textbf{même nombre} non nul.
    \end{propriete}

    
    \begin{exemple}
        Compléter les égalités suivantes.

        \medskip
        
        \begin{tabular}{c|c|c|c|c}
            $\dfrac{10}{7}\quad \quad =\dfrac{}{21}$ 
            & $\dfrac{45}{99} \quad \quad = \dfrac{5}{}$ 
            & $\dfrac{9}{7} \quad \quad = \dfrac{72}{}$ 
            & $\dfrac{120}{48} \quad \quad = \dfrac{}{4}$
            & $\dfrac{18}{12} \quad \quad = \dfrac{}{2}$
        \end{tabular}   
    \end{exemple}

\subsection{Comparer des fractions : Les cas particuliers}

    \begin{propriete}
        Soit $a$ et $b$ deux nombres entiers non nuls. On considère la fraction $\dfrac{a}{b}$.
        \begin{itemize}
            \item Si $a>b$ alors $\dfrac{a}{b}>1$.
            \item Si $a<b$ alors $\dfrac{a}{b}<1$.
            \item Si $a=b$ alors $\dfrac{a}{b}=1$.
        \end{itemize}
    \end{propriete}


    \begin{exemple}
        Comparer les fractions suivantes à 1.

        $\dfrac{4}{7}; \dfrac{9}{11}; \dfrac{13}{7}$ et $\dfrac{45}{45}$.

\subsection{Ranger plusieurs fractions par ordre croissant ou décroissant}
    \begin{methode}
        \begin{itemize}[leftmargin=*]
            \item Pour ranger des fractions par ordre croissant ou décroissant, on commence généralement par les réduire au même dénominateur.

            \item On peut également se servir de la comparaison à 1 pour limiter les calculs.
        \end{itemize}
    \end{methode}


    \begin{exemple}
        Dans les deux cas suivants, ranger les fractions dans l'ordre croissant.
        \begin{enumerate}[leftmargin=*]
            \item $\dfrac{2}{3}; \dfrac{5}{6}; \dfrac{1}{6}; \dfrac{7}{12}; \dfrac{3}{4}; \dfrac{4}{3}; \dfrac{13}{6}; \dfrac{5}{3}$

            \medskip

            \item $\dfrac{4}{3}; \dfrac{11}{6}; \dfrac{4}{13}; \dfrac{17}{12}; \dfrac{7}{6}; \dfrac{7}{13}$
        \end{enumerate}
    \end{exemple}
