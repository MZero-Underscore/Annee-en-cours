\section{L'inégalité triangulaire}
\subsection{Cas général}
    \begin{propriete}[Inégalité triangulaire]
        \begin{minipage}{0.6\linewidth}
            Dans un triangle, la longueur de chaque côté est \textbf{strictement inférieure} à la somme des  longueurs des deux autres côtés.

            \bigskip

            \[\boxed{AB < AC+CB} \quad \boxed{BC < BA+AC} \quad \boxed{CA < CB+BA}\]
        \end{minipage}
        \hfill
        \begin{minipage}{0.39\linewidth}
            \begin{center}
                \includegraphics[scale=0.25]{images/IllustrationTriangle.png}
            \end{center}
        \end{minipage}
    \end{propriete}


    \begin{remarque}
        Cette propriété traduit le fait que \og dans un plan, le chemin le plus court entre deux points est la ligne droite\fg.
    \end{remarque}

\subsection{Un critère pour déterminer si trois poins sont alignés.}
    \begin{propriete}
        Soit A, B et M trois points.

        \begin{itemize}
            \item \textbf{Si} $AM+MB=AB$ \textbf{alors} ces trois points sont alignés et $M \in [AB]$.

            \item \underline{Réciproquement}, \textbf{si} M appartient au segment [AB], \textbf{alors} $AM+MB=AB$.
        \end{itemize}
    \end{propriete}


    \begin{remarque}
        La réciproque \underline{\textbf{ne dit pas}} que le point M est le milieu du segment [AB].
    \end{remarque}

\subsection{Les médiatrices}
\subsubsection{Médiatrice d'un segment}


    \begin{definition}
        La $\textbf{médiatrice d'un segment}$ est la droite qui coupe ce segment en son milieu \underline{et} en formant un angle droit.
    \end{definition}


    \begin{exemple}
        \begin{enumerate}
            \item Tracer un segment [AB] de longueur 3 cm, puis construire sa médiatrice à l'aide d'une règle graduée et d'une équerre.

            \item Placer un point P sur cette médiatrice telle que AP = 2 cm à l'aide du compas. Quelle semble être la longueur BP ? 


            \vspace{2.5cm}
        \end{enumerate}
    \end{exemple}


    \begin{propriete}
        \begin{itemize}
            \item \textbf{Si} un point appartient à la médiatrice d'un segment \textbf{alors} il est équidistant de ses extrémités.

            \item \underline{Réciproquement}, \textbf{si} un point est équidistant des extrémités d'un segment \textbf{alors} il appartient à la médiatrice du segment.
        \end{itemize}
    \end{propriete}


    \begin{exemple}
         \begin{enumerate}
             \item Construire un segment [AB] de longueur quelconque puis, à l'aide de la propriété 1 et du compas, construire sa médiatrice qu'on appellera $d$.

            \item Placer un point C sur la médiatrice $d$. Quelle est la nature du triangle ABC ? Justifier.
         \end{enumerate}
    \end{exemple}

\subsection{Les hauteurs}
    \begin{definition}
        \begin{minipage}{0.5\linewidth}
            Dans un triangle, une \textbf{hauteur} est une droite qui passe par un sommet et qui est perpendiculaire au côté opposé à ce sommet.
