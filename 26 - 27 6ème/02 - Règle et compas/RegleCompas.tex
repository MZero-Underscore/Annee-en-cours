\section{Point, droite, demi-droite et segment}
\subsection{Le point}
    \begin{definition}[Point]
        \textbf{Un point} désigne un emplacement, il n'a aucune dimension (pas de longueur, pas d'aire...).\\
        Au collège, on écrit toujours le nom d'un point en lettre majuscule. On représente usuellement un point par une croix.
    \end{definition}

    \begin{exemple}
        Voici trois illustration.
        \begin{center}
            \includegraphics[scale=0.5]{images/points.png}
        \end{center}
    \end{exemple}
    

\subsection{La droite}
    \begin{definition}[Droite]
        \textbf{Une droite} est une ligne rectiligne illimitée des deux côtés. Elle se trace avec une règle.        
    \end{definition}


    \begin{notation}
        On trouve trois notations pour une droite :
        \begin{itemize}
            \item La droite $d$ ou $(d)$: la lettre d ne désigne aucun point, c'est juste le nom de la droite.
            \begin{center}
                \includegraphics[scale=0.5]{images/droite1.png}
            \end{center}
            
            \item La droite (AB) ou (BA) : les lettres A et B représentent deux points de la droite. On s'en sert pour nommer la droite.
            \begin{center}
                \includegraphics[scale=0.5]{images/droite2.png}
            \end{center}
            
            \item La droite (xy) ou (yx) : ici, les lettres minuscules x et y désignent chacune une des extrémités de la partie de la droite qui est dessinée et ne sont pas des points.
            \begin{center}
                \includegraphics[scale=0.5]{images/droite3.png}
            \end{center}
        \end{itemize}
    \end{notation}


    \begin{propriete}
        Par deux points distincts, il passe une unique droite.
    \end{propriete}
    

\subsection{Le symbole \og appartient à \fg}
    \begin{notation}
        Quand on ajoute un point sur une figure ou que l'on veut décrire la situation d'un point, on utilise deux symboles particuliers. Prenons exemple sur la figure suivante :
        \begin{center}
            \includegraphics[scale=0.5]{images/SymboleAppartient.png}
        \end{center}
        \begin{itemize}
            \item Le point M est sur la droite $d$. On dit que le point M appartient à la droite (d). On note $M \in (d)$.
            \item Le point N n'est pas sur la droite $d$. On dit que le point N n'appartient pas à la droite (d). On note $N \notin (d)$.        
        \end{itemize}
    \end{notation}

\subsection{La demi-droite}
    \begin{definition}[Demi-droite]
        Une \textbf{demi-droite} est une portion de droite limitée d'un seul côté par un point qu'on appelle \textbf{son origine}.
    \end{definition}
      

    \begin{notation}
        On trouve deux notations pour une demi-droite :
        \begin{itemize}
            \item La demi-droite $[AB)$ : l'origine est le point A et le point B est un autre point quelconque.
            \begin{center}
                \includegraphics[scale=0.5]{images/DemiDroite1.png}
            \end{center}
            
            \item La demi-droite $[Ax)$ : ici, la lettre x désigne l'extrémité de la demi-droite qui est illimitée.
            \begin{center}
                \includegraphics[scale=0.5]{images/DemiDroite2.png}
            \end{center}
        \end{itemize}
    \end{notation}

\subsection{Le segment}
    \begin{definition}[Segment]
        Un \textbf{segment} est une portion de droite délimitée par deux points. Ces points s'appellent les \textbf{extrémités} du segment.

        \begin{center}
            \includegraphics[scale=0.4]{images/Segment.png}
        \end{center}
    \end{definition}


    \begin{notation}
        Le segment d'extrémités A et B se note [AB] ou [BA]. 
        On peut mesurer un segment (à l'aide d'une règle par exemple). La longueur du segment [AB] se note AB.
    \end{notation}

\section{Distance entre deux points}
    \begin{propriete}
        Le segment [AB] est le plus court chemin allant du point A au point B.
    \end{propriete}

    
    \begin{definition}[Distance entre deux points]
        La \textbf{distance entre deux points} A et B est la \textbf{longueur du segment} [AB]. Cette longueur est notée AB. 
    \end{definition}


    \begin{propriete}[Inégalité triangulaire]
        \begin{minipage}{0.5\linewidth}
            Soit A, B et C trois points quelconques du plan.
            \begin{itemize}
                \item On a toujours l'inégalité $AB+BC \geq AC$.
    
                \item De plus, $AC+CB=AB$ si, et seulement si, $C\in[AB]$.
            \end{itemize}
        \end{minipage}
        \hfill
        \begin{minipage}{0.49\linewidth}
            \begin{center}
                \includegraphics[scale=0.5]{images/InegalitéTriangulaire.png}
            \end{center}
        \end{minipage}        
    \end{propriete}


    
    \begin{exemple}


        \begin{minipage}{0.5\linewidth}
            On considère la figure ci-contre.

            \begin{enumerate}
                \item Compléter.

                AC = \points[2] cm \quad CB = \points[2] cm \\
                AD = \points[2] cm \quad DB = \points[2] cm \\

                \item Compléter avec le bon symbole parmi <, > ou =.
                
                $AC+CB \points[2] AB$ \quad\quad $AD+DB \points[2] AB$
            \end{enumerate}
        \end{minipage}

\section{Cercle et disque}
\subsection{Le cercle}

    \begin{definition}[Cercle]
        Un \textbf{cercle} est l'ensemble de tous les points situés à une \textbf{\underline{même distance}} d'un même point appelé \textbf{le centre}. Cette distance est appelée le \textbf{rayon} du cercle.

        \begin{minipage}{0.7\linewidth}
            \begin{itemize}
                \item Un \textbf{rayon} est un segment reliant le centre à un point du cercle.
    
                \item Un \textbf{diamètre} est un segment reliant deux points du cercle et passant par le centre du cercle.
    
                \item Une \textbf{corde} est un segment reliant deux points distincts du cercle.
            \end{itemize}
        \end{minipage}
        \hfill
        \begin{minipage}{0.29\linewidth}
            \begin{center}
                \includegraphics[scale=0.6]{images/Cercle.png}
            \end{center}
        \end{minipage}
    \end{definition}


    \begin{remarque}
        Les mots \og rayon \fg et \og diamètre \fg désignent à la fois un segment et une longueur.
    \end{remarque}


    \begin{exemple}
        \begin{enumerate}
            \item Construire un cercle $\mathcal{C}$ de centre F et de rayon 6cm.
            \item Tracer en rouge un rayon de ce cercle.
            \item Tracer en vert un diamètre de ce cercle puis tracer en noir une corde qui n'est pas un diamètre de ce cercle.
        \end{enumerate}        
    \end{exemple}

\subsection{Le disque}
    \begin{definition}[Disque]
        \begin{minipage}{0.5\linewidth}
            Le \textbf{disque de centre O et de rayon r} est l'\textbf{ensemble de tous les points situés à une distance de O inférieure ou égale à r}. Il s'agit de la surface délimitée par le cercle de même centre et de même rayon.
