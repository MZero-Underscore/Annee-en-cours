                donc $27 + \frac{2}{100} < 27,2$

                $52 \text{ dixièmes et } 35,2$

                \medskip

                \medskip

                donc $52 \text{ dixièmes} < 35,2$
            \end{minipage}
            \hfill \vline \hfill

                $\frac{307}{10} + \frac{1}{100} \text{ et } 30 + \frac{71}{100}$
    
                \medskip 

                $\frac{307}{10} + \frac{1}{100}=30,71$

                $ 30 + \frac{71}{100}=30,71$

                \medskip

    \begin{exercice}
        Les deux parties sont indépendantes.

        \begin{enumerate}[leftmargin=*]
            \item Nathalie possède un billet de 5€. Elle achète une boisson à 1,50€, un cookie à 1,25€ et 1 pain au chocolat à 85 centimes. 
            
            Combien lui reste-t-il après avoir payé ?

            \begin{ZoneReponse}{4cm}  
            
                1,50+1,25+0,85=3,6\\ \\ 
                Les articles coûtent 3,60 €.\\ \\ 
                5-3,60=1,40\\ \\ 
                Il lui reste 1,40 € après avoir payé.  
            \end{ZoneReponse}
         
            \item J’ai deux crayons. Le noir mesure 2,5 cm de moins que le vert. Le crayon noir mesure 17,8 cm. 
            
            Quelle est la longueur du crayon vert ?

             \begin{ZoneReponse}{4cm}  
            
                Il faut remarquer que le crayon noir est le plus court des deux.\\ \\ 
                17,80+2,50=20,30\\ \\ 
                Le crayon vert mesure 20,30 cm\\          
            \end{ZoneReponse}
        \end{enumerate}
        
    \end{exercice}
