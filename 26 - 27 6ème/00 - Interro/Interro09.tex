            \item \textbf{Angles supplémentaires :} \color{blue} Deux angles sont dits supplémentaires si la somme de leurs mesures est égale à $180^{\circ}$.
            
            \item \textbf{Angles adjacents :} \color{blue} Deux angles sont adjacents s'ils ont un sommet commun, un côté commun et s'ils sont situés de part et d'autre de ce côté commun.
            
            \item \textbf{Angles opposés par le sommet :} \color{blue} Deux angles sont opposés par le sommet s'ils ont un sommet commun et si leurs côtés sont dans le prolongement l'un de l'autre. Ils ont alors la même mesure.
        \end{enumerate}
    \end{exerciceNote}

    \begin{exerciceNote}{3}
        1. Construire les angles suivants en utilisant le segment $[AB]$ comme côté : \\
        $\widehat{BAC} = 55^{\circ}$ \quad $\widehat{ABD} = 140^{\circ}$

        2. Construire en vert la bissectrice de l'angle $\widehat{ABD}$.
        
        \textbf{Tracé de la correction :}
        \begin{center}
        \begin{tikzpicture}[scale=1]
            % 1. Segment de base
            \draw[thick] (0,0) coordinate(A) node[left]{$A$} -- (5,0) coordinate(B) node[right]{$B$};
            
            % 2. Angle BAC = 55° (Sommet A, direction 55°)
            \draw[blue, thick] (A) -- (55:4.5) coordinate(C) node[above]{$C$};
            \draw[red, thick] (0.8,0) arc (0:55:0.8) node[midway, right, xshift=2pt] {$55^\circ$};
            
            % 3. Angle ABD = 140° (Sommet B, direction 180-140 = 40°)
            % On part de B(5,0) et on trace vers le haut-droite pour que l'angle avec [BA] soit de 140°
            \draw[blue, thick] (B) -- ++(40:4.5) coordinate(D) node[above]{$D$};
            \draw[red, thick] (4.2,0) arc (180:40:0.8) node[midway, above left] {$140^\circ$};
            
            % 4. Bissectrice de l'angle ABD (Vert)
            % La bissectrice est au milieu de 180° (vers A) et 40° (vers D)
            % (180 + 40) / 2 = 110°
            \draw[ForestGreen, thick, dashed] (B) -- ++(110:5) node[above]{bissectrice};
            
            % Codage de la bissectrice (deux arcs simples)
            \draw[ForestGreen] (4.4,0) arc (180:110:0.6) node[midway, left, yshift=2pt] {$\times$};
            \draw[ForestGreen] (4.4,0) arc (110:40:0.6) node[midway, right, yshift=3pt] {$\times$};
        \end{tikzpicture}
        \end{center}
        \color{blue}
        \textbf{Note :} L'angle $\widehat{ABD}$ est bien obtus. La bissectrice partage l'angle de $140^{\circ}$ en deux angles égaux de $70^{\circ}$.
    \end{exerciceNote}
    

     % --- EXERCICE 4 : CALCULS ---
    \begin{exerciceNote}{2}
        \color{blue}
        Les points $A, O, B$ sont alignés, donc $\widehat{AOB}$ est un angle plat qui mesure $180^{\circ}$.\\
        Les angles $\widehat{AOC}$ et $\widehat{COB}$ sont donc supplémentaires.\\
        $\widehat{COB} = 180^{\circ} - \widehat{AOC} = 180^{\circ} - 42^{\circ} = \mathbf{138^{\circ}}$.
    \end{exerciceNote}
