    \begin{exercice}
        \begin{enumerate}
            \item Comment se lit le nombre 4\,705 ?
        
            \item Comment s'écrit le nombre soixante-dix-mille-huit-cent-deux en chiffres ?

            \points[40]
        
            \item À quel nombre correspond la décomposition suivante ?\\
            50\,000 + 9\,000 + 600 + 80 + 4

            \points[40]
        
            \item À quel nombre correspond la décomposition suivante ?\\
            65 milliers + 47 dizaines + 3 unités
            
            \points[40]

            \item Quel est le plus petit nombre entier à quatre chiffres différents ?
            
            \points[40]

            \item Quel est le plus grand nombre entier à six chiffres différents ?

            \points[40]
        
            \item Quels nombres entiers peuvent être intercalés entre les nombres 995 et \np{1005} ?

            \points[40]
        
            \item Donner deux nombres entiers permettant d'encadrer le nombre \np{654209}.
            
            \points[40]
        \end{enumerate}
    \end{exercice}
