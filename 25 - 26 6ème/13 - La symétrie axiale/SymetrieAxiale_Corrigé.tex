\section{Définition et constructions}

    \begin{definition}
        Deux figures sont \textbf{symétriques par rapport à une droite $(d)$} si elles se superposent exactement par pliage le long de cette droite. La droite $(d)$ est l'axe de symétrie.
    \end{definition}
    
    
    \begin{definition}
        Le symétrique d'un point $M$ par rapport à une droite $(d)$ est le point $M'$ tel que :
        \begin{itemize}
            \item Si $M$ n'est pas sur $(d)$, alors $(d)$ est la \textbf{médiatrice} du segment $[MM']$.
            \item Si $M$ est sur $(d)$, alors son symétrique est lui-même ($M = M'$).
        \end{itemize}
    \end{definition}
    
    
    \begin{center}
        \begin{tikzpicture}[scale=0.8]
            \draw[thick, red] (0,-1.5) -- (0,1.5) node[above] {$(d)$};
            \coordinate (M) at (-2, 0.5);
            \coordinate (Mp) at (2, 0.5);
            \draw[dashed, gray] (M) -- (Mp);
            \node at (M) {$\times$}; \node[left] at (M) {$M$};
            \node at (Mp) {$\times$}; \node[right] at (Mp) {$M'$};
            % Codage médiatrice
            \draw[blue, thick] (0, 0.5) -- (0, 0.7) -- (-0.2, 0.7) -- (-0.2, 0.5);
            \node[blue] at (-1, 0.5) {$/$};
            \node[blue] at (1, 0.5) {$/$};
        \end{tikzpicture}
    \end{center}

\section{Propriétés de la symétrie axiale}
    
    \begin{propriete}
        La symétrie axiale conserve :
        \begin{itemize}
            \item \textbf{Les longueurs} (le symétrique d'un segment est un segment de même longueur).
            \item \textbf{L'alignement} (les symétriques de points alignés sont alignés).
            \item \textbf{Les angles} et \textbf{les aires}.
        \end{itemize}
    \end{propriete}
    
    
    \begin{methodeTitre}{Exemple 4 : Utiliser les propriétés}
        $EFGH$ est le symétrique de $ABCD$ par rapport à une droite. Si $AB = 5$ cm, quelle est la longueur de $[EF]$ ? Justifie.
        \begin{ZoneReponse}{3cm}
            \color{ForestGreen}
            Puisque la symétrie axiale conserve les longueurs, le symétrique du segment $[AB]$ est le segment $[EF]$ de même longueur. \\
            On a donc : $EF = AB = 5\text{ cm}$.
        \end{ZoneReponse}
    \end{methodeTitre}
    
    
\section{Cas particuliers et figures complexes}
    
    \subsection{Symétrique d'un cercle}
    \begin{propriete}
        Le symétrique d'un cercle de centre $O$ et de rayon $r$ est un cercle de même rayon $r$, dont le centre $O'$ est le symétrique de $O$.
    \end{propriete}

        \begin{tikzpicture}[scale=0.6]
            \draw[red, thick] (0,-2.5) -- (0,2.5) node[above] {$(d)$};

    \end{center}

        Pour tracer le symétrique d'une figure complexe, on trace les symétriques de ses points stratégiques (sommets, centres de cercles).

\section{Rechercher des axes de symétrie}

    \begin{definition}
        On dit qu'une droite est un \textbf{axe de symétrie d'une figure} si, après pliage le long de cette droite, les deux parties de la figure se superposent parfaitement.
    \end{definition}
    
    \begin{exemple}[Reconnaître un axe de symétrie (l'exemple de la banane)]
        Pour savoir si une figure possède un axe de symétrie, on imagine le pliage.
        \begin{itemize}
            \item \textbf{Une figure symétrique :} Un masque ou un cœur possède un axe de symétrie. Les deux moitiés se superposent parfaitement.
            \item \textbf{Une figure asymétrique :} Une \textbf{banane} est courbée sur le côté. Elle ne possède \textbf{aucun axe de symétrie} car si on la plie, la pointe ne tombe pas sur l'autre extrémité !
        \end{itemize}
        
        \begin{center}
            \begin{tikzpicture}[scale=0.6]
                % Figure 1 : Le masque/coeur (avec axe)
                \begin{scope}[shift={(-4,0)}]
                    \draw[thick, fill=purple!10] (0,-2) .. controls (-2,-0.5) and (-2,1.5) .. (0,0.5) .. controls (2,1.5) and (2,-0.5) .. (0,-2);
                    \draw[dashed, red, very thick] (0,-2.5) -- (0,2) node[above] {Axe de symétrie};
                    \node[below] at (0,-2.5) {1 axe de symétrie};
                \end{scope}
                
                % Figure 2 : La banane
                \begin{scope}[shift={(4,0)}]
                    \draw[very thick, fill=yellow!50] (-1.5,-1) to[bend right=40] (1.5,1) to[bend left=50] (1.3,1.3) to[bend left=30] (-1.7,-0.7) -- cycle;
                    \draw[thick, brown] (-1.5,-1) to[bend right=35] (1.4,1.1);
                    \node[below, font=\bfseries] at (0,-2.5) {La banane};
                    \node[below=0.5cm, color=red!80!black] at (0,-2.5) {0 axe de symétrie (Asymétrique)};
                \end{scope}
            \end{tikzpicture}
        \end{center}
    \end{exemple}
    
    
    \begin{propriete}
        Certaines figures géométriques possèdent des axes de symétrie particuliers :
        \begin{itemize}
            \item Le \textbf{rectangle} possède 2 axes de symétrie (ses médiatrices).
            \item Le \textbf{losange} possède 2 axes de symétrie (ses diagonales).
            \item Le \textbf{carré} possède 4 axes de symétrie (médiatrices et diagonales).
            \item Le \textbf{cercle} possède une infinité d'axes de symétrie (tous ses diamètres).
        \end{itemize}
    \end{propriete}
    
    
    \begin{exemple}[Tracer des axes]
        Trace, en rouge, tous les axes de symétrie possibles pour les figures suivantes :
        \begin{ZoneReponse}{6cm}
        \begin{center}
        \begin{tikzpicture}[scale=1]
            % Un triangle isocèle
            \draw[thick] (-5,0) -- (-3,0) -- (-4,2) -- cycle;
            \node at (-4, -0.5) {Triangle isocèle};
            
            % Une croix
            \draw[thick] (0,0) -- (1,0) -- (1,1) -- (2,1) -- (2,2) -- (1,2) -- (1,3) -- (0,3) -- (0,2) -- (-1,2) -- (-1,1) -- (0,1) -- cycle;
            \node at (0.5, -0.5) {Croix};
            
            % Un hexagone régulier
            \begin{scope}[shift={(5,1)}]
                \foreach \a in {0,60,...,300} { \draw[thick] (\a:1.5) -- (\a+60:1.5); }
            \end{scope}
            \node at (5, -0.5) {Hexagone régulier};
            
            % --- CORRECTION ÉLÈVE (Tracés imposés en rouge par la consigne) ---
            % 1. Axe du triangle isocèle
            \draw[red, dashed, very thick] (-4, -0.3) -- (-4, 2.3);
            
            % 2. Axes de la croix (4 axes)
            \draw[red, dashed, very thick] (0.5, -0.3) -- (0.5, 3.3);
            \draw[red, dashed, very thick] (-1.3, 1.5) -- (2.3, 1.5);
            \draw[red, dashed, very thick] (-1, 0) -- (2, 3);
            \draw[red, dashed, very thick] (-1, 3) -- (2, 0);
            
            % 3. Axes de l'hexagone régulier (6 axes)
            \begin{scope}[shift={(5,1)}]
                \draw[red, dashed, very thick] (-1.8, 0) -- (1.8, 0);
                \draw[red, dashed, very thick] (60:1.8) -- (240:1.8);
                \draw[red, dashed, very thick] (120:1.8) -- (300:1.8);
                \draw[red, dashed, very thick] (30:1.8) -- (210:1.8);
                \draw[red, dashed, very thick] (90:1.8) -- (270:1.8);
                \draw[red, dashed, very thick] (150:1.8) -- (330:1.8);
            \end{scope}
        \end{tikzpicture}
        \end{center}
        \end{ZoneReponse}
    \end{exemple}
    
    
    \begin{methodeTitre}{Astuce pour trouver l'axe}
        Pour trouver l'axe de symétrie entre deux points $A$ et $A'$, il suffit de tracer la médiatrice du segment $[AA']$. Pour une figure, on peut essayer de faire de même avec deux sommets correspondants.
    \end{methodeTitre}
