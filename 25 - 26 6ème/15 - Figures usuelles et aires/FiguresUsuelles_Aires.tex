\subsection{Les quadrilatères}

    \begin{definition}
        \begin{itemize}
            \item \textbf{Rectangle} : C'est un quadrilatère qui possède 4 angles droits.
            \item \textbf{Losange} : C'est un quadrilatère qui possède 4 côtés de même longueur.
            \item \textbf{Carré} : C'est un quadrilatère qui possède 4 angles droits et 4 côtés de même longueur.
        \end{itemize}
    \end{definition}

\subsection{Les triangles particuliers}

    \begin{definition}
        \begin{itemize}
            \item \textbf{Triangle rectangle} : C'est un triangle qui possède un angle droit. Le plus grand côté (en face de l'angle droit) s'appelle l'hypoténuse.
            \item \textbf{Triangle isocèle} : C'est un triangle qui possède au moins deux côtés de même longueur.
            \item \textbf{Triangle équilatéral} : C'est un triangle qui possède ses trois côtés de même longueur.
            \item \textbf{Triangle rectangle isocèle} : C'est un triangle qui possède un angle droit et deux côtés de même longueur.
        \end{itemize}
    \end{definition}

    \vspace{0.5cm}

\subsection{Constructions de figures}

    \begin{exercice}
        Effectuer les constructions géométriques suivantes :
        \begin{enumerate}
            \item \textbf{Le rectangle :} Tracer un rectangle $ABCD$ tel que $AB = 6 \text{ cm}$ et $BC = 3 \text{ cm}$.
            \item \textbf{Le losange :} Tracer un losange $EFGH$ tel que la longueur de ses côtés soit de $5 \text{ cm}$ et la diagonale $EG$ mesure $8 \text{ cm}$.
            \item \textbf{Le triangle isocèle :} Tracer un triangle $IJK$ isocèle en $I$ tel que $JK = 4 \text{ cm}$ et $IJ = 6 \text{ cm}$.
            \item \textbf{Le triangle rectangle isocèle :} Tracer un triangle $LMN$ rectangle et isocèle en $L$ tel que $LM = 5 \text{ cm}$.
        \end{enumerate}
    \end{exercice}

\subsection{Conversion des aires}

    \begin{remarque}
        \begin{itemize}
            \item L'\textbf{aire} d'une figure est la mesure de sa surface intérieure. L'unité légale est le mètre carré ($m^2$), mais en géométrie sur cahier, on utilise très souvent le centimètre carré ($cm^2$).

            \item  Pour mesurer des superficies agraires (terrains, champs, forêts), on utilise parfois d'autres unités : 
            \begin{itemize}
                \item L'\textbf{are} (noté \textbf{a}) : $1 \text{ a} = 1 \text{ dam}^2 = 100 \text{ m}^2$.
                \item L'\textbf{hectare} (noté \textbf{ha}) : $1 \text{ ha} = 1 \text{ hm}^2 = 10\ 000 \text{ m}^2$.
            \end{itemize}
        \end{itemize}
    \end{remarque}
    
    
    \begin{propriete}
        Pour convertir des unités d'aire, on utilise un tableau de conversion où chaque unité est divisée en \textbf{deux colonnes} (unités et dizaines), car $1 \text{ m}^2 = 100 \text{ dm}^2$.
        
        \begin{center}
        \renewcommand{\arraystretch}{1.5}
        \small
        \begin{tabular}{|c|c|c|c|c|c|c|c|c|c|c|c|c|c|}
        \hline
        \multicolumn{2}{|c|}{\textbf{km}$^2$} & \multicolumn{2}{c|}{\textbf{hm}$^2$} & \multicolumn{2}{c|}{\textbf{dam}$^2$} & \multicolumn{2}{c|}{\textbf{m}$^2$} & \multicolumn{2}{c|}{\textbf{dm}$^2$} & \multicolumn{2}{c|}{\textbf{cm}$^2$} & \multicolumn{2}{c|}{\textbf{mm}$^2$} \\
        \multicolumn{2}{|c|}{} & \multicolumn{2}{c|}{(\textit{ha})} & \multicolumn{2}{c|}{(\textit{a})} & \multicolumn{2}{c|}{} & \multicolumn{2}{c|}{} & \multicolumn{2}{c|}{} & \multicolumn{2}{c|}{} \\
        \hline
        d & u & d & u & d & u & d & u & d & u & d & u & d & u \\
        \hline
          &   &   &   &   &   &   &   &   &   &   &   &   &   \\
          &   &   &   &   &   &   &   &   &   &   &   &   &   \\
        \hline
        \end{tabular}
        \end{center}
    \end{propriete}
    
    
    \begin{methode}
        Pour placer un nombre dans le tableau, on repère le \textbf{chiffre des unités} et on le place dans la colonne de droite (u) de l'unité de départ.
    \end{methode}
    

    \begin{exemple}
        Compléter les conversions suivantes :
        
        \vspace{0.3cm}

        \begin{itemize}
            \item[$\bullet$] $5 \text{ m}^2 = \dots\dots\dots\dots\dots\dots \text{ dm}^2$
            \item[$\bullet$] $12 \text{ cm}^2 = \dots\dots\dots\dots\dots\dots \text{ mm}^2$
            \item[$\bullet$] $3,5 \text{ dam}^2 = \dots\dots\dots\dots\dots\dots \text{ m}^2$
            \item[$\bullet$] $27,1 \text{ dm}^2 = \dots\dots\dots\dots\dots\dots \text{ m}^2$

            \item[$\bullet$] $450 \text{ m}^2 = \dots\dots\dots\dots\dots\dots \text{ a}$
            \item[$\bullet$] $0,15 \text{ km}^2 = \dots\dots\dots\dots\dots\dots \text{ ha}$
            \item[$\bullet$] $\np{654123} \text{ cm}^2 = \dots\dots\dots\dots\dots\dots \text{ m}^2$

\subsection{Formules d'aires}

    \begin{propriete}
        Voici les formules permettant de calculer l'aire des figures usuelles :
        
        \vspace{0.4cm}
        \begin{center}
        \begin{tikzpicture}
            % --- RECTANGLE ---
            \begin{scope}[xshift=0cm]
                \draw[thick] (0,0.5) rectangle (2.5,2);
                \node[below] at (1.25,0.5) {$L$};
                \node[right] at (2.5,1.25) {$l$};
                \node[below, align=center] at (1.25,0.1) {\textbf{Rectangle} \\[0.1cm] $\mathcal{A} = L \times l$};
            \end{scope}
        
            % --- CARRÉ ---
            \begin{scope}[xshift=4.5cm]
                \draw[thick] (0,0.5) rectangle (1.5,2);
                \node[below] at (0.75,0.5) {$c$};
                \node[right] at (1.5,1.25) {$c$};
                \node[below, align=center] at (0.75,0.1) {\textbf{Carré} \\[0.1cm] $\mathcal{A} = c \times c=c^2$};
            \end{scope}
        
            % --- TRIANGLE RECTANGLE ---
            \begin{scope}[xshift=8.2cm]
                \draw[thick] (0,2) -- (0,0.5) node[midway, left] {$b$} -- (2,0.5) node[midway, below] {$a$} -- cycle;
                \draw (0.2,0.5) -- (0.2,0.7) -- (0,0.7);
                \node[below, align=center] at (1,0.1) {\textbf{Triangle rectangle} \\[0.1cm] $\mathcal{A} = \dfrac{a \times b}{2}$};
            \end{scope}
        
            % --- DISQUE ---
            \begin{scope}[xshift=12.7cm]
                \draw[thick] (0,1.25) circle (0.75cm);
                \draw[thick, ->] (0,1.25) -- (0.75,1.25) node[midway, above] {$R$};
                \fill (0,1.25) circle (1.5pt);
                \node[below, align=center] at (0,0.1) {\textbf{Disque} \\[0.1cm] $\mathcal{A} = \pi \times R \times R=\pi\times R^2$};
            \end{scope}
        \end{tikzpicture}
        \end{center}
        \vspace{0.2cm}
    \end{propriete}

\subsection{Exemples de calculs}

    \begin{exercice}
        Pour chacune des situations ci-dessous, tu dois présenter ton travail en 3 étapes :
        \begin{itemize}
            \item[$\bullet$] Tracer une \textbf{figure à main levée} en y reportant les mesures et les codages.
            \item[$\bullet$] Écrire la \textbf{formule} littérale de l'aire correspondant à la figure.
            \item[$\bullet$] Remplacer les lettres par les nombres et faire le \textbf{calcul} (sans oublier l'unité finale !).
        \end{itemize}
        
        \vspace{0.3cm}
        \textbf{Calculer l'aire des figures suivantes :}
        \begin{enumerate}
            \item Un rectangle de longueur $7 \text{ cm}$ et de largeur $4 \text{ cm}$.
            \item Un carré de côté $6 \text{ m}$.
            \item Un triangle rectangle dont les côtés formant l'angle droit mesurent $5 \text{ cm}$ et $8 \text{ cm}$.
            \item Un disque de rayon $4 \text{ cm}$ (on prendra $\pi \approx 3,14$).
            \item Un disque de \textbf{diamètre} $10 \text{ dm}$. (\textit{Attention au \og piège \fg !})
        \end{enumerate}
    \end{exercice}
