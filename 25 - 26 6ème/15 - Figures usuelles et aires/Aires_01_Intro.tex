    \begin{exercice}[Aire ou Périmètre ?]
        \begin{minipage}{0.5\linewidth}
            On considère le réseau de lignes (quadrillage) ci-contre. On prend :
                \begin{itemize}
                    \item Comme unité de longueur : le côté d'un carreau, noté \textbf{u.l.}
                    \item Comme unité de surface (d'aire) : un petit carreau du quadrillage, noté $\mathcal{U}$.
            \end{itemize}
        \end{minipage}
        \hfill
        \begin{minipage}{0.49\linewidth}
            \begin{center}
                \begin{tikzpicture}[scale=0.5]
                    \draw[help lines, dashed, gray!60] (0,0) grid (18,6);
                    
                    % Unités de référence
                    \draw[ultra thick, Red] (1,5) -- (2,5) node[midway, above] {\small 1 u.l.};
                    \draw[ultra thick, fill=blue!20] (1,3) rectangle (2,4);
                    \node at (1.5,2.5) {\small 1 $\mathcal{U}$};
                
                    % Figure A (Rectangle 4x3)
                    \draw[ultra thick, fill=green!20] (4,1) rectangle (8,4);
                    \node at (6,2.5) {\Large \textbf{A}};
                
                    % Figure B (Forme en L ou escalier - Périmètre plus grand, même aire)
                    % Aire = 12 carreaux. Exemple: rectangle 2x5 + 2 = 12. 
                    % Base 5, hauteur 3. (5,1) -> (10,1) -> (10,2) -> (7,2) -> (7,4) -> (5,4) -> cycle
                    \draw[ultra thick, fill=orange!20] (11,1) -- (16,1) -- (16,2) -- (13,2) -- (13,4) -- (11,4) -- cycle;
                    \node at (12,2.5) {\Large \textbf{B}};
                \end{tikzpicture}
            \end{center}
        \end{minipage}
           
    
        \begin{enumerate}
            \item En comptant les unités, compléter le tableau suivant :
            
            \begin{center}
            \renewcommand{\arraystretch}{1.5}
            \begin{tabularx}{0.8\textwidth}{|X|c|c|}
            \hline
             & \textbf{Figure A} & \textbf{Figure B} \\
            \hline
            \textbf{Périmètre} (en u.l.) & \dots\dots\dots\dots\dots & \dots\dots\dots\dots\dots \\
            \hline
            \textbf{Aire} (en $\mathcal{U}$) & \dots\dots\dots\dots\dots & \dots\dots\dots\dots\dots \\
            \hline
            \end{tabularx}
            \end{center}
        
            \item Que peut-on en conclure ? Réciproquement, deux figures qui auraient la même aire auraient-elles forcément le même périmètre ? \newline
            \lignesPoints{2}
        \end{enumerate}
    \end{exercice}
