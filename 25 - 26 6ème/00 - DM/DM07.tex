% --- EXERCICE 2 : PROBLÈME AVEC DES DÉCIMAUX ---
    \begin{exerciceTitre}{La sortie au cinéma}
        La coopérative de la classe organise une sortie au cinéma. Elle dispose au départ d'une cagnotte de $245$~\euro.
        \begin{enumerate}
            \item Le maître achète 28 places à $4,50$~\euro{} l'unité. Calcule le montant total payé pour les places.
            \item Après cet achat, il s'aperçoit qu'il doit aussi payer le supplément pour la séance 3D, soit $1,25$~\euro{} par élève. Il paye ce supplément pour les 28 élèves.
            \item Quelle somme reste-t-il finalement dans la cagnotte de la classe après avoir payé les places et le supplément ?
        \end{enumerate}
    
        \begin{ZoneReponse}{3.5cm}
            \textbf{Correction :} \\
            1. $28 \times 4,5 = \mathbf{126}$ \euro. Le montant des places est de 126 \euro. \\
            2. $28 \times 1,25 = \mathbf{35}$ \euro. Le montant du supplément est de 35 \euro. \\
            3. Total dépensé : $126 + 35 = 161$ \euro. \\
            Somme restante : $245 - 161 = \mathbf{84}$ \euro.
        \end{ZoneReponse}
    \end{exerciceTitre}
    

% --- EXERCICE 3 : RECONNAISSANCE DE PROPORTIONNALITÉ ---
    \begin{exerciceTitre}{Reconnaître la proportionnalité}
        Indiquer si les situations suivantes relèvent de la proportionnalité ou non, en justifiant brièvement :
        \begin{enumerate}
            \item La taille d'une personne en fonction de son âge.
            \item Le prix de 5 kg de pommes sachant qu'un kg coûte 2,50 \euro. 
            \item La distance parcourue par un train roulant à vitesse constante en fonction du temps.
        \end{enumerate}
        
        \begin{ZoneReponse}{3cm}
            \textbf{Correction :} \\
            1. \textbf{Non}, car on ne grandit pas de la même manière chaque année (on finit par s'arrêter). \\
            2. \textbf{Oui}, car le prix est obtenu en multipliant la masse par un nombre fixe ($2,50$). \\
            3. \textbf{Oui}, car si la vitesse est constante, on parcourt deux fois plus de distance en deux fois plus de temps.
        \end{ZoneReponse}
    \end{exerciceTitre}

% --- EXERCICE 4 : TABLEAUX À COMPLÉTER ---
    \begin{exerciceTitre}{Résoudre des problèmes de proportionnalité}
        Compléter le tableau de proportionnalité suivant. On dessinera les flèches autour pour indiquer les manipulations effectuées.
        
        \begin{center}
        \begin{tabularx}{0.8\textwidth}{|l|*6{>{\centering\arraybackslash}X|}}
        \hline
        Nombre de livres & 2 & 3 & 6 & 9 & 15 & 17\\ \hline
        Prix en \euro & $\mathbf{7}$ & 10,50 & $\mathbf{21}$ & $\mathbf{31,50}$ & $\mathbf{52,50}$ & $\mathbf{59,50}$ \\ \hline
        \end{tabularx}
        \end{center}
        
        \begin{ZoneReponse}{3cm}
            \textbf{Logique de calcul :} \\            
            $\bullet$ Pour 6 livres : double de 3 livres $\rightarrow 10,5 \times 2 = \mathbf{21}$ \euro. \\
            $\bullet$ Pour 9 livres : triple de 3 livres $\rightarrow 10,5 \times 3 = \mathbf{31,50}$ \euro. \\
            $\bullet$ Pour 15 livres : $15 = 6 + 9$ donc  $21 + 31,5 = \mathbf{52,50}$ \euro.\\
            $\bullet$ Pour 2 livres : Trois fois moins que 6 livres $\rightarrow 21\div3=7$ \euro.
        \end{ZoneReponse}
    \end{exerciceTitre}
    

% --- EXERCICE 5 : PROBLÈME DE SYNTHÈSE ---
    \begin{exerciceTitre}{Cuisiner avec les mathématiques}
        Pour préparer un gâteau pour 4 personnes, il faut 200 g de farine et 3 œufs. 
        \begin{enumerate}
            \item Quelle quantité de farine faut-il pour 12 personnes ?
            \item Combien d'œufs faut-il prévoir pour 10 personnes ?
        \end{enumerate}
        
        \begin{ZoneReponse}{5cm}
            \textbf{Correction :} \\
            1. $12$ personnes, c'est le triple de $4$ ($4 \times 3$). Il faut donc $200 \times 3 = \mathbf{600}$ \textbf{g de farine}. 

            \medskip
            
            2. Pour 10 personnes : \\
            Pour 4 pers. $\rightarrow$ 3 œufs. Pour 8 pers. $\rightarrow$ 6 œufs. \\
            Pour 2 pers. (la moitié de 4) $\rightarrow$ 1,5 œuf. \\
            Total pour 10 pers ($8+2$) : $6 + 1,5 = \mathbf{7,5}$ \textbf{œufs}.
        \end{ZoneReponse}
    \end{exerciceTitre}


% --- EXERCICE 6 : COMPARAISON ET OPTIMISATION ---
    \begin{exerciceTitre}{Le meilleur prix}
        \begin{itemize}
            \item \textbf{Magasin A :} Les 3 gourdes sont vendues au prix de $13,50$~\euro.
            \item \textbf{Magasin B :} Les 5 gourdes sont vendues au prix de $22$~\euro.
        \end{itemize}
        \begin{enumerate}
            \item Dans quel magasin la gourde est-elle la moins chère ? Justifier par un calcul.
            \item L'association a besoin de 15 gourdes. Calcule le prix total à payer dans le magasin le moins cher.
        \end{enumerate}
    
        \begin{ZoneReponse}{4cm}
            \textbf{Correction :} \\
            1. Prix à l'unité : \\
            Magasin A : $13,50 \div 3 = \mathbf{4,50}$ \euro / gourde. \\
            Magasin B : $22 \div 5 = \mathbf{4,40}$ \euro / gourde. \\
            C'est au \textbf{Magasin B} que c'est le moins cher. 

            \medskip
            
            2. Pour 15 gourdes au Magasin B ($5 \times 3$) : $22 \times 3 = \mathbf{66}$ \textbf{\euro}.
        \end{ZoneReponse}
    \end{exerciceTitre}
