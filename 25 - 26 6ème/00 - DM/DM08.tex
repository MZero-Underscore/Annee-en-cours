    \begin{exercice}[Voyage vers Mars]
        \begin{enumerate}
            \item Compléter le tableau de proportionnalité suivant :
            
            \begin{center}
            \renewcommand{\arraystretch}{1.5}
            \begin{tabular}{|l|c|c|c|c|}
            \hline
            Volume de données (en Mo) & 10 & 20 & 50 & \textcolor{red}{\textbf{100}} \\ \hline
            Prix (en \euro) & 15 & \textcolor{red}{\textbf{30}} & \textcolor{red}{\textbf{75}} & 150 \\ \hline
            \end{tabular}
            \end{center}
            \textcolor{red}{\textit{Explications possibles : } \\
            - Pour passer de la ligne du haut à celle du bas, on multiplie par 1,5 (Coefficient de proportionnalité).\\
            - Pour 20 Mo : c'est le double de 10 Mo, donc $15 \times 2 = 30$ \euro.\\
            - Pour 50 Mo : c'est 5 fois 10 Mo, donc $15 \times 5 = 75$ \euro.\\
            - Pour 150 \euro : c'est 10 fois 15 \euro, donc $10 \times 10 = 100$ Mo.}
            
            \vspace{0.3cm}
            \item Une photo de la planète Mars pèse 12 Mo. Quel sera le prix pour envoyer cette photo ? \\
            \textcolor{red}{\textbf{Réponse :} On utilise le coefficient de proportionnalité : $12 \times 1,5 = \mathbf{18 \text{ \euro}}$. \\
            (Ou on peut décomposer : 10 Mo = 15 \euro{} et 2 Mo = 3 \euro, donc $15 + 3 = 18$ \euro).}
        \end{enumerate}
    \end{exercice}
