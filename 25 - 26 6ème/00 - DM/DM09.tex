    \begin{exercice}
        \textbf{Calcul d'un horaire}
        
        \textit{Énoncé : Un train part à 14 h 36 min. Le trajet dure 2 h 47 min. À quelle heure arrive-t-il ?}

        \begin{tcolorbox}[colback=white, colframe=gray, title=Correction]
            On additionne les heures et les minutes séparément :
            \begin{itemize}
                \item $14\text{ h } + 2\text{ h } = 16\text{ h}$
                \item $36\text{ min } + 47\text{ min } = 83\text{ min}$
            \end{itemize}
            Comme $83\text{ min } = 1\text{ h } 23\text{ min}$, on ajoute cette heure au total :
            $16\text{ h } + 1\text{ h } 23\text{ min } = 17\text{ h } 23\text{ min}$.\\
            \textbf{Le train arrive à 17 h 23 min.}
        \end{tcolorbox}
    \end{exercice}

    % CORRECTION EXO 2
    \begin{exercice}
        \textbf{Calcul d'une durée écoulée}
        
        \textit{Énoncé : Début du film à 20 h 50 min, fin à 22 h 15 min. Quelle est la durée du film ?}

        \begin{tcolorbox}[colback=white, colframe=gray, title=Correction]
            On utilise la méthode des compléments (ligne du temps) :
            \begin{itemize}
                \item De $20\text{ h } 50$ à $21\text{ h } 00$ : \textbf{$10\text{ min}$}
                \item De $21\text{ h } 00$ à $22\text{ h } 00$ : \textbf{$1\text{ h}$}
                \item De $22\text{ h } 00$ à $22\text{ h } 15$ : \textbf{$15\text{ min}$}
            \end{itemize}
            Total : $1\text{ h } + 10\text{ min } + 15\text{ min } = 1\text{ h } 25\text{ min}$.\\
            \textbf{La durée du film est de 1 h 25 min.}
        \end{tcolorbox}
    \end{exercice}

    % CORRECTION EXO 4
    \begin{exercice}
        \textbf{Opérations et conversions sur les durées}
        
        \textit{Énoncé : Effectuer l'addition et la soustraction de durées, puis convertir 54 h en jours/heures et 3 h 25 min en minutes.}

        \begin{minipage}{0.49\linewidth}
            \begin{tcolorbox}[colback=white, colframe=gray, title=Correction - Addition]
                On additionne d'abord chaque unité séparément :
                \begin{itemize}
                    \item Secondes : $30\text{ s} + 45\text{ s} = 75\text{ s}$
                    \item Minutes : $45\text{ min} + 30\text{ min} = 75\text{ min}$
                    \item Heures : $2\text{ h} + 1\text{ h} = 3\text{ h}$
                \end{itemize}
                On convertit ensuite les dépassements (en commençant par les secondes) :
                \begin{itemize}
                    \item $75\text{ s} = 1\text{ min } 15\text{ s}$. On ajoute la minute aux $75\text{ min}$, ce qui donne $76\text{ min}$.
                    \item $76\text{ min} = 1\text{ h } 16\text{ min}$. On ajoute l'heure aux $3\text{ h}$, ce qui donne $4\text{ h}$.
                \end{itemize}
                \textbf{Résultat final : $4\text{ h } 16\text{ min } 15\text{ s}$}
            \end{tcolorbox}
        \end{minipage}
        \hfill \vline \hfill
        \begin{minipage}{0.49\linewidth}
            \begin{tcolorbox}[colback=white, colframe=gray, title=Correction - Soustraction]
                On ne peut pas soustraire $30\text{ s}$ à $10\text{ s}$, ni $40\text{ min}$ à $15\text{ min}$. On fait donc des emprunts :
                \begin{itemize}
                    \item On emprunte $1\text{ min}$ (soit $60\text{ s}$) aux $15\text{ min}$ : la durée devient $4\text{ h } 14\text{ min } 70\text{ s}$.
                    \item On emprunte $1\text{ h}$ (soit $60\text{ min}$) aux $4\text{ h}$ : la durée devient $3\text{ h } 74\text{ min } 70\text{ s}$.
                \end{itemize}
                On peut maintenant soustraire terme à terme :
                \begin{itemize}
                    \item $70\text{ s} - 30\text{ s} = 40\text{ s}$
                    \item $74\text{ min} - 40\text{ min} = 34\text{ min}$
                    \item $3\text{ h} - 1\text{ h} = 2\text{ h}$
                \end{itemize}
                \textbf{Résultat final : $2\text{ h } 34\text{ min } 40\text{ s}$}
            \end{tcolorbox}
        \end{minipage}    

        \medskip
        \begin{minipage}{0.5\linewidth}
            \begin{tcolorbox}[colback=white, colframe=gray, title=Correction B1]
                Un jour dure 24 heures. 
                $2 \times 24 = 48\text{ heures}$. Il reste $54 - 48 = 6\text{ heures}$.\\
                Donc $54\text{ heures} = \textbf{2}\text{ jours et }\textbf{6}\text{ heures}$.
            \end{tcolorbox}
        \end{minipage}
        \hfill
        \begin{minipage}{0.49\linewidth}
            \begin{tcolorbox}[colback=white, colframe=gray, title=Correction B2]
                Une heure dure 60 minutes. 
                $3\text{ h} = 3 \times 60 = 180\text{ minutes}$.\\
                On ajoute les minutes restantes : $180 + 25 = 205\text{ minutes}$.\\
                Donc $3\text{ h } 25\text{ min} = \textbf{205}\text{ minutes}$.
            \end{tcolorbox}
        \end{minipage} 
    \end{exercice}
