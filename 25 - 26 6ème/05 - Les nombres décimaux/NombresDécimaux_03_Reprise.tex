    \section*{Correction (pour l'enseignant)}
        \begin{enumerate}
            \item \textbf{Total :} $3,45 + 2,80 + 1,15 = 7,40$ \euro. \textbf{Rendu :} $10 - 7,40 = 2,60$ \euro.
            \item \textbf{Morceaux :} $3 \times 2,4 = 7,2$ m. \textbf{Reste :} $15,5 - 7,2 = 8,3$ m.
            \item \textbf{Contenu :} $4,25 - 0,45 = 3,8$ kg. \textbf{Une boîte :} $3,8 \div 10 = 0,38$ kg (ou $380$ g).
        \end{enumerate}
