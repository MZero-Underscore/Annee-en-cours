    \begin{exercice}
        Pour effectuer le calcul $A=1,5\times2,3$, on peut écrire : 

        $\begin{array}{ll}
            A= & 1,5\times2,3 \\
            A= & 15\times0,1\times23\times0,1 \\
            A= & 15\times23\times0,1\times0,1 \\
            A= & 15\times23\times0,01 \\
            A= & 345 \times 0,01 \\
            A= & 3,45
        \end{array}$

        Poser les calculs suivants 
        \begin{enumerate}
            \item $B=1,4\times0,7$

            \item $C=1,23\times7,8$

            \item $D=0,04\times78,4$
        \end{enumerate}
    \end{exercice}
