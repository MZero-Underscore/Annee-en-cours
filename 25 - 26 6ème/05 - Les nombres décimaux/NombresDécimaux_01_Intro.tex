\setcounter{exo}{1}
\enteteActivite{1}{\monTitre}{\thechapter}

    \begin{exercice}
        Aide Margaux à former le plus grand nombre de quatre chiffres possibles pour ouvrir la porte.
        \begin{center}
            \includegraphics[scale=0.5]{images/Intro_Exo01.png}
        \end{center}
    \end{exercice}\stepcounter{exo}


    \begin{exercice}
        \begin{enumerate}[leftmargin=*]
            \item Quel est le chiffre des dizaines dans le nombre 43,87 ?

            \points[10]
        
            \item Quel est le chiffre des centièmes dans le nombre 8,107 ?

            \points[10]

        
            \item Quelle est l'écriture décimale correspondant au nombre $3 \times 10 + 4 \times 0,1$ ?

            \points[10]

            \item Quelle est l'écriture décimale correspondant au nombre $300 + 40 + 5 + 0,6 + 0,09$ ?

            \points[10]
    
            \item Quelle est l'écriture décimale correspondant au nombre $\frac{26}{10}$ ?

            \points[10]

            \item Quelle est l'écriture décimale correspondant au nombre $\frac{5}{100}$ ?

            \points[10]

            \item Ranger dans l'ordre croissant les nombres suivants : $5,4 ; \quad  5 ; \quad  5,04$.

            \points[10]
      
            \item Ranger dans l'ordre décroissant les nombres suivants : $12,61 \quad  12,02$ et $12,6$.

            \points[10]

        \end{enumerate}
    \end{exercice}
