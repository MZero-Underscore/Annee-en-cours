\section{Prendre la fraction d'une quantité}

    \begin{proprieteTitre}{Prendre la fraction d'un nombre}
        Pour prendre une fraction $\dfrac{a}{b}$ d'une quantité (où $b \neq 0$), on multiplie cette quantité par la fraction $\dfrac{a}{b}$.
    \end{proprieteTitre}
    
    
    \begin{remarque}
        En pratique, pour calculer $\dfrac{a}{b} \times \text{Quantité}$, on peut procéder de trois manières différentes :
        \begin{itemize}
            \item Diviser la quantité par $b$, puis multiplier le résultat par $a$.
            \item Multiplier la quantité par $a$, puis diviser le résultat par $b$.
            \item Calculer la valeur décimale de la fraction $\dfrac{a}{b}$ (quand elle existe !), puis la multiplier par la quantité.
        \end{itemize}
    \end{remarque}

\section{Appliquer un pourcentage}
    
    \begin{definitionTitre}{Pourcentage}
        Un \textbf{pourcentage} est une fraction dont le dénominateur est égal à $100$. 
        L'écriture « $x\,\%$ » se lit « $x$ pour cent » et est égale à la fraction $\dfrac{x}{100}$.
    \end{definitionTitre}

    
    \begin{proprieteTitre}{Appliquer un pourcentage}
        Appliquer un pourcentage de $x\,\%$ à une quantité revient à multiplier cette quantité par la fraction $\dfrac{x}{100}$. 
        Cela correspond à une situation de proportionnalité.
    \end{proprieteTitre}

\section{Additionner des fractions}
    
    \begin{proprieteTitre}{Somme de fractions de même dénominateur}
        Pour \textbf{additionner} (ou \textbf{soustraire}) deux fractions:
        \begin{itemize}
            \item On réduit les nombres au \textbf{même dénominateur}.
            \item On additionne (ou on soustrait) les numérateurs et on garde le dénominateur commun.
        \end{itemize}
    \end{proprieteTitre}
