        \begin{center}
            \begin{tikzpicture}[x=2cm, y=1cm]
    
                % 1. On trace la ligne principale avec une flèche au bout
                \draw[->, >=stealth, thick] (0,0) -- (5.5,0);
            
                % 2. On trace les petites graduations (les dixièmes)
                % On boucle de 0 à 5 par pas de 0.1
                \foreach \x in {0, 0.1, ..., 5} {
                    \draw (\x, -0.05) -- (\x, 0.05);
                }
            
                % 3. On trace les grandes graduations (les entiers) avec les chiffres
                \foreach \x in {0, 1, 2, 3, 4, 5} {
                    \draw[thick] (\x, -0.15) -- (\x, 0.15);
                    \node[below=6pt] at (\x, 0) {\textbf{\x}};
                }
            
                % 4. On place des points à lire (A, B, C)
                % Point A à 1.2
                \filldraw[red] (1.2,0) circle (2pt) node[above=8pt] {A};
