\section*{Partie 1 : Travailler avec des nombres entiers}
Dans cette partie, on ne travaille qu'avec des nombres entiers. Aucun résultat ne doit être donné sous forme d'un nombre décimal.

    \begin{exerciceNote}{3}
        Poser les divisions \textbf{euclidiennes} suivantes et donner l'égalité correspondante.

        \medskip

        \begin{minipage}{0.33\linewidth}
            \quad \Division[Solution]{32548}{6}

            \[\np{32548}=6\times\mathbf{5424}+\mathbf{4}\]
    
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.33\linewidth}
            \quad \Division[Solution]{8179}{13}

            \[\np{8179}=13\times\mathbf{629}+\mathbf{2}\]
            
        \end{minipage}        
        \hfill\vline\hfill
        \begin{minipage}{0.33\linewidth}
            \quad \Division[Solution]{8765}{4}
            
            \[\np{8765}=4\times\mathbf{2191}+\mathbf{1}\]

        \end{minipage}
    \end{exerciceNote}


    \begin{exerciceNote}{5}
        \begin{enumerate}[leftmargin=*]
            \item Compléter les phrases suivantes :
            \begin{itemize}
                \item Un nombre entier est divisible par 2 si \textbf{son chiffre des unités est 0, 2, 4, 6 ou 8 (il est pair).}

                \medskip
                
                \item Un nombre entier est divisible par 3 si \textbf{la somme de ses chiffres est un multiple de 3.}
            \end{itemize}

            \item On sait que $135=5\times27$ et que $60=12\times5$. \\
            Écrire 4 phrases utilisant les mots \textbf{multiple, diviseur, divisible} et \textbf{divise}. 

            \begin{ZoneReponse}{3cm}
            \begin{itemize}
                \item 135 est un \textbf{multiple} de 5 (ou de 27).
                \item 5 est un \textbf{diviseur} de 60 (ou de 135).
                \item 60 est \textbf{divisible} par 12 (ou par 5).
                \item 12 \textbf{divise} 60.
            \end{itemize}
            \end{ZoneReponse}

            \item Entourer les divisions euclidiennes qui ont pu être posées.

            \begin{minipage}{0.45\linewidth}
                \[333=25\times13+8\]
                \begin{center}
                    \fbox{division de 333 par 25} \quad \quad $\cancel{\text{division de 333 par 8}}$
                    \\ \quad (car $8 < 25$) \quad \hfill
                \end{center}
            \end{minipage}
            \hfill \vline \hfill
            \begin{minipage}{0.45\linewidth}
                \[654=42\times15+24\]
                \fbox{division de 654 par 42} \quad \quad $\cancel{\text{division de 654 par 15}}$
                \\ \quad (car $24 < 42$) \hfill
            \end{minipage} 

            \begin{ZoneReponse}{1.8cm}
                Pour la division de 333 par 8, ce n'est pas la bonne écriture. Ici 8 joue le rôle du reste.
                
                Pour la division de 624 par 15, le reste (24) est plus grand que le diviseur (15), ce qui est impossible dans une division euclidienne finie.
            \end{ZoneReponse}
        \end{enumerate}
    \end{exerciceNote}

        \begin{enumerate}[leftmargin=*]

        \item \begin{enumerate}[leftmargin=*]
            \item Écrire un nombre à 5 chiffres divisible par 3 et 5 mais pas par 2 et 9.
            \begin{ZoneReponse}{1.5cm}
            Exemple : \textbf{10 005}. 
            Justification : Il finit par 5 (donc /5 et pas /2). La somme $1+0+0+0+5=6$ (donc /3 mais pas /9).
            \end{ZoneReponse}
    
            \item Plus grand : \textbf{99 975}. Plus petit : \textbf{10 005}.
            \end{enumerate}
        \end{enumerate}
    \end{exerciceNote}

    \begin{exerciceNote}{3}
        Le club de mathématiques dispose de \textbf{458 bonbons} au caramel et de \textbf{327 bonbons} à la menthe.

        \begin{enumerate}[leftmargin=*]
            \item Ils décident de faire des sachets contenant chacun \textbf{12 bonbons au caramel}.
            \begin{enumerate}[leftmargin=*]
                \item Combien de sachets complets peuvent-ils préparer ?
                \item Combien de bonbons au caramel restera-t-il après la préparation ?

                \begin{ZoneReponse}{2cm}
                    $458 = 12 \times 38 + 2$. \\
                    a) On peut préparer \textbf{38 sachets} complets. \\
                    b) Il restera \textbf{2 bonbons}.
                \end{ZoneReponse}
            \end{enumerate}
            \item Pour les bonbons à la menthe, ils veulent préparer \textbf{15 sachets identiques}.
            \begin{enumerate}[leftmargin=*]
                \item Combien de bonbons y aura-t-il dans chaque sachet ?
                \item Combien en manque-t-il pour pouvoir en mettre un de plus dans chaque sachet ?

                \begin{ZoneReponse}{2.5cm}
                    $327 = 15 \times 21 + 12$. \\
                    a) Il y aura \textbf{21 bonbons} par sachet. \\
                    b) Il reste 12 bonbons. Pour en avoir 15 (un de plus par sachet), il en manque $15 - 12 = \mathbf{3}$.
                \end{ZoneReponse}
            \end{enumerate}
        \end{enumerate}
    \end{exerciceNote}

    \begin{exerciceNote}{3}
        \subsection*{1. La liste de courses}
        À l'aide des informations ci-dessous, vérifier que le montant total des achats de Léa est de \textbf{40,53} €.
        \begin{itemize}
            \item \textbf{3 paquets} de gâteaux à \textbf{2,85 €} l'unité ;
            \item \textbf{2 bouteilles} de jus de fruits à \textbf{1,74 €} l'unité ;
            \item \textbf{1,5 kg} de bonbons vendus à \textbf{6,40 €} le kilogramme ;
            \item Un gâteau au chocolat à \textbf{18,90 €}.
        \end{itemize}
        
        \begin{ZoneReponse}{2.5cm}
        $3 \times 2,85 = 8,55$ \euro \\
        $2 \times 1,74 = 3,48$ \euro \\
        $1,5 \times 6,40 = 9,60$ \euro \\
        Total : $8,55 + 3,48 + 9,60 + 18,90 = \mathbf{40,53}$ \textbf{\euro}. Le compte est bon.
        \end{ZoneReponse}
        
        \subsection*{2. À la caisse}
        Léa paye avec son billet de \textbf{50 €}. 
        \\ \textbf{Question :} Quelle somme la caissière doit-elle lui rendre ?
        
        \begin{ZoneReponse}{1.2cm}
        $50 - 40,53 = \mathbf{9,47}$ \textbf{\euro}. La caissière rend 9,47 \euro.
        \end{ZoneReponse}
        
        \subsection*{3. Le partage des frais}
        
        Le coût total de la fête est réparti équitablement entre les \textbf{8 enfants} présents.
        \\ \textbf{Question :} Quel est le prix de la fête pour un seul enfant ? (on arrondira au centime)
        
        \begin{ZoneReponse}{1.5cm}
        $40,53 \div 8 = 5,06625$. \\
        Arrondi au centime, cela fait \textbf{5,07 \euro} par enfant.
        \end{ZoneReponse}
    \end{exerciceNote}
