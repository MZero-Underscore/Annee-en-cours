\section{Vocabulaire et notations}
    
    \begin{definition}
        Un \textbf{angle} est une portion de plan délimitée par deux demi-droites de même origine.
        \begin{itemize}[label=\textbullet]
            \item L'origine commune des deux demi-droites est appelée le \textbf{sommet} de l'angle.
            \item Les deux demi-droites sont appelées les \textbf{côtés} de l'angle.
        \end{itemize}
    \end{definition}

    
    \begin{remarque}
        On note un angle avec trois lettres surmontées d'un chapeau. La lettre du milieu désigne toujours le \textbf{sommet}. 
        Les lettres aux extrémités désignent soit des points (lettres majuscules), soit le nom des demi-droites (lettres minuscules).
    \end{remarque}
    
    
    \begin{exemple}
        \begin{center}
            \begin{tabular}{cc}
                \begin{tikzpicture}[scale=1, baseline=(current bounding box.center)]
                    % Premier angle : avec des points (Majuscules)
                    \draw[thick] (2.5,1.5) -- (0,0) node[left] {$A$} -- (3,0);
                    \fill (2,1.2) circle (1.5pt) node[above left] {$C$};
                    \fill (2.2,0) circle (1.5pt) node[below] {$B$};
                    \draw[->,>=latex, green!55!black] (0.6,0) arc (0:31:0.6);
                    \node at (3.5,0.7) {\textbf{Points :} $\widehat{BAC}$};
                \end{tikzpicture} 
                & 
                \begin{tikzpicture}[scale=1, baseline=(current bounding box.center)]
                    % Deuxième angle : avec des noms de demi-droites (minuscules)
                    \draw[thick] (0,0) node[left] {$O$} -- (30:2.5) node[right] {$x$};
                    \draw[thick] (0,0) -- (0:3) node[right] {$y$};
                    \draw[->,>=latex, green!55!black] (0.6,0) arc (0:30:0.6);
                    \node at (3.5,0.7) {\textbf{Directions :} $\widehat{xOy}$};
                \end{tikzpicture} \\
            \end{tabular}
        \end{center}
    \end{exemple}
    
    
\section{Mesure et nature des angles}


    \begin{definition}
        Le \textbf{degré} est une unité de mesure d'angle. 

        
        Un angle plein (un tour complet) mesure \textbf{360 degrés}, ce que l'on note \textbf{360$^{\circ}$}.
        
        Un degré ($1^{\circ}$) représente donc $\frac{1}{360}$ d'un tour complet.
    \end{definition}
      
    
    \begin{exemple}
        Visualisation du partage d'un tour complet :
        \begin{center}
        \begin{tikzpicture}[scale=1.5]
            % Cercle complet en fond léger
            \draw[gray!20, thin] (0,0) circle (2);
            
            % Marquage du centre
            \fill (0,0) circle (1pt) node[below] {$O$};
            
            % Dessin de quelques secteurs de 1 degré (exagérés pour la visibilité ou très fins)
            \foreach \a in {0,1,2,3,4,5} {
                \draw[blue!70] (0,0) -- (\a:2);
            }
            
            % Un secteur coloré pour montrer "1 part"
            \fill[blue!20] (0,0) -- (0:2) arc (0:1:2) -- cycle;
            
            % Annotations
            \draw[<->, >=latex] (2.2,0) arc (0:360:2.2);
            \node at (2.8,1) {\small 1 tour = $360^{\circ}$};
            
            \draw[red, thick, <-] (0.5,0.02) -- (1.5,0.8) node[right, color=black] {Un petit écart vaut $1^{\circ}$};
        \end{tikzpicture}
        \end{center}
    \end{exemple}


    \begin{remarque}
        L'instrument utilisé pour mesurer ou construire un angle en degrés est le \textbf{rapporteur}. 
        La plupart des rapporteurs sont gradués de $0^{\circ}$ à $180^{\circ}$ (un demi-tour).
    \end{remarque}

\section{Construction d'un angle}

    
    \begin{methode}
        Pour construire un angle $\widehat{ABC}$ de $70^{\circ}$ :
        \begin{enumerate}
            \item On trace un premier côté $[BA)$.
            \item On place le centre du rapporteur sur le sommet $B$.
            \item On aligne le zéro du rapporteur avec le côté $[BA)$.
            \item On fait une marque à $70^{\circ}$ et on trace la demi-droite $[BC)$.
        \end{enumerate}
    \end{methode}

\section{La bissectrice d'un angle}


    \begin{definition}
        La \textbf{bissectrice} d'un angle est la demi-droite qui partage cet angle en deux angles de même mesure.
        \begin{center}
        \begin{tikzpicture}[scale=1.2]
            % Les côtés de l'angle
            \draw[thick] (0,0) node[left]{$A$} -- (3,0) node[right]{$B$};
            \draw[thick] (0,0) -- (40:3) node[right]{$C$};
            
            % La bissectrice
            \draw[thick, dashed, blue!85!white] (0,0) -- (20:3.5) node[right]{$d$};
            
            % Les arcs de codage (couleur de la boîte propriété pour le rappel de l'égalité)
            \draw[red!85!white, thick] (0.8,0) arc (0:20:0.8);
            \draw[red!85!white, thick] (20:0.8) arc (20:40:0.8);
            
            % Les symboles de codage (petits traits)
            \node[red!85!white] at (10:0.8) {/};
            \node[red!85!white] at (30:0.8) {/};
            
            \node[blue!85!white] at (2.5,0.5) {bissectrice};
        \end{tikzpicture}
        \end{center}

        Il s'agit de \textbf{l'axe de symétrie de l'angle}.
    \end{definition}
    
    
    \begin{propriete}
        \begin{minipage}[c]{0.55\textwidth}
            Si la demi-droite $[Ad)$ est la bissectrice de l'angle $\widehat{BAC}$, alors elle partage l'angle en deux angles égaux :
            \[ \widehat{BAd} = \widehat{dAC} = \widehat{BAC} \div 2 \]
        
            \textbf{Exemple de calcul :} \\
            Si $\widehat{BAC} = 70^{\circ}$, alors : \\
            $\widehat{BAd} = 70 \div 2 = \mathbf{35^{\circ}}$
        \end{minipage}
        \hfill
        \begin{minipage}[c]{0.4\textwidth}
            \centering
            \begin{tikzpicture}[scale=1.1]
                % Côtés
                \draw[thick] (0,0) node[left]{$A$} -- (3,0) node[right]{$B$};
                \draw[thick] (0,0) -- (50:3) node[above right]{$C$};
                % Bissectrice
                \draw[thick, dashed, blue!85!white] (0,0) -- (25:3.5) node[right]{$d$};
                % Arcs et codages
                \draw[red!85!white, thick] (0.7,0) arc (0:25:0.7) node[midway, right, xshift=2pt] {\small $35^{\circ}$};
                \draw[red!85!white, thick] (25:0.7) arc (25:50:0.7) node[midway, above right] {\small $35^{\circ}$};
                \node[red!85!white] at (12.5:0.7) {/};
                \node[red!85!white] at (37.5:0.7) {/};
            \end{tikzpicture}
        \end{minipage}
    \end{propriete}
    
    
    \begin{methode}[Construction au compas]
        Pour tracer la bissectrice de l'angle $\widehat{BAC}$ sans rapporteur :
        \begin{enumerate}
            \item On trace un arc de cercle de centre $A$ qui coupe les deux côtés de l'angle en deux points $M$ et $N$.
            \item En gardant le même écartement (ou un autre), on trace deux arcs de cercle : l'un de centre $M$, l'autre de centre $N$, situés à l'intérieur de l'angle.
            \item Ces deux arcs se coupent en un point $P$. 
            \item La demi-droite $[AP)$ est la bissectrice de l'angle $\widehat{BAC}$.
        \end{enumerate}
    \end{methode}
    
    
    \begin{exemple}
        Soit $\widehat{EFG} = 76^{\circ}$. Construire l'angle puis sa bissectrice :
        \begin{enumerate}
            \item à l'aide du rapporteur.
            \item à l'aide du compas.
        \end{enumerate}
    \end{exemple}
