\section{Arithmétique}    
\textbf{L'arithmétique}, c'est l'étude des \textbf{nombres entiers} et uniquement des nombres entiers.  
\subsection{La division euclidienne}

    On parle de \textbf{division euclidienne} lorsque l'on veut \textbf{diviser un nombre entier par un nombre entier} (non nul) et que l'on souhaite obtenir \textbf{un quotient entier}. Il arrive alors que la division ne soit pas totale et que l'on trouve \textbf{un reste} (qui est également un nombre entier).

    \begin{definition}
        Effectuer la division euclidienne d'un nombre entier $a$ par un nombre entier non nul $b$ c'est trouver deux uniques entiers $q$ et $r$ tels que :
        \[a=bq+r \text{ et } 0\leq r<b\]        
    \end{definition}


    \begin{remarque}
        En cm2, on a plus l'habitude de voir la division posée :

        \medskip

        \begin{minipage}{0.3\linewidth}
            $\begin{array}{cc|c }
                 a & & b \\ \cline{3-3}
                   & & q \\
                   & r &   
            \end{array}$
        \end{minipage}
        \hfill
        \begin{minipage}{0.65\linewidth}
            Le nombre $a$ s'appelle le \textbf{dividende} et le nombre $b$ s'appelle le \textbf{diviseur}.

            \medskip
            
            Le nombre $q$ s'appelle le \textbf{quotient} et le nombre $r$ s'appelle le \textbf{reste}.
        \end{minipage}
    \end{remarque}

\subsection{Multiple et diviseur}
    
    \begin{definition}
        Lorsque \textbf{le reste} de la division euclidienne de $a$ par $b$ \textbf{est nul} (vaut zéro). On dit que :

        \begin{itemize}
            \item $b$ est un \textbf{diviseur} de $a$.
            \item $b$ \textbf{divise} $a$.
            \item $a$ est un \textbf{multiple} de $b$.
        \end{itemize}
    \end{definition}

\subsection{Critères de divisibilité}
    Il faut apprendre par cœur ces critères, ils permettent de gagner du temps dans plusieurs chapitres.

    \begin{propriete}
        Un nombre entier est \textbf{divisible par :}
            \begin{itemize}
                \item 2 si son chiffre des unités est 0; 2; 4; 6 ou 8;
                \item 3 si la somme de ses chiffres est divisible par 3;
                \item 4 si le nombre formé par les chiffres des dizaines et des unités est divisible par 4.
                \item 5 si son chiffre des unités est 0 ou 5;
                \item 9 si la somme de ses chiffres est divisible par 9.
            \end{itemize}
    \end{propriete}

\section{La division décimale}
    On traitera deux cas et \textbf{le diviseur restera entier}. 
    \begin{itemize}
        \item Le dividende est décimal (avec partie décimale non nulle)
        \item Le dividende est entier mais on ne souhaite pas avoir de reste.
    \end{itemize}
    
    \begin{methode}
        \begin{itemize}
            \item Dans les deux cas, on commence par diviser la partie entière du dividende comme pour une division euclidienne.

            \medskip

            \item \textbf{Dès que la division de la partie entière est terminée, il faut placer immédiatement une virgule au quotient.}

            \medskip

            \item Ensuite, on poursuit la division avec le chiffre des dixièmes puis des centièmes, \ldots jusqu'à essayer d'obtenir un reste nul. On ajoute des zéros \og inutiles \fg à la droite du dividende si besoin.
        \end{itemize}
    \end{methode}


    \begin{remarque}
        \textbf{\underline{Attention ! }} Certaines divisions décimales sont infinies, il faudra alors savoir arrêter les calculs.   
    \end{remarque}
