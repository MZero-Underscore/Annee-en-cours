    \begin{exerciceTitre}{Sens de la fraction et quotient}
        \begin{enumerate}
            \item Un agriculteur possède un grand champ. Il cultive du blé sur les $\frac{5}{11}$ de son champ et du maïs sur les $\frac{4}{11}$.
            \begin{enumerate}
                \item Quelle culture occupe la plus grande surface ? Justifier.
                \item Quelle fraction du champ est cultivée au total ?
                \item Reste-t-il une partie du champ non cultivée ? Si oui, l'exprimer sous forme de fraction.
            \end{enumerate}
            \item Recopier et compléter les égalités suivantes par le nombre exact (sous forme de fraction si besoin) :
            
            \vspace{0.2cm}
            a) $8 \times \dots = 5$ \hfill b) $\dots \times 12 = 7$ \hfill c) $9 \times \frac{4}{9} = \dots$ \hfill d) $\dots \times \frac{13}{4} = 13$ \hfill
        \end{enumerate}
    \end{exerciceTitre}
    
    
    \begin{exerciceTitre}{Repérage sur une demi-droite graduée}
        \begin{enumerate}
            \item Sur le cahier, tracer une demi-droite graduée en prenant 12 petits carreaux (ou 12 cm) pour une unité.
            \item Placer les points suivants sur cette demi-droite :
            \[ A\left(\frac{7}{12}\right) \qquad B\left(\frac{1}{2}\right) \qquad C\left(\frac{5}{4}\right) \qquad D\left(\frac{4}{3}\right) \qquad E\left(\frac{3}{2}\right) \qquad F\left(\frac{17}{12}\right) \]
            \textit{Indication : Il faudra utiliser la règle des fractions égales pour transformer certaines fractions afin de leur trouver un dénominateur commun.}
        \end{enumerate}
    \end{exerciceTitre}
    
    
    \begin{exerciceTitre}{Fractions égales et pourcentages}
        \begin{enumerate}
            \item Simplifier au maximum les fractions suivantes en justifiant (écrire les étapes de calcul) : 
            \[ \frac{30}{45} \qquad ; \qquad \frac{24}{32} \qquad ; \qquad \frac{42}{60} \]
            \item Dans un collège de 600 élèves, $25\,\%$ des élèves sont inscrits à l'Association Sportive.
            \begin{enumerate}
                \item À quelle fraction simplifiée correspond un pourcentage de $25\,\%$ ?
                \item Calculer le nombre d'élèves inscrits à l'Association Sportive en utilisant cette fraction.
                \item Parmi ces inscrits, le tiers pratique le badminton. Combien d'élèves pratiquent le badminton au sein du collège ?
            \end{enumerate}
        \end{enumerate}
    \end{exerciceTitre}
    
    
    \begin{exerciceTitre}{Comparaison et rangement}
        \begin{enumerate}
            \item Comparer les fractions suivantes en utilisant les symboles $<$, $>$ ou $=$ (justifier brièvement en citant la règle du cours utilisée) :
            
            \vspace{0.2cm}
            a) $\frac{14}{9}$ et $\frac{11}{9}$ \hfill b) $\frac{7}{15}$ et $\frac{7}{12}$ \hfill c) $\frac{8}{8}$ et $1$ \hfill d) $\frac{17}{18}$ et $1$ \hfill
            
            \vspace{0.2cm}
            \item Ranger les fractions suivantes dans l'ordre \textbf{croissant} (du plus petit au plus grand) : 
            \[ \frac{3}{5} \qquad ; \qquad \frac{3}{10} \qquad ; \qquad \frac{3}{2} \qquad ; \qquad \frac{3}{8} \qquad ; \qquad \frac{3}{4} \]
        \end{enumerate}
    \end{exerciceTitre}
    
    
    \begin{exerciceTitre}{Décomposition et problème}
        \begin{enumerate}
            \item Décomposer les fractions suivantes sous la forme d'une somme d'un nombre entier et d'une fraction strictement inférieure à 1 :
            \[ \frac{23}{5} \qquad ; \qquad \frac{45}{8} \qquad ; \qquad \frac{31}{6} \]
            \item \textbf{Problème :} Pour préparer un tournoi, un professeur d'EPS a besoin de distribuer 45 balles de tennis. Les balles sont vendues par boîtes de 8.
            \begin{enumerate}
                \item En utilisant la décomposition de la fraction $\frac{45}{8}$ calculée à la question 1, déterminer le nombre de boîtes complètes qu'il devra utiliser.
                \item Combien de boîtes le professeur devra-t-il acheter au minimum au magasin pour avoir suffisamment de balles ? Justifier.
            \end{enumerate}
        \end{enumerate}
    \end{exerciceTitre}
