    \begin{exercice}[Le jardinier (Partage et sens de la fraction)]
        Un jardinier possède un terrain rectangulaire qu'il décide de fleurir. 
        Il consacre les $\dfrac{3}{8}$ du terrain aux roses. Sur ce qu'il reste du terrain après avoir planté les roses, il consacre $\dfrac{2}{5}$ de cette nouvelle surface aux tulipes. Le reste est laissé en herbe.
        \begin{enumerate}
            \item Représenter le terrain par un grand rectangle divisé en $40$ petites cases de même taille (par exemple un quadrillage de $5$ cases sur $8$).
            \item Hachurer en rouge ou colorier la partie du terrain réservée aux roses.
            \item Hachurer en bleu ou colorier la partie du terrain réservée aux tulipes (\textit{attention à bien lire l'énoncé !}).
            \item En déduire quelle fraction du terrain total représente la partie laissée en herbe.
        \end{enumerate}
    \end{exercice}
    
    \begin{exercice}[La fraction comme quotient]
        \begin{enumerate}
            \item Compléter les égalités suivantes avec des fractions ou des entiers sans poser d'opération :
            \begin{enumerate}
                \item $7 \times \ldots\ldots = 5$ \hfill
                \item $11 \times \dfrac{4}{11} = \ldots\ldots$ \hfill
                \item $\ldots\ldots \times \dfrac{13}{6} = 13$ \hfill
                \item $8 \times \ldots\ldots = 20$
            \end{enumerate}
            \vspace{0.2cm}
            \item Pour une activité de travaux manuels, le professeur doit découper une ficelle de $15$~cm en $4$ morceaux de longueurs identiques. 
            \begin{enumerate}
                \item Quelle est la longueur exacte d'un morceau, écrite sous la forme d'une fraction ?
                \item Donner l'écriture décimale de cette longueur.
            \end{enumerate}
            \item \textbf{Défi :} Trouver le nombre manquant dans l'égalité suivante : $\dfrac{5}{3} \times \ldots\ldots = 5$.
        \end{enumerate}
    \end{exercice}
    
    \begin{exercice}[Placements stratégiques sur la demi-droite graduée]
        \begin{enumerate}
            \item Tracer une demi-droite graduée en prenant une unité d'une longueur de $12$~carreaux (ou $12$~cm).
            \item Placer le plus précisément possible les points suivants :
            \[ A\left(\dfrac{1}{3}\right) \quad ; \quad B\left(\dfrac{1}{2}\right) \quad ; \quad C\left(\dfrac{3}{4}\right) \quad ; \quad D\left(\dfrac{5}{6}\right) \quad ; \quad E\left(\dfrac{13}{12}\right) \]
            \item Sur cette même demi-droite, placer un point $M$ situé exactement au milieu du segment $[BD]$. 
            \item Lire l'abscisse du point $M$ sur la demi-droite graduée et l'écrire sous la forme d'une fraction la plus simple possible.
        \end{enumerate}
    \end{exercice}
    
    \begin{exercice}[Le partage de la tablette de chocolat]
        Trois amis, Anna, Bilal et Chloé, se partagent une tablette de chocolat composée de $24$ carrés identiques.
        \begin{itemize}
            \item Anna prend $\dfrac{1}{4}$ de la tablette.
            \item Bilal prend $\dfrac{1}{3}$ de la tablette.
            \item Chloé prend $\dfrac{5}{12}$ de la tablette.
        \end{itemize}
        
        \begin{enumerate}
            \item Sans faire de calculs (en utilisant uniquement un dessin de la tablette ou un axe gradué), expliquer comment on peut s'assurer qu'à eux trois, ils ont mangé la totalité de la tablette.
            \item Combien de carrés de chocolat chacun a-t-il mangés ? Est-ce que cela confirme la réponse à la question précédente ?
        \end{enumerate}
    \end{exercice}
