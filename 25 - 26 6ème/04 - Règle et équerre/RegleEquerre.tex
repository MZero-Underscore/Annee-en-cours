\section{Les droites}
\subsection{Rappels}
    \begin{definition}
        Soit A ,B et C trois points deux à deux distincts.
        \begin{enumerate}
            \item Les points A, B et C sont \textbf{alignés} si on peut tracer une droite passant par ces trois points.
            \item La droite $(AB)$ est \textbf{l'ensemble de tous les points alignés} avec les points A et B.
        \end{enumerate}
    \end{definition}

\subsection{Positions relatives entre deux droites}

    \begin{definition}
        \begin{itemize}
            \item Deux droites sont \textbf{sécantes} si elles se coupent en un unique point que l'on appelle $\textbf{le point d'intersection}$.

            \item Deux droites sont \textbf{perpendiculaires} si elles sont sécantes \underline{et} qu'elles forment un \textbf{angle droit}.

            \item Deux droites sont \textbf{parallèles} si elles ne sont pas sécantes.
        \end{itemize}
    \end{definition}


    \begin{exemple}
        \begin{minipage}{0.3\linewidth}

\section{Médiatrice d'un segment}
\subsection{Rappels}
    \begin{definition}
        Soit A et B deux points distincts.
        \begin{itemize}
            \item Le segment [AB] est la partie de la droite située entre les points A et B.
            \item Le milieu du segment [AB] est le point du segment qui est situé à égale distance des points A et B.
        \end{itemize}
    \end{definition}


    \begin{exemple}
        Construire un segment [AB] tel que $AB=7$ cm puis construire son milieu qu'on appellera I.

        \vspace{1cm}
    \end{exemple}


\subsection{Médiatrice d'un segment}
    \begin{definition}
        La $\textbf{médiatrice d'un segment}$ est la droite qui coupe ce segment en son milieu \underline{et} en formant un angle droit.
    \end{definition}


    \begin{exemple}
        \begin{enumerate}
            \item Tracer un segment [AB] de longueur 3 cm, puis construire sa médiatrice à l'aide d'une règle graduée et d'une équerre.

            \item Placer un point P sur cette médiatrice telle que AP = 2 cm à l'aide du compas. Quelle semble être la longueur BP ? 


            \vspace{3cm}
        \end{enumerate}
    \end{exemple}


    \begin{propriete}
        La médiatrice d'un segment est l'ensemble des points équidistants des extrémités de ce segment.
    \end{propriete}
