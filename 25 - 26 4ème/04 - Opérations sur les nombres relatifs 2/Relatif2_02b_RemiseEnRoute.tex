    \begin{exercice}
        \begin{enumerate}
            \item ABC est un triangle rectangle en B, AB = 10cm et BC=7cm. Calculer CA au mm près.

            \item EFG est un triangle rectangle en G tel que GE = 5 m et GF = 7 m. Calculer EF au mm près.
        \end{enumerate}
    \end{exercice}\stepcounter{exo}


    \begin{exercice}
        Sans calculatrice, effectuer : 

        $A=-4+5-(-15)-13$ \quad $B=5-(-5+3)-27$ \quad $C=(14-5+(-12))-(14-20-3)$
    \end{exercice}\stepcounter{exo}


    \begin{exercice}
        Sans calculatrice, effectuer :

        $A=-5\times(-2)-17$ \quad
        $B=\big(15+2\times(-3)\big)\div(-3)$ \quad 
        $C=15-15\times$ \quad
        $D=\big(8\times(-3)+(-4)\times5\big)\div(-11)$
    \end{exercice}\stepcounter{exo}


    \begin{exercice}

        \medskip

        $A=(-3)\times(-4)\times(-5)\times6\times7\times8$ \quad
        $B=\dfrac{(-4)\times5\times(-3)}{6\times(-2)\times(-13)}$ \quad 
        $C=(-5)\times(-4-5)\times(-6+10)$  
    \end{exercice}
