\newcommand{\monTitre}{Opérations sur les nombres relatifs - Partie 2}

    
    \setcounter{exo}{1}
    \enteteActivite{1}{\monTitre}{\thechapter}

    \begin{exercice}
        \begin{enumerate}
            \item Décomposer le produit $A=5\times(-3)$ en somme et donne sa valeur.

            \item Même consigne avec $B=(-4)\times3$ puis $C=(-8)\times5$.

            \item Énoncer une règle permettant d'effectuer les multiplications entre deux nombres de signes différents.
            
            \item Compléter les égalités suivantes.

            a. $(-4)\times3=\points[4]$ \quad b. $5\times(-10)=\points[4]$ \quad 
            c. $\points[4]\times9=-81$ \quad  d. $(-7)\times\points[4]=-49$ 
        \end{enumerate}
    \end{exercice}\stepcounter{exo}


    \begin{exercice}
        \begin{enumerate}
            \item Calculer sans calculatrice.

            $A=26\times98+26\times2$ \quad $B=(-13)\times991+(-13)\times9$

            \item Même consigne.

            $C=(-79)\times5+(-79)\times(-5)$  \quad $D=(-23)\times(-4)+(-23)\times4$

            \item Que représente le produit $(-79)\times(-5)$ pour le produit $(-79)\times5$ ?

            Même question pour le produit $(-23)\times(-4)$ pour $(-23)\times4$ ?

            \item Énoncer une règle permettant d'effectuer les multiplications entre deux nombres de même signe.

            \item Compléter les égalités suivantes.

            a. $(-4)\times(-3)=\points[4]$ \quad b. $(-7)\times(-10)=\points[4]$ \quad 
            c. $\points[4]\times(-7)=63$ \quad  d. $(-6)\times\points[4]=54$ 
        \end{enumerate}
    \end{exercice}\stepcounter{exo}
