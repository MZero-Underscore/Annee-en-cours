\section{Multiplication et division}   
\subsection{Multiplier}
    
    \begin{propriete} 
        Le produit de deux nombres relatifs de signes différents est un nombre négatif. 
        
        La distance à zéro de ce produit est le produit des distances à zéro des 2 facteurs.
    \end{propriete}
    
    \begin{exemple}
        Effectuer les calculs suivants.

        $A = 3\times(-15)$ puis $B = -6 \times 4$
    \end{exemple}
    
    \begin{propriete} 
        Le produit de deux nombres négatifs est un nombre positif. 
        
        La distance à zéro de ce produit est le produit des distances à zéro des 2 facteurs.
    \end{propriete}
    
    \begin{exemple}
        Effectuer les calculs suivants.

        $A = -3\times(-15)$ puis $B = -4 \times (-1,5)$
    \end{exemple}
    
    \begin{propriete}
        Multiplier un nombre par \((-1)\) revient à prendre \textbf{l'opposé} du nombre.
    \end{propriete}
    
    \begin{exemple}
        Effectuer les calculs suivants.
        
        $A = 7\times (-1) \quad  B = -1 \times (-5)$ puis $C = -1,5\times (-1) $
    \end{exemple}
    
\subsection{Diviser}
    
    \begin{propriete} 
        Le quotient de deux nombres relatifs de signes différents est un nombre négatif. 
        
        La distance à zéro de ce quotient s'obtient en calculant le quotient des distances à zéro des deux nombres.
    \end{propriete}
    
    \begin{exemple}
        Effectuer les calculs suivants.
        
        $A = -21 \div 3 $ puis $B=91\div(-7)$
    \end{exemple}
    
    \begin{remarque} 
        La version littérale de la propriété ci-dessus peut s'écrire :
        
        Soit \( a \) et \( b \) deux nombres positifs (\( b \) non nul).
        \[
        \frac{-a}{b} = \frac{a}{-b} = -\frac{a}{b}
        \]
    \end{remarque}
    
    \begin{propriete} 
        Le quotient de deux nombres relatifs de même signe est un nombre positif. La distance à zéro de ce quotient s'obtient en calculant le quotient des distances à zéro des deux nombres.
    \end{propriete}
    
    \begin{exemple}
        Effectuer les calculs suivants.
        
        $A = -99 \div (-3) $ puis $B=-121\div(-11)$
    \end{exemple}
    
\subsection{Signe d'un produit ou d'un quotient}
    
    \begin{proprieteTitre}{Signe d'un produit/quotient}
        Pour \textbf{déterminer le signe d'une suite de multiplications et/ou de divisions}, il suffit de compter le \textbf{nombre de facteurs négatifs}.
        \begin{itemize}
            \item Si ce nombre est pair alors le résultat sera positif.
            \item Si ce nombre est impair alors le résultat sera négatif.
        \end{itemize}
    \end{proprieteTitre}
    
    \begin{exemple}
        \begin{enumerate}
            \item Sans effectuer les calculs, déterminer le signe des expressions suivantes.
            \begin{itemize}
                \item $A=(-25)\times4\times(-17)\div(-52)$
                \item $B=12\times(-4)\div(-27)\times 13$
                \item Attention aux pièges : $C=-15\times(-5+7)\times3\times(-4)$
            \end{itemize}
        
            \item Effectuer les calculs suivants.
            \begin{itemize}
                \item $A=5+(-5)\times4\times(-10)$
                \item $B=36\div(-4)+5\times(-12)\div(-3)$
            \end{itemize}
        \end{enumerate}
    \end{exemple}
