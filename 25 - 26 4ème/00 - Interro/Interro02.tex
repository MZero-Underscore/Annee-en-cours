    \begin{exerciceNote}{6}
        \begin{minipage}{0.5\linewidth}
             $CAT$ est rectangle en $C$ tel que $CA = 12$ cm et $CT = 17$ cm. 
    
            Calculer $AT$ au mm près.
    
            \begin{center}
                \begin{center}
                    \includegraphics[scale=0.5]{images/Interro02b_exo03a.png}
                \end{center}
            \end{center}
                
            Dans \textcolor{red}{le triangle CAT rectangle en C}
        
            D'après \textcolor{red}{le théorème de Pythagore} :
            \begin{align*}
                AT^2 & = CA^2+CT^2 \\
                AT^2 & = 12^2+17^2 \\
                AT^2 & = 144+289 \\
                AT^2 & = 433 \\ 
            \end{align*}
            Donc $AT = \sqrt{433} \simeq 20,8$ cm (arrondi au mm près)
            
        \end{minipage}
        \vline
        \hfill
        \begin{minipage}{0.49\linewidth}
             $SEL$ est rectangle en $E$ tel que $SL = 26$ m et $SE = 8$ m. 
    
            Calculer $EL$ au mm près.
    
            \begin{center}
                \begin{center}
                    \includegraphics[scale=0.5]{images/Interro02b_exo03b.png}
                \end{center}
            \end{center}
                
            Dans \textcolor{red}{le triangle SEL rectangle en E}
        
            D'après \textcolor{red}{le théorème de Pythagore} :
            \begin{align*}
                SL^2 & = SE^2+EL^2 \\
                26^2 & = 8^2+EL^2 \\
                676  & = 64+EL^2 \\
                EL^2 & = 676-64 \\
                EL^2 & = 612
            \end{align*}
            Donc $SL = \sqrt{612} \simeq 24,739$ m (arrondi au mm près)
        \end{minipage}
    \end{exerciceNote}\stepcounter{exo}


    \begin{exerciceNote}{2}
        \begin{minipage}{0.7\linewidth}
            Un navire quitte le port de Brest avec 100 passagers à bord. Le capitaine note ensuite l'évolution du nombre de passagers à chaque escale.

            \begin{enumerate}
                \item Combien de passagers y a-t-il dans le navire après la 5ème escale ?

                100+45-60+90+130-230=55

                \textcolor{red}{Il reste 55 passagers dans le bateau après la 5è escale.}

                \item Sachant que chaque passager transporte en moyenne 16,5 kg de bagages, quelle est la masse totale de bagages dans le navire après la 5ème escale.

                $16,5\times55=907,5$

                \textcolor{red}{La masse totale des bagages est de 907,5 kg.}
                
            \end{enumerate}

        \end{minipage}
