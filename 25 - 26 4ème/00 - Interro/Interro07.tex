    \begin{exerciceNote}{4}
        \begin{enumerate}[leftmargin=*]
            \item Les tableaux suivants sont-ils des tableaux de proportionnalité ? Justifier par un calcul.

            \begin{minipage}{0.48\linewidth}
                \begin{center}
                \begin{tabular}{|c|c|c|c|}
                \hline
                3 & 5 & 8 & 10 \\ \hline
                10,5 & 17,5 & 28 & 35 \\ \hline
                \end{tabular}
                \end{center}

                \medskip

                \begin{ZoneReponse}{2.5cm}
                On calcule les rapports : \\
                $\frac{10,5}{3} = 3,5$ ; $\frac{17,5}{5} = 3,5$ ; $\frac{28}{8} = 3,5$ ; $\frac{35}{10} = 3,5$. \\
                Tous les rapports sont égaux, c'est un \textbf{tableau de proportionnalité}.
                \end{ZoneReponse}
                
            \end{minipage}
            \hfill \vline \hfill
            \begin{minipage}{0.48\linewidth}
                \begin{center}
                \begin{tabular}{|c|c|c|c|}
                    \hline
                    2 & 4 & 6 & 9 \\ \hline
                    5 & 10 & 15 & 22 \\ \hline
                \end{tabular}
                \end{center}

                \medskip

                \begin{ZoneReponse}{2.5cm}
                $\frac{5}{2} = 2,5$ ; $\frac{10}{4} = 2,5$ ; $\frac{15}{6} = 2,5$ ;\\
                mais $\frac{22}{9} \approx 2,44$. \\
                Les rapports ne sont pas tous égaux ($2,5 \neq 2,44$), ce \textbf{n'est pas} de la proportionnalité.
                \end{ZoneReponse}
            \end{minipage}

            \medskip

            \item Compléter le tableau de proportionnalité suivant. Indiquer les calculs dans la zone prévue en dessous. 
            \begin{center}
                \begin{tabular}{|l|c|c|c|}
                \hline
                Masse d'oranges (kg) & 2,5 & 4 & \textbf{6} \\ \hline
                Prix payé (€) & 3,75 & \textbf{6} & 9 \\ \hline
                \end{tabular}
            \end{center}

            \begin{ZoneReponse}{1.8cm} 
            Le coefficient est $3,75 \div 2,5 = 1,5$ €/kg. \\
            Calculs : $4 \times 1,5 = \mathbf{6}$ € \ et \ $9 \div 1,5 = \mathbf{6}$ kg.
            \end{ZoneReponse}

            \item Compléter les égalités suivantes.

            \medskip

            \begin{minipage}{0.3\linewidth}
                \begin{center}                
                    3,42 m$^3$ = $\mathbf{3\,420}$ L
    
                    \medskip
    
                    3 h 20 min = $\mathbf{3,33}$ h 
                \end{center}
            \end{minipage}
            \hfill \vline \hfill
            \begin{minipage}{0.3\linewidth}
