\section{Vocabulaire des probabilités}

    \begin{definitionTitre}{Expérience aléatoire}
        Une expérience est dite \textbf{aléatoire} lorsqu'on en connaît tous les résultats possibles, mais qu'on ne peut pas prévoir avec certitude lequel va se produire.
    \end{definitionTitre}
    
    
    \begin{itemize}[label=$\bullet$]
        \item Chaque résultat possible d'une expérience aléatoire est appelé une \textbf{issue}.
        \item L'\textbf{univers} est l'ensemble de toutes les issues possibles.
    \end{itemize}

        On lance un dé équilibré à 6 faces et on regarde le nombre inscrit sur la face supérieure.

    \end{exemple}
    
    
    \begin{definitionTitre}{Événement}
        Un \textbf{événement} est un ensemble d'issues. Il est généralement noté par une lettre majuscule ($A$, $B$, $C$...).
        \begin{itemize}
            \item Un événement \textbf{élémentaire} ne contient qu'une seule issue.
            \item Un événement \textbf{impossible} ne peut pas se produire (probabilité égale à 0).
            \item Un événement \textbf{certain} se produit toujours (probabilité égale à 1).
        \end{itemize}
    \end{definitionTitre}

\section{Calculer une probabilité}
    
    \begin{proprieteTitre}{Loi d'équirépartition}
        Lorsque chaque issue a la même chance de se produire (on parle d'\textbf{équiprobabilité}), la probabilité d'un événement $A$ se calcule ainsi :
        \[ P(A) = \frac{\text{Nombre d'issues favorables à } A}{\text{Nombre total d'issues possibles}} \]
    \end{proprieteTitre}

    
    \begin{remarque}
        Une probabilité est un nombre toujours compris entre \textbf{0} et \textbf{1}. Elle peut s'exprimer sous forme d'une fraction, d'un nombre décimal ou d'un pourcentage.
    \end{remarque}

\section{Événement contraire}
    
    \begin{definition}
        L'événement \textbf{contraire} d'un événement $A$ est l'ensemble de toutes les issues qui ne réalisent pas $A$. On le note souvent $\overline{A}$ (lire \og $A$ barre \fg) ou \og non $A$ \fg.
    \end{definition}

    
    \begin{propriete}
        La somme de la probabilité d'un événement et de celle de son contraire est égale à 1 :
        \[ P(A) + P(\overline{A}) = 1 \quad \text{donc} \quad P(\overline{A}) = 1 - P(A) \]
    \end{propriete}

    
    \begin{exemple}
        Si la probabilité de gagner à un jeu est $P(G) = \frac{3}{10}$, alors la probabilité de perdre est :
        $P(\overline{G}) = 1 - \frac{3}{10} = \frac{7}{10}$.
    \end{exemple}

    
    \begin{exerciceTitre}{La roue de la fortune}
        Une roue de loterie est divisée en 20 secteurs identiques :
        \begin{itemize}
            \item 1 secteur permet de gagner le \textbf{Gros Lot} ;
            \item 4 secteurs permettent de gagner une \textbf{Peluche} ;
            \item Tous les autres secteurs sont \textbf{Perdants}.
        \end{itemize}
        
        On fait tourner la roue. On considère l'événement $G$ : \og Le joueur gagne un lot (Gros Lot ou Peluche) \fg.
        
        \begin{enumerate}
            \item Quelle est la probabilité $P(G)$ de gagner un lot ? 
            \item Définir par une phrase l'événement contraire $\overline{G}$.
            \item Calculer la probabilité $P(\overline{G})$ de ne rien gagner.
        \end{enumerate}
    \end{exerciceTitre}
