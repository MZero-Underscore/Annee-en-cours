    \begin{exerciceTitre}{Simplifier et réduire des expressions}
        Simplifier et réduire au maximum les expressions suivantes en regroupant les termes de même nature :
        
        \begin{minipage}{0.45\linewidth}
            \begin{enumerate}
                \item $C = 4x + 9 - 2x + 5$
                \item $D = 3x^2 - 5x + 7 - x^2 + 8x - 10$
                \item $E = 5 \times x \times 3 \times x - 7 \times x$
            \end{enumerate}
        \end{minipage}
        \hfill \vline \hfill
        \begin{minipage}{0.45\linewidth}
            \begin{enumerate}
                \setcounter{enumi}{3}
                \item $F = 8 - (-4x) + 12 - 10x$
                \item $G = 5x^2 + 8x - 4x^2 + 14x$
                \item $H = 4x\times2x^2-14x^3$
            \end{enumerate}
        \end{minipage}

        \begin{ZoneReponse}{5cm}
        \end{ZoneReponse}
    \end{exerciceTitre}
    
    \begin{exerciceTitre}{Développement simple (Distributivité)}
        Développer et simplifier les expressions suivantes.
        
        \begin{minipage}{0.45\linewidth}
            \begin{enumerate}
                \item $I = 3(4x + 5)$
                \item $J = x(7 - 2x)$
                \item $K = -5(2x - 4)$
            \end{enumerate}
        \end{minipage}
        \hfill \vline \hfill
        \begin{minipage}{0.45\linewidth}
            \begin{enumerate}
                \setcounter{enumi}{3}
                \item $L = 5x(2x-6)$
                \item $M = -5(5x+6x^2)$
                \item $N = -1(2x-6)$
            \end{enumerate}
        \end{minipage}

        \begin{ZoneReponse}{7cm}
        \end{ZoneReponse}
    \end{exerciceTitre}

    \begin{exerciceTitre}{Application géométrique}

            \item Exprimer l'aire $\mathcal{A}$ du rectangle en fonction de $x$. Développer et réduire l'expression.
            \item Calculer l'aire pour $x = 4$ cm.
        \end{enumerate}
        \begin{ZoneReponse}{7cm}
        \end{ZoneReponse}
    \end{exerciceTitre}
