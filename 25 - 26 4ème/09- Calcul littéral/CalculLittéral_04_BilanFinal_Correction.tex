\section{Évaluer et Simplifier}

    \begin{exercice}[Calculer une expression - CORRIGÉ]
    \begin{itemize}
        \item \textbf{Pour $x = 3$ :}
        \begin{enumerate}
            \item $A = 5 \times 3 - 7 = 15 - 7 = 8$
            \item $B = 3^2 - 4 \times 3 + 1 = 9 - 12 + 1 = -2$
            \item $C = (3 + 3)(2 \times 3 - 5) = 6 \times (6 - 5) = 6 \times 1 = 6$
        \end{enumerate}
        \item \textbf{Pour $x = -2$ :}
        \begin{enumerate}
            \item $A = 5 \times (-2) - 7 = -10 - 7 = -17$
            \item $B = (-2)^2 - 4 \times (-2) + 1 = 4 + 8 + 1 = 13$
            \item $C = (-2 + 3)(2 \times (-2) - 5) = 1 \times (-4 - 5) = 1 \times (-9) = -9$
        \end{enumerate}
    \end{itemize}
    \end{exercice}
    
    \begin{exercice}[Simplifier une écriture - CORRIGÉ]
    \begin{enumerate}
        \item $D = 3a + 20$
        \item $E = x^2 + 2xy$
        \item $F = n - 6b$ \textit{(Note : $1 \times n = n$ et $4+2=6$)}
        \item $G = 7(x + 3)$ \textit{(Note : $0 \times x = 0$)}
    \end{enumerate}
    \end{exercice}
    
    \section{Réduction et Développement}
    
    \begin{exercice}[Réduire une expression - CORRIGÉ]
    \begin{enumerate}
        \item $H = 4x - 2x + 9 + 3 = 2x + 12$
        \item $I = 5a^2 - 4a^2 - 3a + 8a + 2 = a^2 + 5a + 2$
        \item $J = 7x - 3x - 4 + 2x - 5 = 6x - 9$
    \end{enumerate}
    \end{exercice}
    
    \begin{exercice}[Simple Distributivité - CORRIGÉ]
    \begin{enumerate}
        \item $K = 3 \times x + 3 \times 4 = 3x + 12$
        \item $L = -2 \times 5x - (-2) \times 3 = -10x + 6$
        \item $M = 2x^2 + 7x - 3x^2 = -x^2 + 7x$
    \end{enumerate}
    \end{exercice}
    
    \begin{exercice}[Double Distributivité - CORRIGÉ]
    \begin{enumerate}
        \item $N = x^2 + 5x + 3x + 15 = x^2 + 8x + 15$
        \item $O = 6x^2 + 2x - 12x - 4 = 6x^2 - 10x - 4$
        \item $P = 10x - 15 - 2x^2 + 3x = -2x^2 + 13x - 15$
    \end{enumerate}
    \end{exercice}
    
    \section{Factorisation et Défis}
    
    \begin{exercice}[Factoriser - CORRIGÉ]
    \begin{enumerate}
        \item $Q = 5(x + 3)$ \textit{(facteur commun : 5)}
        \item $R = a(7a - 3)$ \textit{(facteur commun : $a$)}
        \item $S = 4x(x + 2)$ \textit{(facteur commun : $4x$)}
    \end{enumerate}
    \end{exercice}
    
    \begin{exercice}[Pyramide de calcul - CORRIGÉ]
    \begin{itemize}
        \item Étage 2 (gauche) : $2x + 3$
        \item Étage 2 (droite) : $3 + x - 5 = x - 2$
        \item \textbf{Sommet :} $(2x + 3) + (x - 2) = 3x + 1$
    \end{itemize}
    \end{exercice}
    
    \begin{exercice}[Défi : Aire - CORRIGÉ]
    \begin{enumerate}
        \item $\mathcal{A}_1 = (x+5)^2 = (x+5)(x+5) = x^2 + 10x + 25$
        \item $\mathcal{A}_2 = (x+2)^2 = (x+2)(x+2) = x^2 + 4x + 4$
        \item $\text{Aire grise} = \mathcal{A}_1 - \mathcal{A}_2 = (x^2 + 10x + 25) - (x^2 + 4x + 4)$ \\
        $\text{Aire grise} = x^2 + 10x + 25 - x^2 - 4x - 4 = 6x + 21$
    \end{enumerate}
    \end{exercice}
    
    \begin{exercice}[Programme de calcul - CORRIGÉ]
    \begin{enumerate}
        \item Pour 4 : $(4 - 10) \times 3 = -6 \times 3 = -18$. Pour -3 : $(-3 - 10) \times 3 = -13 \times 3 = -39$.
        \item $R = 3(n - 10)$ ou $R = 3n - 30$.
        \item \textbf{Bonus :} On résout $3(n - 10) = -9 \Rightarrow n - 10 = -3 \Rightarrow n = 7$.
    \end{enumerate}
    \end{exercice}
