\section{Rappels}

    \begin{definition} 
        Soit \( a \) et \( b \) deux nombres quelconques (\( b \) est non nul). 
        
        Le nombre qui, multiplié par \( b \), donne \( a \), s'appelle \textbf{le quotient de \( a \) par \( b \)}.         
        On le note \( \frac{a}{b} \).

        \[  b \times \boxed{\frac{a}{b}} = a \] 
    \end{definition}
    

    \begin{remarque}
        Le nombre $\frac{a}{b}$ est le résultat de la division de \( a \) par \( b \).
    \end{remarque}
    
     
    \begin{exemple} 
        Compléter les égalités suivantes pour qu'elles soient vraies.
        
        \smallskip

        \begin{minipage}{0.24\linewidth}
            \[4\times \ldots = 28\]
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.24\linewidth}
            \[\ldots \times 7 = 2\]

        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.24\linewidth}
            \[3\times \ldots = 14\]
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.24\linewidth}
            \[10\times \ldots = 5\]
        \end{minipage}
    \end{exemple}
    
    
    \begin{definition}
        Lorsque les nombres \( a \) et \( b \) sont deux nombres entiers  ($b$ non nul), on dit que \( \frac{a}{b} \) est une \textbf{fraction}.
    \end{definition}
    
    
    \begin{definition} 
        \textbf{Simplifier une fraction}, c'est l'écrire avec les plus petits nombres entiers possibles au numérateur et au dénominateur. 
        
        On fait en sorte que le plus grand diviseur commun du numérateur et du dénominateur soit 1.

        La fraction obtenue est dite $\textbf{irréductible}$.
    \end{definition}


    \begin{exemple}
        Donner l'écriture simplifiée des fractions suivantes.

        \begin{minipage}{0.24\linewidth}
            $A=\dfrac{15}{25}$

            \vspace{1cm}
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.24\linewidth}
            $B=\dfrac{77}{88}$

            \vspace{1cm}
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.24\linewidth}
            $C=\dfrac{36}{28}$

            \vspace{1cm}
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.24\linewidth}
            $D=\dfrac{45}{95}$

            \vspace{1cm}
        \end{minipage}
    \end{exemple}
    
    
    \section{Additionner et soustraire}
    
    \begin{definition}
        Pour \textbf{additionner (ou soustraire)} deux nombres en écriture fractionnaire :
        \begin{itemize}
            \item On réduit les nombres au même dénominateur.
            \item On additionne (ou on soustrait) les numérateurs et on garde le dénominateur commun.
        \end{itemize}
    \end{definition}
    
    \begin{exemple} 
        Calculer les expressions suivantes. Donner les résultats sous forme irréductible.

        \begin{minipage}{0.24\linewidth}
            $A=\dfrac{3}{5}+\dfrac{7}{10}$

            \vspace{1.5cm}
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.24\linewidth}
            $B=\dfrac{-2}{7}-\dfrac{2}{3}$

            \vspace{1.5cm}
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.24\linewidth}
            $C=-4+\dfrac{55}{3}$

            \vspace{1.5cm}
        \end{minipage}
    \end{exemple}
    
    
\section{Multiplier}
\subsection{Cas général}
    
    \begin{definition}
        Pour \textbf{multiplier} des nombres en écriture fractionnaire, on multiplie les numérateurs entre eux et les dénominateurs entre eux.
    \end{definition}
    
    \begin{exemple} 
         Calculer les expressions suivantes. Donner les résultats sous forme irréductible.

        \begin{minipage}{0.3\linewidth}
            $A=\dfrac{4}{5}\times\dfrac{3}{7}$

            \vspace{1.5cm}
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.3\linewidth}
            $B=\dfrac{-4}{9}\times\dfrac{5}{-3}$

            \vspace{1.5cm}
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.3\linewidth}
            $C=-5\times\dfrac{5}{7}$

            \vspace{1.5cm}
        \end{minipage}
    \end{exemple}
    
    
\subsection{Simplifier avant de calculer}
    
    Lorsqu'on doit multiplier deux fractions avec des nombres entiers un peu grands, on commence par décomposer ces nombres sous forme de produits et on cherche des simplifications avant de se lancer dans les multiplications.
    
    \begin{exemple}
         Effectuer les opérations suivantes. Donner les résultats sous forme irréductible.

         \begin{minipage}{0.45\linewidth}
            $D=\dfrac{44}{15}\times\dfrac{25}{33}$

            \vspace{1.5cm}
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.45\linewidth}
            $E=\dfrac{18}{-35}\times\dfrac{-14}{27}$

            \vspace{1.5cm}
        \end{minipage}
    \end{exemple}

\section{Diviser}
    
    \begin{definition} 
        Un nombre et \textbf{son inverse} sont tels que leur produit vaut 1. Pour un nombre \( x \) non nul, on note son inverse \( \dfrac{1}{x} \).
    \end{definition}
    
    \begin{exemple} 
        Donner l'inverse des nombres suivants : 

        \medskip

        \begin{minipage}{0.24\linewidth}
            $4$

            \vspace{0.5cm}
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.24\linewidth}
            $\dfrac{2}{3}$

            \vspace{0.5cm}
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.24\linewidth}
            $\dfrac{-7}{3}$

            \vspace{0.5cm}
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.24\linewidth}
            $\dfrac{-1}{5}$

            \vspace{0.5cm}
        \end{minipage}
    \end{exemple}

    
    \begin{remarque}
        \begin{itemize}
            \item Zéro n'a pas d'inverse.
            \item Un nombre et son inverse sont de même signe.
        \end{itemize}
    \end{remarque}
    
    \begin{propriete} 
        Diviser par un nombre non nul revient à multiplier par son inverse.
    \end{propriete}
      
   
    \begin{exemple} 
        Calculer les expressions suivantes. Donner les résultats sous forme irréductible.

        \begin{minipage}{0.3\linewidth}
            $A=\dfrac{5}{6}\div\dfrac{7}{3}$

            \vspace{1.5cm}
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.3\linewidth}
            $B=3\div\dfrac{-2}{5}$

            \vspace{1.5cm}
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.3\linewidth}
            $C=\dfrac{8}{5}\div(-4)$

            \vspace{1.5cm}
        \end{minipage}
    \end{exemple}

\section{Comparer des fractions}
\subsection{Cas général}

    \begin{propriete}
        Pour \textbf{comparer} deux fractions, on commence par les réduire au même dénominateur. On compare ensuite les numérateurs obtenus.
    \end{propriete}

    \begin{exemple}
        \begin{enumerate}
            \item Comparer les couples de fractions suivants.

            \begin{minipage}{0.3\linewidth}
                $\dfrac{1}{2}$ et $\dfrac{5}{6}$
    
                \vspace{2cm}
            \end{minipage}
            \hfill\vline\hfill
            \begin{minipage}{0.3\linewidth}
                $\dfrac{7}{5}$ et $\dfrac{4}{3}$
    
                \vspace{2cm}
            \end{minipage}
            \hfill\vline\hfill
            \begin{minipage}{0.3\linewidth}
                $\dfrac{7}{5}$ et $\dfrac{4}{3}$
    
                \vspace{1.5cm}
            \end{minipage}

            \item Ranger les fractions suivantes par ordre croissant.

            $-\dfrac{1}{4} \quad -\dfrac{5}{4} \quad \dfrac{5}{9} \quad \dfrac{4}{7} \quad \dfrac{3}{4} \quad -\dfrac{9}{2} \quad \dfrac{6}{4} \quad \dfrac{9}{7}$

            \vspace{3cm}
            
        \end{enumerate}
    \end{exemple}

\subsection{Cas particuliers}

    \begin{propriete}
        \begin{itemize}
            \item Si deux fractions ont le même numérateur alors la plus grande est celle qui possède le plus petit dénominateur.
            \item Une fraction dont le numérateur est plus petit que le dénominateur est plus petite que 1.
        \end{itemize}
    \end{propriete}


    \begin{exemple}
        Comparer les couples de fractions suivants.

        \begin{minipage}{0.3\linewidth}
            $\dfrac{4}{7}$ et $\dfrac{4}{9}$

            \vspace{2cm}
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.3\linewidth}
            $\dfrac{5}{11}$ et $\dfrac{5}{7}$

            \vspace{2cm}
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.3\linewidth}
            $\dfrac{7}{5}$ et $\dfrac{1}{7}$

            \vspace{1.5cm}
        \end{minipage}
    \end{exemple}
