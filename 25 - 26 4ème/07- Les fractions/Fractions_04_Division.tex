    \begin{exercice}
        Compléter les égalités suivantes.
        
        \medskip
        
        \begin{minipage}{0.24\linewidth}
            \[\dfrac{3}{4}\times\points[2] = \dfrac{15}{28}\]

            \[\dfrac{3}{7}\times\points[2] = \dfrac{21}{21}\]
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.24\linewidth}
            \[\dfrac{12}{5}\times\points[2] = \dfrac{36}{25}\]

            \[\dfrac{5}{8}\times\points[2] = \dfrac{40}{40}\]
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.24\linewidth}
            \[\dfrac{3}{4}\times\points[2] = \dfrac{3}{7}\]

            \[\dfrac{13}{6}\times\points[2] = 1\]
        \end{minipage}
        \hfill\vline\hfill
        \begin{minipage}{0.24\linewidth}
              \[\dfrac{15}{14}\times\points[2] = \dfrac{7}{5}\]

            \[\dfrac{14}{-5}\times\points[2] = 1\]
        \end{minipage}
    \end{exercice}\stepcounter{exo}


    \begin{exercice}
        \begin{enumerate}[leftmargin=*]
            \item Essayer de comprendre cette suite de calculs.

            $A=\dfrac{4}{7}\div\dfrac{3}{5}
            =\dfrac{\dfrac{4}{7}}{\dfrac{3}{5}}
            =\dfrac{\dfrac{4}{7}\times\dfrac{5}{3}}{\dfrac{3}{5}\times\dfrac{5}{3}}
            =\dfrac{\dfrac{4}{7}\times\dfrac{5}{3}}{1}
            =\dfrac{4}{7}\times\dfrac{5}{3}
            =\dfrac{20}{21}$

            \item Essayer de reproduire cet enchaînement pour calculer : 

            \bigskip

            $B=\dfrac{10}{3}\div\dfrac{7}{11}$

            \bigskip

            \item Même consigne avec :

            \bigskip

            $C=\dfrac{14}{33}\div\dfrac{35}{44}$

            \bigskip

            \item En transformant directement la division en multiplication, effectuer les calculs suivants :

                    \medskip
        
            \begin{minipage}{0.30\linewidth}
                $D=\dfrac{4}{5}\div\dfrac{7}{2}$
            \end{minipage}
            \hfill\vline\hfill
            \begin{minipage}{0.30\linewidth}
                $E=\dfrac{2}{9}\div\dfrac{11}{5}$
            \end{minipage}
            \hfill\vline\hfill
            \begin{minipage}{0.30\linewidth}
                $E=\dfrac{4}{5}\div8$
            \end{minipage}
        \end{enumerate}
    \end{exercice}\stepcounter{exo}
