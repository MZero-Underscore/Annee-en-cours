\enteteActivite{\monTitre}{1}

    \begin{exercice}
        \begin{enumerate}[leftmargin=*]
            \item Donner la liste des 15 premiers multiples de 3 (en commençant par 3 lui même).

            \item Parmi les nombres suivants, lesquels sont-ils dans la table de 3 ?
            \[6; 22; 35; 39; 156; 729; 742\]

            \item Parmi les nombres suivants, lesquels sont dans la table de 5 ?
            \[75; 81; 124; 155; 225; 1000\]

            \item Donner 4 diviseurs distincts de 160.

            \item Poser la division de 134 par 5, donner le quotient et le reste.

            \item Écrire 42 sous la forme d'un produit avec les plus petits nombres entiers possibles.

            \item Faire de même avec 180.
        \end{enumerate}
    \end{exercice}
