\section{Vocabulaire et notions de base}
    \begin{definitionTitre}{Équation et inconnue}
        Une \textbf{équation} est une égalité mathématique qui comporte un ou plusieurs nombres inconnus. Ces nombres sont cachés derrière des lettres (le plus souvent $x$), appelées les \textbf{inconnues}.
    \end{definitionTitre}

    
    \begin{definitionTitre}{Solution d'une équation}
        Une \textbf{solution} d'une équation est une valeur de l'inconnue pour laquelle l'égalité est vraie.
        
        \textbf{Résoudre} une équation, c'est trouver \textbf{toutes} les solutions possibles de cette équation.
    \end{definitionTitre}


    \begin{remarque}
        \begin{itemize}
            \item Une équations peut avoir une solution, plusieurs solutions, voire aucune.
            \item les expressions placées à gauche et à droite du symbole \og = \fg s'appellent les \textbf{membres} de l’équation.
        \end{itemize}    
    \end{remarque}

    
    \begin{exemple}
        On considère l'équation suivante, dont l'inconnue est $x$ : \quad $3x - 4 = 5x+8$.
        \begin{enumerate}
            \item Le nombre $2$ est-il une solution de cette équation ?
            \item Le nombre $-6$ est-il une solution de cette équation ?
        \end{enumerate}

        \end{ZoneReponse}
    \end{exemple}
    
    
\section{Les règles de la « balance »}
    Pour résoudre une équation de manière systématique sans deviner le résultat, on utilise les deux propriétés fondamentales des égalités.
    
    
    \begin{proprieteTitre}{Opérations sur les égalités}
        \begin{itemize}
            \item \underline{Règle n°1} : Une égalité reste vraie si on \textbf{ajoute} ou si on \textbf{soustrait} un même nombre à ses deux membres.
            \item \underline{Règle n°2} : Une égalité reste vraie si on \textbf{multiplie} ou si on \textbf{divise} ses deux membres par un même nombre non nul.
        \end{itemize}  
    \end{proprieteTitre}

    
    \begin{remarque}
        \textbf{Une équation fonctionne comme une balance en équilibre.}
        
        Tout ce qui est effectué sur le membre de gauche (à gauche du signe $=$) doit obligatoirement être effectué sur le membre de droite afin de maintenir l'équilibre.
    \end{remarque}
