    \begin{exerciceTitre}{Échauffement : Équations techniques}
        \begin{enumerate}
            \item $5x - (2x + 3) = 4 + 2(x - 1)$ \\
            On supprime les parenthèses et on développe :
            \begin{align*}
                5x - 2x - 3 &= 4 + 2x - 2 \\
                3x - 3 &= 2x + 2 \\
                3x - 2x &= 2 + 3 \\
                \mathbf{x} &\mathbf{= 5}
            \end{align*}
        
            \item $3(2x - 4) - 2(x + 5) = 6x - 2$ \\
            On développe les deux blocs :
            \begin{align*}
                6x - 12 - 2x - 10 &= 6x - 2 \\
                4x - 22 &= 6x - 2 \\
                4x - 6x &= -2 + 22 \\
                -2x &= 20 \\
                x &= \frac{20}{-2} \\
                \mathbf{x} &\mathbf{= -10}
            \end{align*}
        
            \item $\frac{2x}{3} + 4 = x - 1$ \\
            On multiplie chaque terme par 3 pour éliminer le dénominateur :
            \begin{align*}
                2x + 12 &= 3(x - 1) \\
                2x + 12 &= 3x - 3 \\
                2x - 3x &= -3 - 12 \\
                -x &= -15 \\
                \mathbf{x} &\mathbf{= 15}
            \end{align*}
        
            \item $4x - 7 = 9x + 18$
            \begin{align*}
                4x - 9x &= 18 + 7 \\
                -5x &= 25 \\
                x &= \frac{25}{-5} \\
                \mathbf{x} &\mathbf{= -5}
            \end{align*}
        \end{enumerate}
    \end{exerciceTitre}
    
    
    \begin{exerciceTitre}{Les deux terrains (Égalité de périmètres)}
        \begin{itemize}
            \item[$\bullet$] \textbf{Étape 1 : Expressions des périmètres} \\
            $\mathcal{P}_{\text{triangle}} = 3 \times (x + 4) = 3x + 12$ \\
            $\mathcal{P}_{\text{rectangle}} = 2 \times (\text{Longueur} + \text{Largeur}) = 2 \times (x + 1 + x) = 2 \times (2x + 1) = 4x + 2$
            
            \item[$\bullet$] \textbf{Étape 2 : Mise en équation et résolution} \\
            On cherche $x$ tel que $\mathcal{P}_{\text{triangle}} = \mathcal{P}_{\text{rectangle}}$ :
            \begin{align*}
                3x + 12 &= 4x + 2 \\
                12 - 2 &= 4x - 3x \\
                \mathbf{x} &\mathbf{= 10\text{ m}}
            \end{align*}
            
            \item[$\bullet$] \textbf{Étape 3 : Calcul du périmètre commun} \\
            En remplaçant $x$ par 10 : $\mathcal{P} = 3 \times (10 + 4) = 3 \times 14 = \mathbf{42\text{ m}}$. \\
            (Vérification avec le rectangle : $4 \times 10 + 2 = 42\text{ m}$).
        \end{itemize}
    \end{exerciceTitre}
    
    
    \begin{exerciceTitre}{L'extension du rectangle (Égalité d'aires)}
        \begin{enumerate}
            \item \textbf{Nouvelles dimensions :} La longueur devient $7 + x$ et la largeur devient $4 - 1 = 3\text{ cm}$.
            \item \textbf{Mise en équation et résolution :} \\
            Aire initiale $= 7 \times 4 = 28\text{ cm}^2$. \\
            Nouvelle aire $= 3 \times (7 + x) = 21 + 3x$.
            \begin{align*}
                21 + 3x &= 28 \\
                3x &= 28 - 21 \\
                3x &= 7 \\
                \mathbf{x} &\mathbf{= \frac{7}{3}\text{ cm}} \approx 2,33\text{ cm}
            \end{align*}
        \end{enumerate}
    \end{exerciceTitre}
    
    
    \begin{exerciceTitre}{Le défi de Pythagore}
        \begin{enumerate}
            \item Comme $x > 1$, on a $x + 7 > x - 1$ et $x + 7 < x + 7$. Le côté le plus long est donc $\mathbf{AC = x + 7}$.
            \item Si le triangle est rectangle, d'après le théorème de Pythagore, on doit avoir : $AC^2 = AB^2 + BC^2$.
            \begin{align*}
                (x + 7)^2 &= (x - 1)^2 + (x + 6)^2 \\
                x^2 + 14x + 49 &= (x^2 - 2x + 1) + (x^2 + 12x + 36) \\
                x^2 + 14x + 49 &= 2x^2 + 10x + 37 \\
                0 &= 2x^2 - x^2 + 10x - 14x + 37 - 49 \\
                \mathbf{x^2 - 4x - 12} &\mathbf{= 0}
            \end{align*}
            Cette équation est équivalente à $(x - 6)(x + 2) = 0$. Un produit de facteurs est nul si l'un au moins des facteurs est nul :
            \begin{center}
                $x - 6 = 0 \implies \mathbf{x = 6}$ \quad ou \quad $x + 2 = 0 \implies \mathbf{x = -2}$
            \end{center}
            Comme une longueur géométrique doit être positive, on élimine la solution négative. La seule valeur possible est donc $\mathbf{x = 6}$.
            \item Les longueurs exactes des trois côtés sont :
            \begin{itemize}
                \item $AB = 6 - 1 = \mathbf{5\text{ cm}}$
                \item $BC = 6 + 6 = \mathbf{12\text{ cm}}$
                \item $AC = 6 + 7 = \mathbf{13\text{ cm}}$
            \end{itemize}
        \end{enumerate}
    \end{exerciceTitre}
    
    
    \begin{exerciceTitre}{Le rectangle d'or caché}
        \begin{enumerate}
            \item \textbf{Aire totale :} $\mathcal{A} = \text{Aire}_A + \text{Aire}_{\text{Carré}} + \text{Aire}_B = 2x + x^2 + 5x = \mathbf{x^2 + 7x}$.
            \item 
            \begin{enumerate}
                \item L'aire totale étant de $90\text{ cm}^2$, on obtient bien l'équation : $\mathbf{x^2 + 7x = 90}$.
                \item Développons l'expression proposée :
                \begin{align*}
                    (x - 5)(x + 18) &= x^2 + 18x - 5x - 90 \\
                    &= x^2 + 7x - 90
                \end{align*}
                L'équation $x^2 + 7x = 90$ étant équivalente à $x^2 + 7x - 90 = 0$, elle revient bien à résoudre $\mathbf{(x - 5)(x + 18) = 0}$.
                \item Un produit de facteurs est nul si l'un au moins des facteurs est nul :
                \begin{center}
                    $x - 5 = 0 \implies \mathbf{x = 5}$ \quad ou \quad $x + 18 = 0 \implies \mathbf{x = -18}$
                \end{center}
                Une longueur ne pouvant pas être négative, on rejette $x = -18$. La solution du problème est donc $\mathbf{x = 5\text{ cm}}$.
            \end{enumerate}
        \end{enumerate}
    \end{exerciceTitre}

    \begin{exerciceTitre}{Le trapèze partagé}
        \begin{itemize}
            \item Aire de la partie orange (rectangle) : $\mathcal{A}_{\text{orange}} = 4 \times x = 4x$
            \item Aire de la partie verte (triangle) : $\mathcal{A}_{\text{verte}} = \frac{4 \times (x - 2)}{2} = 2(x - 2) = 2x - 4$
            \item \textbf{Mise en équation et résolution :} On souhaite que $\mathcal{A}_{\text{orange}} = 3 \times \mathcal{A}_{\text{verte}}$.
            \begin{align*}
                4x &= 3 \times (2x - 4) \\
                4x &= 6x - 12 \\
                4x - 6x &= -12 \\
                -2x &= -12 \\
                x &= \frac{-12}{-2} \\
                \mathbf{x} &\mathbf{= 6\text{ cm}}
            \end{align*}
        \end{itemize}
    \end{exerciceTitre}
    
    
    \begin{exerciceTitre}{La magie des nombres consécutifs}
        \begin{enumerate}
            \item Si le premier nombre est $x,$, les deux nombres entiers suivants sont $\mathbf{x + 1}$ et $\mathbf{x + 2}$.
            \item \textbf{Mise en équation et résolution :}
            \begin{align*}
                x + (x + 1) + (x + 2) &= 201 \\
                3x + 3 &= 201 \\
                3x &= 201 - 3 \\
                3x &= 198 \\
                x &= \frac{198}{3} \\
                \mathbf{x} &\mathbf{= 66}
            \end{align*}
            \item Les trois nombres entiers consécutifs cherchés sont donc $\mathbf{66}$, $\mathbf{67}$ et $\mathbf{68}$.
            \item \textbf{Défi :} Traduisons la contrainte par une équation :
            \begin{align*}
                3x + 3 &= 2026 \\
                3x &= 2023 \\
                x &= \frac{2023}{3} \approx 674,33
            \end{align*}
            La solution de l'équation n'étant pas un nombre entier, il est \textbf{impossible} de trouver trois entiers consécutifs dont la somme vaut $2026$.
        \end{enumerate}
    \end{exerciceTitre}
    
    
    \begin{exerciceTitre}{Le choix du programme}
        \begin{enumerate}
            \item 
            \begin{itemize}
                \item Avec le nombre $5$ : $5 \longrightarrow 5 + 3 = 8 \longrightarrow 8 \times 4 = 32 \longrightarrow 32 - 2 \times 5 = \mathbf{22}$.
                \item Avec le nombre $-2$ : $-2 \longrightarrow -2 + 3 = 1 \longrightarrow 1 \times 4 = 4 \longrightarrow 4 - 2 \times (-2) = 4 + 4 = \mathbf{8}$.
                \item \textit{Remarque :} On peut conjecturer que le résultat final correspond toujours au double du nombre de départ auquel on ajoute 12.
            \end{itemize}
            \item 
            \begin{enumerate}
                \item Expression du résultat final en fonction de $n$ : $\mathbf{4(n + 3) - 2n}$.
                \item Développement et réduction : $4n + 12 - 2n = \mathbf{2n + 12}$. Notre conjecture est validée.
            \end{enumerate}
            \item On cherche à résoudre l'équation :
            \begin{align*}
                2n + 12 &= 42 \\
                2n &= 42 - 12 \\
                2n &= 30 \\
                n &= \frac{30}{2} \\
                \mathbf{n} &\mathbf{= 15}
            \end{align*}
            Pour obtenir $42$, il fallait choisir le nombre $\mathbf{15}$ au départ.
        \end{enumerate}
    \end{exerciceTitre}
