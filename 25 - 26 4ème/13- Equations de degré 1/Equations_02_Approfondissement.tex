    \begin{tcolorbox}[colback=yellow!5, colframe=yellow!60!black, title=Méthode : Résoudre un problème géométrique, arc=2mm]
    Pour résoudre les problèmes, suis rigoureusement les 4 étapes suivantes :
    \begin{enumerate}
        \item \textbf{Choix de l'inconnue :} Repère la grandeur cherchée et nomme-la $x$ (pense à préciser l'unité).
        \item \textbf{Mise en équation :} Traduis les propriétés géométriques (aire, périmètre, égalité) à l'aide d'une expression contenant $x$.
        \item \textbf{Résolution :} Résous l'équation proprement étape par étape.
        \item \textbf{Conclusion :} Vérifie la cohérence géométrique du résultat (une longueur ne peut pas être négative !) et réponds à la question par une phrase.
    \end{enumerate}
    \end{tcolorbox}

    \begin{exerciceTitre}{Les deux terrains (Égalité de périmètres)}
        Un agriculteur possède deux terrains adjacents comme modélisés ci-dessous :
        \begin{itemize}
            \item Le premier est un \textbf{triangle équilatéral} de côté $x + 4$ (en mètres).
            \item Le deuxième est un \textbf{rectangle} de largeur $x$ et de longueur $x + 1$ (en mètres).
        \end{itemize}
        
        \noindent
        \begin{minipage}{0.45\textwidth}
            \centering
            \begin{tikzpicture}[scale=0.7]
                % Triangle équilatéral
                \coordinate (A) at (0,0);
                \coordinate (B) at (3,0);
                \coordinate (C) at (1.5,2.598);
                \draw[very thick, blue] (A) -- (B) -- (C) -- cycle;
                \node[below] at (1.5,0) {$x+4$};
                % Codage triangle équilatéral
                \node[blue] at (0.6,1.3) {$\times$};
                \node[blue] at (2.4,1.3) {$\times$};
                \node[blue] at (1.5,-0.1) {$\times$};
                \node[above] at (1.5,2.6) {Terrain 1};
            \end{tikzpicture}
        \end{minipage}
        \hfill
        \begin{minipage}{0.45\textwidth}
            \centering
            \begin{tikzpicture}[scale=0.7]
                % Rectangle
                \draw[very thick, ForestGreen] (0,0) rectangle (4,2.5);
                \node[left] at (0,1.25) {$x$};
                \node[below] at (2,0) {$x+1$};
                % Angles droits
                \draw (0,0.3) -- (0.3,0.3) -- (0.3,0);
                \draw (4,0.3) -- (3.7,0.3) -- (3.7,0);
                \node[above] at (2,2.5) {Terrain 2};
            \end{tikzpicture}
        \end{minipage}
        
        \vspace{0.2cm}
        Déterminer la valeur de $x$ pour laquelle le périmètre du triangle est \textbf{égal} au périmètre du rectangle. Calculer alors ce périmètre commun.
    \end{exerciceTitre}


    \begin{exerciceTitre}{L'extension du rectangle (Égalité d'aires)}
        On considère un rectangle de largeur $4\text{ cm}$ et de longueur $7\text{ cm}$. 
        On augmente sa longueur de $x\text{ cm}$ et on diminue sa largeur de $1\text{ cm}$.
        
        \noindent
        \begin{minipage}{0.5\textwidth}
            \begin{enumerate}
                \item Réaliser un schéma à main levée de la nouvelle situation en y inscrivant les nouvelles dimensions en fonction de $x$.
                \item Déterminer la valeur de $x$ pour que l'aire du nouveau rectangle soit \textbf{égale} à l'aire du rectangle initial.
            \end{enumerate}
        \end{minipage}
        \hfill
        \begin{minipage}{0.45\textwidth}
            \centering
            \begin{tikzpicture}[scale=0.6]
                \draw[thick, gray, dashed] (0,0) rectangle (7,4);
                \draw[very thick, purple] (0,0) rectangle (10,3);
                \node[above, gray] at (3.5,4) {$7\text{ cm}$};
                \node[left, gray] at (0,2) {$4\text{ cm}$};
                \draw[<->, purple] (7,-0.4) -- (10,-0.4) node[midway, below] {$+x$};
                \draw[<->, purple] (10.4,3) -- (10.4,4) node[midway, right] {$-1$};
            \end{tikzpicture}
        \end{minipage}
    \end{exerciceTitre}


    \begin{exerciceTitre}{Le défi de Pythagore}
        Un triangle $ABC$ a des côtés dont les longueurs dépendent d'un nombre $x$ supérieur à $1$ :
        $$ AB = x - 1 \quad , \quad BC = x + 7 \quad \text{et} \quad AC = x + 8 $$
        
        \begin{enumerate}
            \item Quel est le côté le plus long de ce triangle ? Justifier.
            \item Déterminer la valeur de $x$ pour que le triangle $ABC$ soit \textbf{rectangle}.
            \item \textit{Question subsidiaire :} Quelles sont alors les longueurs exactes de ses trois côtés ?
        \end{enumerate}
    \end{exerciceTitre}

    \begin{exerciceTitre}{Le rectangle d'or caché}
        On donne la figure ci-dessous, composée d'un carré central de côté $x$ (en cm) entouré par deux rectangles de dimensions fixes ($2\text{ cm} \times x\text{ cm}$ et $5\text{ cm} \times x\text{ cm}$).
        
        \begin{center}
            \begin{tikzpicture}[scale=0.7]
                % Bloc 2x
                \draw[very thick, fill=gray!10] (0,0) rectangle (1.5,3);
                \node at (0.75,1.5) {A};
                \node[below] at (0.75,0) {$2$};
                
                % Carré x*x
                \draw[very thick, fill=blue!5] (1.5,0) rectangle (4.5,3);
                \node at (3,1.5) {Carré};
                \node[below] at (3,0) {$x$};
                \node[right] at (4.5,1.5) {$x$};
                
                % Bloc 5x
                \draw[very thick, fill=gray!10] (4.5,0) rectangle (7,3);
                \node at (5.75,1.5) {B};
                \node[below] at (5.75,0) {$5$};
            \end{tikzpicture}
        \end{center}
        
        \begin{enumerate}
            \item Exprimer l'aire totale de la figure en fonction de $x$.
            \item On sait que l'aire totale de la figure est égale à $90\text{ cm}^2$. 
            \begin{enumerate}
                \item Montrer que cette situation se traduit par l'équation $x^2 + 7x = 90$.
                \item \textit{Le piège :} Cette équation comporte un terme en $x^2$. Montrer que l'équation est équivalente à : $(x - 5)(x + 12) = 0$.
                \item Résoudre cette équation (équation produit-nul) et en déduire la valeur de $x$. Expliquer pourquoi l'une des solutions doit être éliminée.
            \end{enumerate}
        \end{enumerate}
    \end{exerciceTitre}


    \begin{exerciceTitre}{Le trapèze partagé}
        On considère la figure suivante où un rectangle et un triangle rectangle forment un trapèze. Les dimensions sont données en centimètres en fonction de $x$.
        
        \begin{center}
            \begin{tikzpicture}[scale=0.8]
                % Rectangle
                \draw[very thick, fill=orange!5] (0,0) rectangle (4,2.5);
                \node[below] at (2,0) {$4$};
                \node[left] at (0,1.25) {$x$};
                
                % Triangle
                \draw[very thick, fill=green!5] (4,0) -- (7,0) -- (4,2.5) -- cycle;

                % Angle droit triangle
                \draw (4,0.3) -- (4.3,0.3) -- (4.3,0);

        Déterminer la valeur de $x$ pour que l'aire de la partie orange (le rectangle) soit strictement égale au \textbf{triple} de l'aire de la partie verte (le triangle).
    \end{exerciceTitre}

    \begin{exerciceTitre}{La magie des nombres consécutifs}
        On appelle « nombres entiers consécutifs » des nombres entiers qui se suivent, comme $4$ ; $5$ et $6$.
        On cherche à trouver trois entiers consécutifs dont la somme est égale à $201$.
        
        \begin{enumerate}
            \item Si on note $x$ le plus petit de ces trois nombres, exprimer les deux suivants en fonction de $x$.
            \item Écrire une équation qui traduit le problème, puis la résoudre.
            \item En déduire les trois nombres cherchés.
            \item \textit{Défi :} Est-il possible de trouver trois entiers consécutifs dont la somme est égale à $2026$ ? Justifier la réponse par une équation.
        \end{enumerate}
    \end{exerciceTitre}
    
    
    \begin{exerciceTitre}{Le choix du programme}
        On propose le programme de calcul suivant :
        \begin{center}
            \begin{tcolorbox}[colback=gray!5, colframe=gray!40, width=8cm, arc=1mm, center]
            \bfseries
            $\bullet$ Choisir un nombre entier \\
            $\bullet$ Ajouter $3$ \\
            $\bullet$ Multiplier le résultat par $4$ \\
            $\bullet$ Soustraire le double du nombre de départ
            \end{tcolorbox}
        \end{center}
        
        \begin{enumerate}
            \item Tester ce programme en choisissant le nombre $5$, puis le nombre $-2$. Que remarque-t-on ?
            \item On note $n$ le nombre entier choisi au départ. 
            \begin{enumerate}
                \item Exprimer le résultat final du programme en fonction de $n$.
                \item Développer et réduire cette expression. Quelle conjecture peut-on faire ?
            \end{enumerate}
            \item Quel nombre doit-on choisir au départ pour obtenir un résultat final égal à $42$ ? Écrire et résoudre l'équation correspondante.
        \end{enumerate}
    \end{exerciceTitre}
