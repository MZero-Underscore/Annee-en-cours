    \begin{exerciceTitre}{Le principe de la balance (Comprendre l'équilibre)}
        Imaginons une balance à plateaux en parfait équilibre. 
        \begin{itemize}
            \item Sur le plateau de \textbf{gauche} : il y a $4$ petites boîtes identiques (dont on ignore la masse) et un poids de $3$~kg.
            \item Sur le plateau de \textbf{droite} : il y a un seul poids de $15$~kg.
        \end{itemize}
        On désigne par la lettre $x$ la masse d'une petite boîte (en kg).

        \begin{center}
        \begin{tikzpicture}[scale=0.8, every node/.style={transform shape}]
            % --- Le socle (triangle central) ---
            \draw[thick, fill=gray!30] (-1,0) -- (1,0) -- (0,2.5) -- cycle;
            \draw[thick] (-2,0) -- (2,0); % Ligne de sol
            
            % --- Le fléau (ligne horizontale) ---
            \draw[ultra thick, gray!80!black] (-5,2.5) -- (5,2.5);
            \filldraw[black] (0,2.5) circle (0.15); % Point de pivot
            
            % ================= PLATEAU GAUCHE =================
            % Fils d'attache
            \draw[thick, gray] (-5,2.5) -- (-6.5,0);
            \draw[thick, gray] (-5,2.5) -- (-3.5,0);
            % Plateau
            \draw[thick, fill=gray!20] (-6.5,0) to[out=-20,in=200] (-3.5,0) -- (-3.5,-0.2) -- (-6.5,-0.2) -- cycle;
            
            % Les 4 boîtes "x"
            \draw[thick, fill=orange!30, rounded corners=1pt] (-6.1,0.2) rectangle (-5.3,1.0) node[midway] {$x$};
            \draw[thick, fill=orange!30, rounded corners=1pt] (-5.2,0.2) rectangle (-4.4,1.0) node[midway] {$x$};
            \draw[thick, fill=orange!30, rounded corners=1pt] (-6.1,1.1) rectangle (-5.3,1.9) node[midway] {$x$};
            \draw[thick, fill=orange!30, rounded corners=1pt] (-5.2,1.1) rectangle (-4.4,1.9) node[midway] {$x$};
            
            % Le poids de 3 kg
            \draw[thick, fill=teal!40, rounded corners=2pt] (-4.2,0.2) -- (-4.2,1.0) -- (-3.8,1.2) -- (-3.4,1.0) -- (-3.4,0.2) -- cycle;
            \node at (-3.8,0.6) {3 kg};
        
            % ================= PLATEAU DROIT =================
            % Fils d'attache
            \draw[thick, gray] (5,2.5) -- (3.5,0);
            \draw[thick, gray] (5,2.5) -- (6.5,0);
            % Plateau
            \draw[thick, fill=gray!20] (3.5,0) to[out=-20,in=200] (6.5,0) -- (6.5,-0.2) -- (3.5,-0.2) -- cycle;
            
            % Le grand poids de 15 kg (centré)
            \draw[thick, fill=teal!40, rounded corners=2pt] (4.3,0.2) -- (4.3,1.5) -- (5.0,2.0) -- (5.7,1.5) -- (5.7,0.2) -- cycle;
            \node at (5.0,0.9) {\textbf{15 kg}};
            
        \end{tikzpicture}
        \end{center}

        
        \begin{enumerate}
            \item Traduire cette situation d'équilibre par une égalité mathématique comportant la lettre $x$.
            \begin{ZoneReponse}{1cm}
            \end{ZoneReponse}
            \item Si l'on retire le poids de $3$~kg sur le plateau de gauche, que doit-on faire sur le plateau de droite pour que la balance reste bien en équilibre ?
            \begin{ZoneReponse}{1cm}
            \end{ZoneReponse}
            \item En déduire la masse des $4$ boîtes seules sur le plateau, puis calculer la masse $x$ d'une seule boîte.
            \begin{ZoneReponse}{3cm}
            \end{ZoneReponse}
        \end{enumerate}
    \end{exerciceTitre}

    \begin{exerciceTitre}{Tester une solution (Vérifier une égalité)}
        Pour des situations plus complexes, on ne peut pas toujours deviner la solution de tête. 
        On considère l'égalité (l'équation) suivante :
        \[ 5x - 4 = 2x + 11 \]
        Le but de cet exercice est de vérifier si un nombre donné est la solution cachée de cette équation.
        
        \begin{enumerate}
            \item \textbf{Testons pour $x = 4$ :}
            \begin{enumerate}
                \item Remplacer $x$ par la valeur $4$ dans le \textbf{membre de gauche} ($5x - 4$) et calculer son résultat.
                \item Remplacer $x$ par la valeur $4$ dans le \textbf{membre de droite} ($2x + 11$) et calculer son résultat.
                \item L'égalité est-elle vérifiée ? Le nombre $4$ est-il la solution de cette équation ?
            \end{enumerate}
            \begin{ZoneReponse}{4cm}
            \end{ZoneReponse}
            
            \vspace{0.4cm}
            
            \item \textbf{Testons pour $x = 5$ :}
            \begin{enumerate}
                \item Calculer la valeur du membre de gauche pour $x = 5$.
                \item Calculer la valeur du membre de droite pour $x = 5$.
                \item Que peut-on conclure pour le nombre $5$ ?
            \end{enumerate}
            \begin{ZoneReponse}{4cm}
            \end{ZoneReponse}
        \end{enumerate}
    \end{exerciceTitre}
