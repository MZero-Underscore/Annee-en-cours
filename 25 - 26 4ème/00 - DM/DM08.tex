    \begin{exercice}[Développer et réduire] 
        Développer les expressions suivantes en utilisant la distributivité (simple et double) :
        \begin{itemize}
            \item $D = 7(2x - 4) \textcolor{red}{= 7 \times 2x - 7 \times 4 \mathbf{= 14x - 28}}$
            \item $E = (x + 5)(x + 2) \textcolor{red}{= x^2 + 2x + 5x + 10 \mathbf{= x^2 + 7x + 10}}$
            \item $F = (3x - 1)(2x + 4) \textcolor{red}{= 6x^2 + 12x - 2x - 4 \mathbf{= 6x^2 + 10x - 4}}$
        \end{itemize}
    \end{exercice}
    
    \begin{exercice}[Évaluation et Factorisation] 
        1. Calculer la valeur de $G = 4x^2 - 5x + 3$ pour $x = 2$ puis pour $x = -2$. \\
        \textcolor{red}{
        \textbf{Pour $x = 2$ :} $G = 4 \times 2^2 - 5 \times 2 + 3 = 4 \times 4 - 10 + 3 = 16 - 10 + 3 = \mathbf{9}$. \\
        \textbf{Pour $x = -2$ :} $G = 4 \times (-2)^2 - 5 \times (-2) + 3 = 4 \times 4 + 10 + 3 = 16 + 10 + 3 = \mathbf{29}$.
        }
        
        \medskip
        2. Factoriser (trouver le facteur commun) : $H = 6x - 18$ et $I = x^2 + 5x$. \\
        \textcolor{red}{
        $H = \mathbf{6}(x) - \mathbf{6}(3) \mathbf{= 6(x - 3)}$ \\
        $I = \mathbf{x}(x) + \mathbf{x}(5) \mathbf{= x(x + 5)}$
        }
    \end{exercice}
    

    \begin{exercice}[Des figures mystérieuses]
        On considère deux figures géométriques :
        \begin{itemize}
            \item \textbf{Figure 1 :} Un rectangle de longueur $3x + 1$ et de largeur $x + 2$.
            \item \textbf{Figure 2 :} Un triangle dont les longueurs des trois côtés sont $x + 2$, $4x + 1$ et $3x + 3$.
        \end{itemize}
        \textit{L'unité de longueur est le centimètre et $x$ désigne un nombre positif.}
        
        \begin{enumerate}
            \item Exprimer le périmètre de la Figure 1 en fonction de $x$. Développer et réduire. \\
            \textcolor{red}{$P_1 = 2 \times (\text{Longueur} + \text{largeur}) = 2 \times (3x + 1 + x + 2) = 2 \times (4x + 3) \mathbf{= 8x + 6}$.}
            
            \item Exprimer le périmètre de la Figure 2 en fonction de $x$. Réduire cette expression. \\
            \textcolor{red}{$P_2 = (x + 2) + (4x + 1) + (3x + 3) = x + 4x + 3x + 2 + 1 + 3 \mathbf{= 8x + 6}$.}
            
            \item Que peut-on conclure concernant les périmètres de ces deux figures ? \\
            \textcolor{red}{On remarque que $P_1 = P_2$. \\
            Les deux figures ont \textbf{exactement le même périmètre}, quelle que soit la valeur choisie pour $x$.}
            
            \item L'affirmation suivante est-elle vraie : \og Si $x = 10,5$, alors je suis certain que les deux figures ont le même périmètre sans même avoir besoin de le calculer \fg{} ? Justifier. \\
            \textcolor{red}{\textbf{Vrai.} Le calcul littéral réalisé aux questions 1 et 2 prouve que l'égalité $P_1 = P_2$ est valable pour \textbf{toutes} les valeurs de $x$ (donc aussi pour $10,5$).}
        \end{enumerate}
    \end{exercice}
    
    
    \begin{exercice}[Data Centers et flux de données] 
        Un Data Center situé en Isère traite un flux de données de \textbf{540 To} (Téraoctets) en \textbf{12 heures}.
        
        \begin{enumerate}
            \item \textbf{Autour des données}
            \begin{enumerate}
                \item Quel volume de données est traité en 1 heure ? En une semaine complète ? \\
                \textcolor{red}{En 1 heure : $540 \div 12 = \mathbf{45 \text{ To}}$. \\
                Une semaine compte $7 \times 24 = 168$ heures. En une semaine : $45 \times 168 = \mathbf{7\,560 \text{ To}}$.}
                
                \item \textit{Pour se rendre compte :} On estime qu'une vidéo courte sur smartphone pèse environ 20 Mo. \\
                Sachant que \textbf{1 To = 1 000 000 Mo}, combien de ces vidéos représentent un volume de 1 seul To ? \\
                \textcolor{red}{On cherche combien de fois 20 Mo rentrent dans 1 000 000 Mo : \\ $1\,000\,000 \div 20 = \mathbf{50\,000}$. Il faut 50 000 vidéos pour remplir un seul Téraoctet.}
            \end{enumerate}


\newpage
            
            \item \textbf{Autour de l'énergie} \\
            La consommation électrique de ce Data Center est de \textbf{180 MWh} pour traiter ces 540 To (c'est-à-dire en 12 heures).
            \begin{enumerate}
                \item Calculer la consommation électrique totale du Data Center sur un mois de 30 jours. \\
                \textcolor{red}{En 12 heures, il consomme 180 MWh. Donc en 24 heures (1 jour), il consomme le double : $180 \times 2 = 360$ MWh. \\
                Sur 30 jours, la consommation est : $360 \times 30 = \mathbf{10\,800 \text{ MWh}}$.}
                
                \item \textit{Pour se rendre compte :} Un collège de 600 élèves consomme en moyenne \textbf{400 MWh par an}. À combien d'années de fonctionnement de ce collège correspond la consommation d'un seul mois de ce Data Center ? \\
                \textcolor{red}{On divise la consommation mensuelle du Data Center par la consommation annuelle du collège : \\
                $10\,800 \div 400 = \mathbf{27}$. \\
                La consommation d'un seul mois de ce Data Center correspond à \textbf{27 années de fonctionnement} du collège !}
            \end{enumerate}
        \end{enumerate}
    \end{exercice}
