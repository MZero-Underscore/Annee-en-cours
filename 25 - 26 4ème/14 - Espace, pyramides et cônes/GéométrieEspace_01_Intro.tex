        \textbf{1. La pyramide} \\
        \begin{minipage}{0.55\textwidth}

                \item Combien ce solide possède-t-il de \textbf{sommets} au total ? \dotfill
                \item Combien possède-t-il d'\textbf{arêtes} au total ? \dotfill

            \end{itemize}
        \end{minipage}
