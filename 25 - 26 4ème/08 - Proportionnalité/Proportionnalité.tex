\section{Tableau de proportionnalité}
\subsection{Reconnaître une situation de proportionnalité}
    
    \begin{propriete} 
        Dans une situation de proportionnalité, les quotients des nombres de la première ligne par ceux de la seconde sont tous égaux. Le nombre trouvé s'appelle un \textbf{coefficient de proportionnalité}.
    \end{propriete}

\subsection{Calcul de la quatrième proportionnelle}
    
    \begin{propriete} 
        Soit \( a, b, c \) et \( x \) quatre nombres quelconques non nuls. On considère le tableau suivant :
    
        \[
        \begin{array}{c|c}
        a & c \\ \hline
        b & x \\
        \end{array}
        \]
    
        Si ce tableau est un tableau de proportionnalité, alors \( x = \frac{b \times c}{a} \).
    \end{propriete}
       
    \begin{exemple} 
        Compléter les deux tableaux suivants.
        \begin{enumerate}

            \end{tabular}

\section{Calculer et utiliser un pourcentage}
    \begin{definition}
        Un pourcentage est l'expression d'une proportion pour un total de 100.

        Pour indiquer qu'on utilise un pourcentage, on utilise le symbole \%.
    \end{definition}
    

    \begin{remarque}
        Le symbole \% vient de la déformation au fil du temps de l'écriture fractionnaire $\frac{.}{100}.$

        On peut écrire, par exemple : $15\%=\dfrac{15}{100}=0,15$
    \end{remarque}

\subsection{Vitesse moyenne}
    
    \begin{definition} 
        La vitesse moyenne \( v \) d'un mobile qui parcourt une distance \( d \) en un temps \( t \) est donnée par la formule :
    
        \[
        v = \frac{d}{t}
        \]
    \end{definition}

\subsection{Le débit (Grandeur quotient)}
    
    \begin{definitionTitre}{Débit}
        Le \textbf{débit} $D$ est le rapport du volume $V$ par le temps $t$ : 
        \[ D = \frac{V}{t} \]
        Unités courantes : L/s, L/min, $m^3/h$.
    \end{definitionTitre}

\subsection{L'énergie électrique (Grandeur produit)}
    
    \begin{definitionTitre}{Énergie électrique}
        L'énergie électrique $E$ consommée est le produit de la puissance $P$ par la durée d'utilisation $t$ :
        \[ E = P \times t \]
        Unités courantes : le Wattheure ($Wh$) ou le Kilowattheure ($kWh$).
    \end{definitionTitre}
