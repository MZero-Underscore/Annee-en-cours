        \begin{enumerate}
            \item Récapituler ces informations dans la deuxième ligne du tableau ci-dessous.
            
            \begin{center}
                \begin{tabular}{l|c|c|c|c|c}
                    \toprule
                    \textbf{Contenance} & \textbf{0,5\,L} & \textbf{1\,L} & \textbf{2\,L} & \textbf{5\,L} & \textbf{Total} \\
                    \midrule
                    \textbf{Effectif} & & & & & \\
                    \midrule
                    \textbf{Fréquence} & & & & & 1 \\
                    \textbf{Fréquence en \%} & & & & & 100 \\
                    \bottomrule
                \end{tabular}
            \end{center}
            
            \item Combien lui reste-t-il de pots au total ?
            
            \item Complète la ligne « fréquence ».
            
            \item Complète la ligne « fréquence en \% ».
            
            \item Les pots de volume supérieur ou égal à 2\,L représentent-ils moins de 50\% du total ?
        \end{enumerate}
    \end{exercice}
