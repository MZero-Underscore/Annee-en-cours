\section*{1. Division euclidienne}

    \begin{definition}
        Diviser un entier $a$ par un entier $b$ (avec $b \neq 0$), c'est trouver deux entiers $q$ et $r$ tels que :
        \[
            a = b \times q + r \qquad \text{avec} \qquad 0 \leq r < b
        \]
        \begin{itemize}
            \item $a$ est le \textbf{dividende}, \quad $b$ est le \textbf{diviseur},
            \item $q$ est le \textbf{quotient}, \quad $r$ est le \textbf{reste}.
        \end{itemize}
        Lorsque $r = 0$, on dit que $b$ \textbf{divise} $a$ (ou que $a$ est \textbf{divisible} par $b$).
    \end{definition}

    \begin{exemple}

\section*{2. Multiples et diviseurs}

    \begin{definition}
        Soit $a$ et $b$ deux entiers avec $b \neq 0$. On dit que $b$ est un \textbf{diviseur} de $a$ (et que $a$ est un \textbf{multiple} de $b$) si la division euclidienne de $a$ par $b$ a un \textbf{reste nul}.
    \end{definition}

\section*{3. Critères de divisibilité}

    \begin{propriete}
        Quelques critères de divisibilité courants (sans effectuer la division) :

        \medskip
        \begin{center}
        \renewcommand{\arraystretch}{1.8}
        \begin{tabular}{|c|p{9cm}|}
        \hline
        \textbf{Diviseur} & \textbf{Critère} \\ \hline
        $2$  & Le chiffre des \textbf{unités} est $0, 2, 4, 6$ ou $8$. \\ \hline
        $3$  & La \textbf{somme des chiffres} est divisible par $3$. \\ \hline
        $4$  & Le nombre formé par les \textbf{deux derniers chiffres} est divisible par $4$. \\ \hline
        $5$  & Le chiffre des \textbf{unités} est $0$ ou $5$. \\ \hline
        $9$  & La \textbf{somme des chiffres} est divisible par $9$. \\ \hline
        $10$ & Le chiffre des \textbf{unités} est $0$. \\ \hline
        \end{tabular}
        \end{center}
    \end{propriete}
